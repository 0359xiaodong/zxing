54  /**
55   * This is a quick and dirty way to get totals across all the positive black box tests. It is
56   * necessary because we spawn multiple processes when using the standard test-blackbox Ant target.
57   * It would be a shame to change that because it does help with performance. Perhaps we can find a
58   * way to unify these in the future.
59   *
60   * @author dswitkin@google.com (Daniel Switkin)
61   */
62  public final class AllPositiveBlackBoxTester {
63  
64    private static final Logger log = Logger.getLogger(AllPositiveBlackBoxTester.class.getSimpleName());
65  
66    // This list has to be manually kept up to date. I don't know any automatic way to include every
67    // subclass of AbstractBlackBoxTestCase, and furthermore to exclude subclasses of
68    // AbstractNegativeBlackBoxTestCase which derives from it.
69    private static final AbstractBlackBoxTestCase[] TESTS = {
70      new DataMatrixBlackBox1TestCase(),
71      new DataMatrixBlackBox2TestCase(),
72      new Code128BlackBox1TestCase(),
73      new Code128BlackBox2TestCase(),
74      new Code128BlackBox3TestCase(),
75      new Code39BlackBox1TestCase(),
76      new Code39ExtendedBlackBox2TestCase(),
77      new Code39BlackBox3TestCase(),
78      new EAN13BlackBox1TestCase(),
79      new EAN13BlackBox2TestCase(),
80      new EAN13BlackBox3TestCase(),
81      new EAN13BlackBox4TestCase(),
82      new EAN8BlackBox1TestCase(),
83      new ITFBlackBox1TestCase(),
84      new ITFBlackBox2TestCase(),
85      new UPCABlackBox1TestCase(),
86      new UPCABlackBox2TestCase(),
87      new UPCABlackBox3ReflectiveTestCase(),
88      new UPCABlackBox4TestCase(),
89      new UPCABlackBox5TestCase(),
90      new UPCEBlackBox1TestCase(),
91      new UPCEBlackBox2TestCase(),
92      new UPCEBlackBox3ReflectiveTestCase(),
93      new PDF417BlackBox1TestCase(),
94      new PDF417BlackBox2TestCase(),
95      new QRCodeBlackBox1TestCase(),
96      new QRCodeBlackBox2TestCase(),
97      new QRCodeBlackBox3TestCase(),
98      new QRCodeBlackBox4TestCase(),
99      new QRCodeBlackBox5TestCase()
100   };
101 
102   private AllPositiveBlackBoxTester() {
103     System.setProperty("java.util.logging.SimpleFormatter.format", "%4$s: %5$s%6$s%n");
104   }
105 
106   public static void main(String[] args) throws Exception {
107     long start = System.currentTimeMillis();
108     SummaryResults results = new SummaryResults();
109 
110     for (AbstractBlackBoxTestCase test : TESTS) {
111       results.add(test.testBlackBoxCountingResults(false));
112     }
113 
114     log.info(results.toString());
115     log.info(String.format("Total time: %d ms", System.currentTimeMillis() - start));
116   }
117 }
