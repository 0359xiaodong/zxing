177   private static void assertCorrectImage2binary(String fileName, String expected)
178       throws IOException, NotFoundException {
179     Path path = Paths.get("src/test/resources/blackbox/rssexpanded-1/").resolve(fileName);
180     if (!Files.exists(path)) {
181       // Support running from project root too
182       path = Paths.get("core").resolve(path);
183     }
184 
185     BufferedImage image = ImageIO.read(path.toFile());
186     BinaryBitmap binaryMap = new BinaryBitmap(new GlobalHistogramBinarizer(new BufferedImageLuminanceSource(image)));
187     int rowNumber = binaryMap.getHeight() / 2;
188     BitArray row = binaryMap.getBlackRow(rowNumber, null);
189 
190     List<ExpandedPair> pairs;
191     try {
192       RSSExpandedReader rssExpandedReader = new RSSExpandedReader();
193       pairs = rssExpandedReader.decodeRow2pairs(rowNumber, row);
194     } catch (ReaderException re) {
195       fail(re.toString());
196       return;
197     }
198     BitArray binary = BitArrayBuilder.buildBitArray(pairs);
199     assertEquals(expected, binary.toString());
200   }
