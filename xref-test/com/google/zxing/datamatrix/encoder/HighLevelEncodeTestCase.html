264   private static String createBinaryMessage(int len) {
265     StringBuilder sb = new StringBuilder();
266     sb.append("\u00ABÃ¤Ã¶Ã¼Ã©Ã Ã¡-");
267     for (int i = 0; i < len - 9; i++) {
268       sb.append('\u00B7');
269     }
270     sb.append('\u00BB');
271     return sb.toString();
272   }

274   private static void assertStartsWith(String expected, String actual) {
275     if (!actual.startsWith(expected)) {
276       throw new ComparisonFailure(null, expected, actual.substring(0, expected.length()));
277     }
278   }
279 
280   private static void assertEndsWith(String expected, String actual) {
281     if (!actual.endsWith(expected)) {
282       throw new ComparisonFailure(null, expected, actual.substring(actual.length() - expected.length()));
283     }
284   }

368   private static String encodeHighLevel(String msg) {
369     CharSequence encoded = HighLevelEncoder.encodeHighLevel(msg);
370     //DecodeHighLevel.decode(encoded);
371     return visualize(encoded);
372   }

374   /**
375    * Convert a string of char codewords into a different string which lists each character
376    * using its decimal value.
377    *
378    * @param codewords the codewords
379    * @return the visualized codewords
380    */
381   static String visualize(CharSequence codewords) {
382     StringBuilder sb = new StringBuilder();
383     for (int i = 0; i < codewords.length(); i++) {
384       if (i > 0) {
385         sb.append(' ');
386       }
387       sb.append((int) codewords.charAt(i));
388     }
389     return sb.toString();
390   }
