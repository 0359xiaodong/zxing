17  package com.google.zxing.datamatrix.encoder;
18  
19  import org.junit.Assert;
20  import org.junit.Test;
21  
22  import java.util.regex.Pattern;
23  
24  /**
25   * Tests the DataMatrix placement algorithm.
26   */
27  public final class PlacementTestCase extends Assert {
28  
29    private static final Pattern SPACE = Pattern.compile(" ");
30  
31    @Test
32    public void testPlacement() {
33      String codewords = unvisualize("66 74 78 66 74 78 129 56 35 102 192 96 226 100 156 1 107 221"); //"AIMAIM" encoded
34      DebugPlacement placement = new DebugPlacement(codewords, 12, 12);
35      placement.place();
36      String[] expected = {
37          "011100001111",
38          "001010101000",
39          "010001010100",
40          "001010100010",
41          "000111000100",
42          "011000010100",
43          "000100001101",
44          "011000010000",
45          "001100001101",
46          "100010010111",
47          "011101011010",
48          "001011001010"};
49      String[] actual = placement.toBitFieldStringArray();
50      for (int i = 0; i < actual.length; i++) {
51        assertEquals("Row " + i, expected[i], actual[i]);
52      }
53    }
54  
55    private static String unvisualize(CharSequence visualized) {
56      StringBuilder sb = new StringBuilder();
57      for (String token : SPACE.split(visualized)) {
58        sb.append((char) Integer.parseInt(token));
59      }
60      return sb.toString();
61    }
62  
63  }
