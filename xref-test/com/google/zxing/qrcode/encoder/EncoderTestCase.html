38    @Test
39    public void testGetAlphanumericCode() {
40      // The first ten code points are numbers.
41      for (int i = 0; i < 10; ++i) {
42        assertEquals(i, Encoder.getAlphanumericCode('0' + i));
43      }
44  
45      // The next 26 code points are capital alphabet letters.
46      for (int i = 10; i < 36; ++i) {
47        assertEquals(i, Encoder.getAlphanumericCode('A' + i - 10));
48      }
49  
50      // Others are symbol letters
51      assertEquals(36, Encoder.getAlphanumericCode(' '));
52      assertEquals(37, Encoder.getAlphanumericCode('$'));
53      assertEquals(38, Encoder.getAlphanumericCode('%'));
54      assertEquals(39, Encoder.getAlphanumericCode('*'));
55      assertEquals(40, Encoder.getAlphanumericCode('+'));
56      assertEquals(41, Encoder.getAlphanumericCode('-'));
57      assertEquals(42, Encoder.getAlphanumericCode('.'));
58      assertEquals(43, Encoder.getAlphanumericCode('/'));
59      assertEquals(44, Encoder.getAlphanumericCode(':'));
60  
61      // Should return -1 for other letters;
62      assertEquals(-1, Encoder.getAlphanumericCode('a'));
63      assertEquals(-1, Encoder.getAlphanumericCode('#'));
64      assertEquals(-1, Encoder.getAlphanumericCode('\0'));
65    }

67    @Test
68    public void testChooseMode() throws WriterException {
69      // Numeric mode.
70      assertSame(Mode.NUMERIC, Encoder.chooseMode("0"));
71      assertSame(Mode.NUMERIC, Encoder.chooseMode("0123456789"));
72      // Alphanumeric mode.
73      assertSame(Mode.ALPHANUMERIC, Encoder.chooseMode("A"));
74      assertSame(Mode.ALPHANUMERIC,
75                 Encoder.chooseMode("0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ $%*+-./:"));
76      // 8-bit byte mode.
77      assertSame(Mode.BYTE, Encoder.chooseMode("a"));
78      assertSame(Mode.BYTE, Encoder.chooseMode("#"));
79      assertSame(Mode.BYTE, Encoder.chooseMode(""));
80      // Kanji mode.  We used to use MODE_KANJI for these, but we stopped
81      // doing that as we cannot distinguish Shift_JIS from other encodings
82      // from data bytes alone.  See also comments in qrcode_encoder.h.
83  
84      // AIUE in Hiragana in Shift_JIS
85      assertSame(Mode.BYTE,
86                 Encoder.chooseMode(shiftJISString(new byte[]{0x8, 0xa, 0x8, 0xa, 0x8, 0xa, 0x8, (byte) 0xa6})));
87  
88      // Nihon in Kanji in Shift_JIS.
89      assertSame(Mode.BYTE, Encoder.chooseMode(shiftJISString(new byte[]{0x9, 0xf, 0x9, 0x7b})));
90  
91      // Sou-Utsu-Byou in Kanji in Shift_JIS.
92      assertSame(Mode.BYTE, Encoder.chooseMode(shiftJISString(new byte[]{0xe, 0x4, 0x9, 0x5, 0x9, 0x61})));
93    }

202   @Test
203   public void testAppendBytes() throws WriterException {
204     // Should use appendNumericBytes.
205     // 1 = 01 = 0001 in 4 bits.
206     BitArray bits = new BitArray();
207     Encoder.appendBytes("1", Mode.NUMERIC, bits, Encoder.DEFAULT_BYTE_MODE_ENCODING);
208     assertEquals(" ...X" , bits.toString());
209     // Should use appendAlphanumericBytes.
210     // A = 10 = 0xa = 001010 in 6 bits
211     bits = new BitArray();
212     Encoder.appendBytes("A", Mode.ALPHANUMERIC, bits, Encoder.DEFAULT_BYTE_MODE_ENCODING);
213     assertEquals(" ..X.X." , bits.toString());
214     // Lower letters such as 'a' cannot be encoded in MODE_ALPHANUMERIC.
215     try {
216       Encoder.appendBytes("a", Mode.ALPHANUMERIC, bits, Encoder.DEFAULT_BYTE_MODE_ENCODING);
217     } catch (WriterException we) {
218       // good
219     }
220     // Should use append8BitBytes.
221     // 0x61, 0x62, 0x63
222     bits = new BitArray();
223     Encoder.appendBytes("abc", Mode.BYTE, bits, Encoder.DEFAULT_BYTE_MODE_ENCODING);
224     assertEquals(" .XX....X .XX...X. .XX...XX", bits.toString());
225     // Anything can be encoded in QRCode.MODE_8BIT_BYTE.
226     Encoder.appendBytes("\0", Mode.BYTE, bits, Encoder.DEFAULT_BYTE_MODE_ENCODING);
227     // Should use appendKanjiBytes.
228     // 0x93, 0x5f
229     bits = new BitArray();
230     Encoder.appendBytes(shiftJISString(new byte[] {(byte)0x93,0x5f}), Mode.KANJI, bits, Encoder.DEFAULT_BYTE_MODE_ENCODING);
231     assertEquals(" .XX.XX.. XXXXX", bits.toString());
232   }

469   @Test
470   public void testBugInBitVectorNumBytes() throws WriterException {
471     // There was a bug in BitVector.sizeInBytes() that caused it to return a
472     // smaller-by-one value (ex. 1465 instead of 1466) if the number of bits
473     // in the vector is not 8-bit aligned.  In QRCodeEncoder::InitQRCode(),
474     // BitVector::sizeInBytes() is used for finding the smallest QR Code
475     // version that can fit the given data.  Hence there were corner cases
476     // where we chose a wrong QR Code version that cannot fit the given
477     // data.  Note that the issue did not occur with MODE_8BIT_BYTE, as the
478     // bits in the bit vector are always 8-bit aligned.
479     //
480     // Before the bug was fixed, the following test didn't pass, because:
481     //
482     // - MODE_NUMERIC is chosen as all bytes in the data are '0'
483     // - The 3518-byte numeric data needs 1466 bytes
484     //   - 3518 / 3 * 10 + 7 = 11727 bits = 1465.875 bytes
485     //   - 3 numeric bytes are encoded in 10 bits, hence the first
486     //     3516 bytes are encoded in 3516 / 3 * 10 = 11720 bits.
487     //   - 2 numeric bytes can be encoded in 7 bits, hence the last
488     //     2 bytes are encoded in 7 bits.
489     // - The version 27 QR Code with the EC level L has 1468 bytes for data.
490     //   - 1828 - 360 = 1468
491     // - In InitQRCode(), 3 bytes are reserved for a header.  Hence 1465 bytes
492     //   (1468 -3) are left for data.
493     // - Because of the bug in BitVector::sizeInBytes(), InitQRCode() determines
494     //   the given data can fit in 1465 bytes, despite it needs 1466 bytes.
495     // - Hence QRCodeEncoder.encode() failed and returned false.
496     //   - To be precise, it needs 11727 + 4 (getMode info) + 14 (length info) =
497     //     11745 bits = 1468.125 bytes are needed (i.e. cannot fit in 1468
498     //     bytes).
499     StringBuilder builder = new StringBuilder(3518);
500     for (int x = 0; x < 3518; x++) {
501       builder.append('0');
502     }
503     Encoder.encode(builder.toString(), ErrorCorrectionLevel.L);
504   }
