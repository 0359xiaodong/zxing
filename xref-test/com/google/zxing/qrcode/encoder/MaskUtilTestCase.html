28    @Test
29    public void testApplyMaskPenaltyRule1() {
30      ByteMatrix matrix = new ByteMatrix(4, 1);
31      matrix.set(0, 0, 0);
32      matrix.set(1, 0, 0);
33      matrix.set(2, 0, 0);
34      matrix.set(3, 0, 0);
35      assertEquals(0, MaskUtil.applyMaskPenaltyRule1(matrix));
36      // Horizontal.
37      matrix = new ByteMatrix(6, 1);
38      matrix.set(0, 0, 0);
39      matrix.set(1, 0, 0);
40      matrix.set(2, 0, 0);
41      matrix.set(3, 0, 0);
42      matrix.set(4, 0, 0);
43      matrix.set(5, 0, 1);
44      assertEquals(3, MaskUtil.applyMaskPenaltyRule1(matrix));
45      matrix.set(5, 0, 0);
46      assertEquals(4, MaskUtil.applyMaskPenaltyRule1(matrix));
47      // Vertical.
48      matrix = new ByteMatrix(1, 6);
49      matrix.set(0, 0, 0);
50      matrix.set(0, 1, 0);
51      matrix.set(0, 2, 0);
52      matrix.set(0, 3, 0);
53      matrix.set(0, 4, 0);
54      matrix.set(0, 5, 1);
55      assertEquals(3, MaskUtil.applyMaskPenaltyRule1(matrix));
56      matrix.set(0, 5, 0);
57      assertEquals(4, MaskUtil.applyMaskPenaltyRule1(matrix));
58    }

60    @Test
61    public void testApplyMaskPenaltyRule2() {
62      ByteMatrix matrix = new ByteMatrix(1, 1);
63      matrix.set(0, 0, 0);
64      assertEquals(0, MaskUtil.applyMaskPenaltyRule2(matrix));
65      matrix = new ByteMatrix(2, 2);
66      matrix.set(0, 0, 0);
67      matrix.set(1, 0, 0);
68      matrix.set(0, 1, 0);
69      matrix.set(1, 1, 1);
70      assertEquals(0, MaskUtil.applyMaskPenaltyRule2(matrix));
71      matrix = new ByteMatrix(2, 2);
72      matrix.set(0, 0, 0);
73      matrix.set(1, 0, 0);
74      matrix.set(0, 1, 0);
75      matrix.set(1, 1, 0);
76      assertEquals(3, MaskUtil.applyMaskPenaltyRule2(matrix));
77      matrix = new ByteMatrix(3, 3);
78      matrix.set(0, 0, 0);
79      matrix.set(1, 0, 0);
80      matrix.set(2, 0, 0);
81      matrix.set(0, 1, 0);
82      matrix.set(1, 1, 0);
83      matrix.set(2, 1, 0);
84      matrix.set(0, 2, 0);
85      matrix.set(1, 2, 0);
86      matrix.set(2, 2, 0);
87      // Four instances of 2x2 blocks.
88      assertEquals(3 * 4, MaskUtil.applyMaskPenaltyRule2(matrix));

151   @Test
152   public void testApplyMaskPenaltyRule4() {
153     // Dark cell ratio = 0%
154     ByteMatrix matrix = new ByteMatrix(1, 1);
155     matrix.set(0, 0, 0);
156     assertEquals(100, MaskUtil.applyMaskPenaltyRule4(matrix));
157     // Dark cell ratio = 5%
158     matrix = new ByteMatrix(2, 1);
159     matrix.set(0, 0, 0);
160     matrix.set(0, 0, 1);
161     assertEquals(0, MaskUtil.applyMaskPenaltyRule4(matrix));
162     // Dark cell ratio = 66.67%
163     matrix = new ByteMatrix(6, 1);
164     matrix.set(0, 0, 0);
165     matrix.set(1, 0, 1);
166     matrix.set(2, 0, 1);
167     matrix.set(3, 0, 1);
168     matrix.set(4, 0, 1);
169     matrix.set(5, 0, 0);
170     assertEquals(30, MaskUtil.applyMaskPenaltyRule4(matrix));
171   }

173   private static boolean TestGetDataMaskBitInternal(int maskPattern, int[][] expected) {
174     for (int x = 0; x < 6; ++x) {
175       for (int y = 0; y < 6; ++y) {
176         if ((expected[y][x] == 1) != MaskUtil.getDataMaskBit(maskPattern, x, y)) {
177           return false;
178         }
179       }
180     }
181     return true;
182   }
