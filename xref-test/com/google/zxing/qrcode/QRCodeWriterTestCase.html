45    private static BufferedImage loadImage(String fileName) throws IOException {
46      Path file = BASE_IMAGE_PATH.resolve(fileName);
47      if (!Files.exists(file)) {
48        // try starting with 'core' since the test base is often given as the project root
49        file = Paths.get("core/").resolve(BASE_IMAGE_PATH).resolve(fileName);
50      }
51      assertTrue("Please download and install test images, and run from the 'core' directory", Files.exists(file));
52      return ImageIO.read(file.toFile());
53    }

55    // In case the golden images are not monochromatic, convert the RGB values to greyscale.
56    private static BitMatrix createMatrixFromImage(BufferedImage image) {
57      int width = image.getWidth();
58      int height = image.getHeight();
59      int[] pixels = new int[width * height];
60      image.getRGB(0, 0, width, height, pixels, 0, width);
61  
62      BitMatrix matrix = new BitMatrix(width, height);
63      for (int y = 0; y < height; y++) {
64        for (int x = 0; x < width; x++) {
65          int pixel = pixels[y * width + x];
66          int luminance = (306 * ((pixel >> 16) & 0xFF) +
67              601 * ((pixel >> 8) & 0xFF) +
68              117 * (pixel & 0xFF)) >> 10;
69          if (luminance <= 0x7F) {
70            matrix.set(x, y);
71          }
72        }
73      }
74      return matrix;
75    }

77    @Test
78    public void testQRCodeWriter() throws WriterException {
79      // The QR should be multiplied up to fit, with extra padding if necessary
80      int bigEnough = 256;
81      Writer writer = new QRCodeWriter();
82      BitMatrix matrix = writer.encode("http://www.google.com/", BarcodeFormat.QR_CODE, bigEnough,
83          bigEnough, null);
84      assertNotNull(matrix);
85      assertEquals(bigEnough, matrix.getWidth());
86      assertEquals(bigEnough, matrix.getHeight());
87  
88      // The QR will not fit in this size, so the matrix should come back bigger
89      int tooSmall = 20;
90      matrix = writer.encode("http://www.google.com/", BarcodeFormat.QR_CODE, tooSmall,
91          tooSmall, null);
92      assertNotNull(matrix);
93      assertTrue(tooSmall < matrix.getWidth());
94      assertTrue(tooSmall < matrix.getHeight());
95  
96      // We should also be able to handle non-square requests by padding them
97      int strangeWidth = 500;
98      int strangeHeight = 100;
99      matrix = writer.encode("http://www.google.com/", BarcodeFormat.QR_CODE, strangeWidth,
100         strangeHeight, null);
101     assertNotNull(matrix);
102     assertEquals(strangeWidth, matrix.getWidth());
103     assertEquals(strangeHeight, matrix.getHeight());
104   }

106   private static void compareToGoldenFile(String contents,
107                                           ErrorCorrectionLevel ecLevel,
108                                           int resolution,
109                                           String fileName) throws WriterException, IOException {
110 
111     BufferedImage image = loadImage(fileName);
112     assertNotNull(image);
113     BitMatrix goldenResult = createMatrixFromImage(image);
114     assertNotNull(goldenResult);
115 
116     Map<EncodeHintType,Object> hints = new EnumMap<>(EncodeHintType.class);
117     hints.put(EncodeHintType.ERROR_CORRECTION, ecLevel);
118     Writer writer = new QRCodeWriter();
119     BitMatrix generatedResult = writer.encode(contents, BarcodeFormat.QR_CODE, resolution,
120         resolution, hints);
121 
122     assertEquals(resolution, generatedResult.getWidth());
123     assertEquals(resolution, generatedResult.getHeight());
124     assertEquals(goldenResult, generatedResult);
125   }

127   // Golden images are generated with "qrcode_sample.cc". The images are checked with both eye balls
128   // and cell phones. We expect pixel-perfect results, because the error correction level is known,
129   // and the pixel dimensions matches exactly. 
130   @Test
131   public void testRegressionTest() throws Exception {
132     compareToGoldenFile("http://www.google.com/", ErrorCorrectionLevel.M, 99,
133         "renderer-test-01.png");
