17  package com.google.zxing.pdf417.decoder.ec;
18  
19  import org.junit.Assert;
20  
21  import java.security.SecureRandom;
22  import java.util.BitSet;
23  import java.util.Random;
24  
25  /**
26   * @author Sean Owen
27   */
28  abstract class AbstractErrorCorrectionTestCase extends Assert {
29  
30    static void corrupt(int[] received, int howMany, Random random) {
31      BitSet corrupted = new BitSet(received.length);
32      for (int j = 0; j < howMany; j++) {
33        int location = random.nextInt(received.length);
34        if (corrupted.get(location)) {
35          j--;
36        } else {
37          corrupted.set(location);
38          received[location] = random.nextInt(929);
39        }
40      }
41    }
42  
43    static int[] erase(int[] received, int howMany, Random random) {
44      BitSet erased = new BitSet(received.length);
45      int[] erasures = new int[howMany];
46      int erasureOffset = 0;
47      for (int j = 0; j < howMany; j++) {
48        int location = random.nextInt(received.length);
49        if (erased.get(location)) {
50          j--;
51        } else {
52          erased.set(location);
53          received[location] = 0;
54          erasures[erasureOffset++] = location;
55        }
56      }
57      return erasures;
58    }
59  
60    static Random getRandom() {
61      return new SecureRandom(new byte[] {(byte) 0xDE, (byte) 0xAD, (byte) 0xBE, (byte) 0xEF});
62    }
63  
64  }
