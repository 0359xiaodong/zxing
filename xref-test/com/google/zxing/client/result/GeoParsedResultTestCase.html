17  package com.google.zxing.client.result;
18  
19  import com.google.zxing.BarcodeFormat;
20  import com.google.zxing.Result;
21  import org.junit.Assert;
22  import org.junit.Test;
23  
24  /**
25   * Tests {@link com.google.zxing.client.result.GeoParsedResult}.
26   *
27   * @author Sean Owen
28   */
29  public final class GeoParsedResultTestCase extends Assert {
30  
31    private static final double EPSILON = 0.0000000001;
32  
33    @Test
34    public void testGeo() {
35      doTest("geo:1,2", 1.0, 2.0, 0.0, null);
36      doTest("geo:80.33,-32.3344,3.35", 80.33, -32.3344, 3.35, null);
37      doTest("geo:-20.33,132.3344,0.01", -20.33, 132.3344, 0.01, null);
38      doTest("geo:-20.33,132.3344,0.01?q=foobar", -20.33, 132.3344, 0.01, "q=foobar");
39      doTest("GEO:-20.33,132.3344,0.01?q=foobar", -20.33, 132.3344, 0.01, "q=foobar");
40    }
41  
42    private static void doTest(String contents,
43                               double latitude,
44                               double longitude,
45                               double altitude,
46                               String query) {
47      Result fakeResult = new Result(contents, null, null, BarcodeFormat.QR_CODE);
48      ParsedResult result = ResultParser.parseResult(fakeResult);
49      assertSame(ParsedResultType.GEO, result.getType());
50      GeoParsedResult geoResult = (GeoParsedResult) result;
51      assertEquals(latitude, geoResult.getLatitude(), EPSILON);
52      assertEquals(longitude, geoResult.getLongitude(), EPSILON);
53      assertEquals(altitude, geoResult.getAltitude(), EPSILON);
54      assertEquals(query, geoResult.getQuery());
55    }
56  
57  }
