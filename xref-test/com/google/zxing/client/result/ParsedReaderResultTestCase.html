327   private static void doTestResult(String contents,

333   private static void doTestResult(String contents,
334                                    String goldenResult,
335                                    ParsedResultType type,
336                                    BarcodeFormat format) {
337     Result fakeResult = new Result(contents, null, null, format);
338     ParsedResult result = ResultParser.parseResult(fakeResult);
339     assertNotNull(result);
340     assertSame(type, result.getType());
341 
342     String displayResult = result.getDisplayResult();
343     assertEquals(goldenResult, displayResult);
344   }
