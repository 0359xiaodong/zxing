38  /**
39   * This abstract class looks for negative results, i.e. it only allows a certain number of false
40   * positives in images which should not decode. This helps ensure that we are not too lenient.
41   *
42   * @author dswitkin@google.com (Daniel Switkin)
43   */
44  public abstract class AbstractNegativeBlackBoxTestCase extends AbstractBlackBoxTestCase {
45  
46    private static final Logger log = Logger.getLogger(AbstractNegativeBlackBoxTestCase.class.getSimpleName());
47  
48    private static final class TestResult {
49      private final int falsePositivesAllowed;
50      private final float rotation;
51  
52      TestResult(int falsePositivesAllowed, float rotation) {
53        this.falsePositivesAllowed = falsePositivesAllowed;
54        this.rotation = rotation;
55      }
56  
57      public int getFalsePositivesAllowed() {
58        return falsePositivesAllowed;
59      }
60  
61      public float getRotation() {
62        return rotation;
63      }
64    }
65  
66    private final List<TestResult> testResults;
67  
68    // Use the multiformat reader to evaluate all decoders in the system.
69    protected AbstractNegativeBlackBoxTestCase(String testBasePathSuffix) {
70      super(testBasePathSuffix, new MultiFormatReader(), null);
71      testResults = new ArrayList<>();
72    }
73  
74    protected final void addTest(int falsePositivesAllowed, float rotation) {
75      testResults.add(new TestResult(falsePositivesAllowed, rotation));
76    }
77  
78    @Override
79    @Test
80    public void testBlackBox() throws IOException {
81      assertFalse(testResults.isEmpty());
82  
83      List<Path> imageFiles = getImageFiles();
84      int[] falsePositives = new int[testResults.size()];
85      for (Path testImage : imageFiles) {
86        log.info(String.format("Starting %s", testImage));
87        BufferedImage image = ImageIO.read(testImage.toFile());
88        if (image == null) {
89          throw new IOException("Could not read image: " + testImage);
90        }
91        for (int x = 0; x < testResults.size(); x++) {
92          TestResult testResult = testResults.get(x);
93          if (!checkForFalsePositives(image, testResult.getRotation())) {
94            falsePositives[x]++;
95          }
96        }
97      }
98  
99      int totalFalsePositives = 0;
100     int totalAllowed = 0;
101 
102     for (int x = 0; x < testResults.size(); x++) {
103       TestResult testResult = testResults.get(x);
104       totalFalsePositives += falsePositives[x];
105       totalAllowed += testResult.getFalsePositivesAllowed();
106     }
107 
108     if (totalFalsePositives < totalAllowed) {
109       log.warning(String.format("+++ Test too lax by %d images", totalAllowed - totalFalsePositives));
110     } else if (totalFalsePositives > totalAllowed) {
111       log.warning(String.format("--- Test failed by %d images", totalFalsePositives - totalAllowed));
112     }
113 
114     for (int x = 0; x < testResults.size(); x++) {
115       TestResult testResult = testResults.get(x);
116       log.info(String.format("Rotation %d degrees: %d of %d images were false positives (%d allowed)",
117                              (int) testResult.getRotation(), falsePositives[x], imageFiles.size(),
118                              testResult.getFalsePositivesAllowed()));
119       assertTrue("Rotation " + testResult.getRotation() + " degrees: Too many false positives found",
120                  falsePositives[x] <= testResult.getFalsePositivesAllowed());
121     }
122   }
123 
124   /**
125    * Make sure ZXing does NOT find a barcode in the image.
126    *
127    * @param image The image to test
128    * @param rotationInDegrees The amount of rotation to apply
129    * @return true if nothing found, false if a non-existant barcode was detected
130    */
131   private boolean checkForFalsePositives(BufferedImage image, float rotationInDegrees) {
132     BufferedImage rotatedImage = rotateImage(image, rotationInDegrees);
133     LuminanceSource source = new BufferedImageLuminanceSource(rotatedImage);
134     BinaryBitmap bitmap = new BinaryBitmap(new HybridBinarizer(source));
135     Result result;
136     try {
137       result = getReader().decode(bitmap);
138       log.info(String.format("Found false positive: '%s' with format '%s' (rotation: %d)",
139                               result.getText(), result.getBarcodeFormat(), (int) rotationInDegrees));
140       return false;
141     } catch (ReaderException re) {
142       // continue
143     }
144 
145     // Try "try harder" getMode
146     Map<DecodeHintType,Object> hints = new EnumMap<>(DecodeHintType.class);
147     hints.put(DecodeHintType.TRY_HARDER, Boolean.TRUE);
148     try {
149       result = getReader().decode(bitmap, hints);
150       log.info(String.format("Try harder found false positive: '%s' with format '%s' (rotation: %d)",
151                               result.getText(), result.getBarcodeFormat(), (int) rotationInDegrees));
152       return false;
153     } catch (ReaderException re) {
154       // continue
155     }
156     return true;
157   }
