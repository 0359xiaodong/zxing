30  /**
31   * <p>Encapsulates logic that can detect a PDF417 Code in an image, even if the
32   * PDF417 Code is rotated or skewed, or partially obscured.</p>
33   *
34   * @author SITA Lab (kevin.osullivan@sita.aero)
35   * @author dswitkin@google.com (Daniel Switkin)
36   * @author Guenther Grau
37   */
38  public final class Detector {
39  
40    private static final int[] INDEXES_START_PATTERN = {0, 4, 1, 5};
41    private static final int[] INDEXES_STOP_PATTERN = {6, 2, 7, 3};
42    private static final float MAX_AVG_VARIANCE = 0.42f;
43    private static final float MAX_INDIVIDUAL_VARIANCE = 0.8f;
44  
45    // B S B S B S B S Bar/Space pattern
46    // 11111111 0 1 0 1 0 1 000
47    private static final int[] START_PATTERN = {8, 1, 1, 1, 1, 1, 1, 3};
48    // 1111111 0 1 000 1 0 1 00 1
49    private static final int[] STOP_PATTERN = {7, 1, 1, 3, 1, 1, 1, 2, 1};
50    private static final int MAX_PIXEL_DRIFT = 3;
51    private static final int MAX_PATTERN_DRIFT = 5;
52    // if we set the value too low, then we don't detect the correct height of the bar if the start patterns are damaged.
53    // if we set the value too high, then we might detect the start pattern from a neighbor barcode.
54    private static final int SKIPPED_ROW_COUNT_MAX = 25;
55    // A PDF471 barcode should have at least 3 rows, with each row being >= 3 times the module width. Therefore it should be at least
56    // 9 pixels tall. To be conservative, we use about half the size to ensure we don't miss it.
57    private static final int ROW_STEP = 5;
58    private static final int BARCODE_MIN_HEIGHT = 10;
59  
60    private Detector() {
61    }
62  
63    /**
64     * <p>Detects a PDF417 Code in an image. Only checks 0 and 180 degree rotations.</p>
65     *
66     * @param image barcode image to decode
67     * @param hints optional hints to detector
68     * @param multiple if true, then the image is searched for multiple codes. If false, then at most one code will
69     * be found and returned
70     * @return {@link PDF417DetectorResult} encapsulating results of detecting a PDF417 code
71     * @throws NotFoundException if no PDF417 Code can be found
72     */
73    public static PDF417DetectorResult detect(BinaryBitmap image, Map<DecodeHintType,?> hints, boolean multiple)
74        throws NotFoundException {
75      // TODO detection improvement, tryHarder could try several different luminance thresholds/blackpoints or even 
76      // different binarizers
77      //boolean tryHarder = hints != null && hints.containsKey(DecodeHintType.TRY_HARDER);
78  
79      BitMatrix bitMatrix = image.getBlackMatrix();
80  
81      List<ResultPoint[]> barcodeCoordinates = detect(multiple, bitMatrix);
82      if (barcodeCoordinates.isEmpty()) {
83        bitMatrix = bitMatrix.clone();
84        bitMatrix.rotate180();
85        barcodeCoordinates = detect(multiple, bitMatrix);
86      }
87      return new PDF417DetectorResult(bitMatrix, barcodeCoordinates);
88    }

90    /**
91     * Detects PDF417 codes in an image. Only checks 0 degree rotation
92     * @param multiple if true, then the image is searched for multiple codes. If false, then at most one code will
93     * be found and returned
94     * @param bitMatrix bit matrix to detect barcodes in
95     * @return List of ResultPoint arrays containing the coordinates of found barcodes
96     */
97    private static List<ResultPoint[]> detect(boolean multiple, BitMatrix bitMatrix) {
98      List<ResultPoint[]> barcodeCoordinates = new ArrayList<>();
99      int row = 0;
100     int column = 0;
101     boolean foundBarcodeInRow = false;
102     while (row < bitMatrix.getHeight()) {
103       ResultPoint[] vertices = findVertices(bitMatrix, row, column);
104 
105       if (vertices[0] == null && vertices[3] == null) {
106         if (!foundBarcodeInRow) {
107           // we didn't find any barcode so that's the end of searching 
108           break;
109         }
110         // we didn't find a barcode starting at the given column and row. Try again from the first column and slightly
111         // below the lowest barcode we found so far.
112         foundBarcodeInRow = false;
113         column = 0;
114         for (ResultPoint[] barcodeCoordinate : barcodeCoordinates) {
115           if (barcodeCoordinate[1] != null) {
116             row = (int) Math.max(row, barcodeCoordinate[1].getY());
117           }
118           if (barcodeCoordinate[3] != null) {
119             row = Math.max(row, (int) barcodeCoordinate[3].getY());
120           }
121         }
122         row += ROW_STEP;
123         continue;
124       }
125       foundBarcodeInRow = true;
126       barcodeCoordinates.add(vertices);
127       if (!multiple) {
128         break;
129       }
130       // if we didn't find a right row indicator column, then continue the search for the next barcode after the 
131       // start pattern of the barcode just found.
132       if (vertices[2] != null) {
133         column = (int) vertices[2].getX();
134         row = (int) vertices[2].getY();
135       } else {
136         column = (int) vertices[4].getX();
137         row = (int) vertices[4].getY();
138       }
139     }
140     return barcodeCoordinates;
141   }

143   /**
144    * Locate the vertices and the codewords area of a black blob using the Start
145    * and Stop patterns as locators.
146    *
147    * @param matrix the scanned barcode image.
148    * @return an array containing the vertices:
149    *           vertices[0] x, y top left barcode
150    *           vertices[1] x, y bottom left barcode
151    *           vertices[2] x, y top right barcode
152    *           vertices[3] x, y bottom right barcode
153    *           vertices[4] x, y top left codeword area
154    *           vertices[5] x, y bottom left codeword area
155    *           vertices[6] x, y top right codeword area
156    *           vertices[7] x, y bottom right codeword area
157    */
158   private static ResultPoint[] findVertices(BitMatrix matrix, int startRow, int startColumn) {
159     int height = matrix.getHeight();
160     int width = matrix.getWidth();
161 
162     ResultPoint[] result = new ResultPoint[8];
163     copyToResult(result, findRowsWithPattern(matrix, height, width, startRow, startColumn, START_PATTERN),
164         INDEXES_START_PATTERN);
165 
166     if (result[4] != null) {
167       startColumn = (int) result[4].getX();
168       startRow = (int) result[4].getY();
169     }
170     copyToResult(result, findRowsWithPattern(matrix, height, width, startRow, startColumn, STOP_PATTERN),
171         INDEXES_STOP_PATTERN);
172     return result;
173   }
174 
175   private static void copyToResult(ResultPoint[] result, ResultPoint[] tmpResult, int[] destinationIndexes) {
176     for (int i = 0; i < destinationIndexes.length; i++) {
177       result[destinationIndexes[i]] = tmpResult[i];
178     }
179   }

181   private static ResultPoint[] findRowsWithPattern(BitMatrix matrix,
182                                                    int height,
183                                                    int width,
184                                                    int startRow,
185                                                    int startColumn,
186                                                    int[] pattern) {
187     ResultPoint[] result = new ResultPoint[4];
188     boolean found = false;
189     int[] counters = new int[pattern.length];
190     for (; startRow < height; startRow += ROW_STEP) {
191       int[] loc = findGuardPattern(matrix, startColumn, startRow, width, false, pattern, counters);
192       if (loc != null) {
193         while (startRow > 0) {
194           int[] previousRowLoc = findGuardPattern(matrix, startColumn, --startRow, width, false, pattern, counters);
195           if (previousRowLoc != null) {
196             loc = previousRowLoc;
197           } else {
198             startRow++;
199             break;
200           }
201         }
202         result[0] = new ResultPoint(loc[0], startRow);
203         result[1] = new ResultPoint(loc[1], startRow);
204         found = true;
205         break;
206       }
207     }
208     int stopRow = startRow + 1;
209     // Last row of the current symbol that contains pattern
210     if (found) {
211       int skippedRowCount = 0;
212       int[] previousRowLoc = {(int) result[0].getX(), (int) result[1].getX()};
213       for (; stopRow < height; stopRow++) {
214         int[] loc = findGuardPattern(matrix, previousRowLoc[0], stopRow, width, false, pattern, counters);
215         // a found pattern is only considered to belong to the same barcode if the start and end positions
216         // don't differ too much. Pattern drift should be not bigger than two for consecutive rows. With
217         // a higher number of skipped rows drift could be larger. To keep it simple for now, we allow a slightly
218         // larger drift and don't check for skipped rows.
219         if (loc != null &&
220             Math.abs(previousRowLoc[0] - loc[0]) < MAX_PATTERN_DRIFT &&
221             Math.abs(previousRowLoc[1] - loc[1]) < MAX_PATTERN_DRIFT) {
222           previousRowLoc = loc;
223           skippedRowCount = 0;
224         } else {
225           if (skippedRowCount > SKIPPED_ROW_COUNT_MAX) {
226             break;
227           } else {
228             skippedRowCount++;
229           }
230         }
231       }
232       stopRow -= skippedRowCount + 1;
233       result[2] = new ResultPoint(previousRowLoc[0], stopRow);
234       result[3] = new ResultPoint(previousRowLoc[1], stopRow);
235     }
236     if (stopRow - startRow < BARCODE_MIN_HEIGHT) {
237       for (int i = 0; i < result.length; i++) {
238         result[i] = null;
239       }
240     }
241     return result;
242   }

244   /**
245    * @param matrix row of black/white values to search
246    * @param column x position to start search
247    * @param row y position to start search
248    * @param width the number of pixels to search on this row
249    * @param pattern pattern of counts of number of black and white pixels that are
250    *                 being searched for as a pattern
251    * @param counters array of counters, as long as pattern, to re-use 
252    * @return start/end horizontal offset of guard pattern, as an array of two ints.
253    */
254   private static int[] findGuardPattern(BitMatrix matrix,
255                                         int column,
256                                         int row,
257                                         int width,
258                                         boolean whiteFirst,
259                                         int[] pattern,
260                                         int[] counters) {
261     Arrays.fill(counters, 0, counters.length, 0);
262     int patternLength = pattern.length;
263     boolean isWhite = whiteFirst;
264     int patternStart = column;
265     int pixelDrift = 0;
266 
267     // if there are black pixels left of the current pixel shift to the left, but only for MAX_PIXEL_DRIFT pixels 
268     while (matrix.get(patternStart, row) && patternStart > 0 && pixelDrift++ < MAX_PIXEL_DRIFT) {
269       patternStart--;
270     }
271     int x = patternStart;
272     int counterPosition = 0;
273     for (; x < width; x++) {
274       boolean pixel = matrix.get(x, row);
275       if (pixel ^ isWhite) {
276         counters[counterPosition]++;
277       } else {
278         if (counterPosition == patternLength - 1) {
279           if (patternMatchVariance(counters, pattern, MAX_INDIVIDUAL_VARIANCE) < MAX_AVG_VARIANCE) {
280             return new int[] {patternStart, x};
281           }
282           patternStart += counters[0] + counters[1];
283           System.arraycopy(counters, 2, counters, 0, patternLength - 2);
284           counters[patternLength - 2] = 0;
285           counters[patternLength - 1] = 0;
286           counterPosition--;
287         } else {
288           counterPosition++;
289         }
290         counters[counterPosition] = 1;
291         isWhite = !isWhite;
292       }
293     }
294     if (counterPosition == patternLength - 1) {
295       if (patternMatchVariance(counters, pattern, MAX_INDIVIDUAL_VARIANCE) < MAX_AVG_VARIANCE) {
296         return new int[] {patternStart, x - 1};
297       }
298     }
299     return null;
300   }

302   /**
303    * Determines how closely a set of observed counts of runs of black/white
304    * values matches a given target pattern. This is reported as the ratio of
305    * the total variance from the expected pattern proportions across all
306    * pattern elements, to the length of the pattern.
307    *
308    * @param counters observed counters
309    * @param pattern expected pattern
310    * @param maxIndividualVariance The most any counter can differ before we give up
311    * @return ratio of total variance between counters and pattern compared to total pattern size
312    */
313   private static float patternMatchVariance(int[] counters, int[] pattern, float maxIndividualVariance) {
314     int numCounters = counters.length;
315     int total = 0;
316     int patternLength = 0;
317     for (int i = 0; i < numCounters; i++) {
318       total += counters[i];
319       patternLength += pattern[i];
320     }
321     if (total < patternLength) {
322       // If we don't even have one pixel per unit of bar width, assume this
323       // is too small to reliably match, so fail:
324       return Float.POSITIVE_INFINITY;
325     }
326     // We're going to fake floating-point math in integers. We just need to use more bits.
327     // Scale up patternLength so that intermediate values below like scaledCounter will have
328     // more "significant digits".
329     float unitBarWidth = (float) total / patternLength;
330     maxIndividualVariance *= unitBarWidth;
331 
332     float totalVariance = 0.0f;
333     for (int x = 0; x < numCounters; x++) {
334       int counter = counters[x];
335       float scaledPattern = pattern[x] * unitBarWidth;
336       float variance = counter > scaledPattern ? counter - scaledPattern : scaledPattern - counter;
337       if (variance > maxIndividualVariance) {
338         return Float.POSITIVE_INFINITY;
339       }
340       totalVariance += variance;
341     }
342     return totalVariance / total;
343   }
