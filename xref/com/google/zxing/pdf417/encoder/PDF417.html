542   /**
543    * Calculates the necessary number of rows as described in annex Q of ISO/IEC 15438:2001(E).
544    *
545    * @param m the number of source codewords prior to the additional of the Symbol Length
546    *          Descriptor and any pad codewords
547    * @param k the number of error correction codewords
548    * @param c the number of columns in the symbol in the data region (excluding start, stop and
549    *          row indicator codewords)
550    * @return the number of rows in the symbol (r)
551    */
552   private static int calculateNumberOfRows(int m, int k, int c) {
553     int r = ((m + 1 + k) / c) + 1;
554     if (c * r >= (m + 1 + k + c)) {
555       r--;
556     }
557     return r;
558   }

560   /**
561    * Calculates the number of pad codewords as described in 4.9.2 of ISO/IEC 15438:2001(E).
562    *
563    * @param m the number of source codewords prior to the additional of the Symbol Length
564    *          Descriptor and any pad codewords
565    * @param k the number of error correction codewords
566    * @param c the number of columns in the symbol in the data region (excluding start, stop and
567    *          row indicator codewords)
568    * @param r the number of rows in the symbol
569    * @return the number of pad codewords
570    */
571   private static int getNumberOfPadCodewords(int m, int k, int c, int r) {
572     int n = c * r - k;
573     return n > m + 1 ? n - m - 1 : 0;
574   }

576   private static void encodeChar(int pattern, int len, BarcodeRow logic) {
577     int map = 1 << len - 1;
578     boolean last = (pattern & map) != 0; //Initialize to inverse of first bit
579     int width = 0;
580     for (int i = 0; i < len; i++) {
581       boolean black = (pattern & map) != 0;
582       if (last == black) {
583         width++;
584       } else {
585         logic.addBar(last, width);
586 
587         last = black;
588         width = 1;
589       }
590       map >>= 1;
591     }
592     logic.addBar(last, width);
593   }

595   private void encodeLowLevel(CharSequence fullCodewords,
596                               int c,
597                               int r,
598                               int errorCorrectionLevel,
599                               BarcodeMatrix logic) {
600 
601     int idx = 0;
602     for (int y = 0; y < r; y++) {
603       int cluster = y % 3;
604       logic.startRow();
605       encodeChar(START_PATTERN, 17, logic.getCurrentRow());
606 
607       int left;
608       int right;
609       if (cluster == 0) {
610         left = (30 * (y / 3)) + ((r - 1) / 3);
611         right = (30 * (y / 3)) + (c - 1);
612       } else if (cluster == 1) {
613         left = (30 * (y / 3)) + (errorCorrectionLevel * 3) + ((r - 1) % 3);
614         right = (30 * (y / 3)) + ((r - 1) / 3);
615       } else {
616         left = (30 * (y / 3)) + (c - 1);
617         right = (30 * (y / 3)) + (errorCorrectionLevel * 3) + ((r - 1) % 3);
618       }
619 
620       int pattern = CODEWORD_TABLE[cluster][left];
621       encodeChar(pattern, 17, logic.getCurrentRow());
622 
623       for (int x = 0; x < c; x++) {
624         pattern = CODEWORD_TABLE[cluster][fullCodewords.charAt(idx)];
625         encodeChar(pattern, 17, logic.getCurrentRow());
626         idx++;
627       }
628 
629       if (compact) {
630         encodeChar(STOP_PATTERN, 1, logic.getCurrentRow()); // encodes stop line for compact pdf417
631       } else {
632         pattern = CODEWORD_TABLE[cluster][right];
633         encodeChar(pattern, 17, logic.getCurrentRow());
634 
635         encodeChar(STOP_PATTERN, 18, logic.getCurrentRow());
636       }
637     }
638   }

640   /**
641    * Generates the barcode logic.
642    *
643    * @param msg        the message to encode
644    */
645   public void generateBarcodeLogic(String msg, int errorCorrectionLevel) throws WriterException {
646 
647     //1. step: High-level encoding
648     int errorCorrectionCodeWords = PDF417ErrorCorrection.getErrorCorrectionCodewordCount(errorCorrectionLevel);
649     String highLevel = PDF417HighLevelEncoder.encodeHighLevel(msg, compaction, encoding);
650     int sourceCodeWords = highLevel.length();
651 
652     int[] dimension = determineDimensions(sourceCodeWords, errorCorrectionCodeWords);
653 
654     int cols = dimension[0];
655     int rows = dimension[1];
656 
657     int pad = getNumberOfPadCodewords(sourceCodeWords, errorCorrectionCodeWords, cols, rows);
658 
659     //2. step: construct data codewords
660     if (sourceCodeWords + errorCorrectionCodeWords + 1 > 929) { // +1 for symbol length CW
661       throw new WriterException(
662           "Encoded message contains to many code words, message to big (" + msg.length() + " bytes)");
663     }
664     int n = sourceCodeWords + pad + 1;
665     StringBuilder sb = new StringBuilder(n);
666     sb.append((char) n);
667     sb.append(highLevel);
668     for (int i = 0; i < pad; i++) {
669       sb.append((char) 900); //PAD characters
670     }
671     String dataCodewords = sb.toString();
672 
673     //3. step: Error correction
674     String ec = PDF417ErrorCorrection.generateErrorCorrection(dataCodewords, errorCorrectionLevel);
675     String fullCodewords = dataCodewords + ec;
676 
677     //4. step: low-level encoding
678     barcodeMatrix = new BarcodeMatrix(rows, cols);
679     encodeLowLevel(fullCodewords, cols, rows, errorCorrectionLevel, barcodeMatrix);
680   }

682   /**
683    * Determine optimal nr of columns and rows for the specified number of
684    * codewords.
685    *
686    * @param sourceCodeWords number of code words
687    * @param errorCorrectionCodeWords number of error correction code words
688    * @return dimension object containing cols as width and rows as height
689    */
690   private int[] determineDimensions(int sourceCodeWords, int errorCorrectionCodeWords) throws WriterException {
691     float ratio = 0.0f;
692     int[] dimension = null;
693 
694     for (int cols = minCols; cols <= maxCols; cols++) {
695 
696       int rows = calculateNumberOfRows(sourceCodeWords, errorCorrectionCodeWords, cols);
697 
698       if (rows < minRows) {
699         break;
700       }
701 
702       if (rows > maxRows) {
703         continue;
704       }
705 
706       float newRatio = ((17 * cols + 69) * DEFAULT_MODULE_WIDTH) / (rows * HEIGHT);
707 
708       // ignore if previous ratio is closer to preferred ratio
709       if (dimension != null && Math.abs(newRatio - PREFERRED_RATIO) > Math.abs(ratio - PREFERRED_RATIO)) {
710         continue;
711       }
712 
713       ratio = newRatio;
714       dimension = new int[] {cols, rows};
715     }
716 
717      // Handle case when min values were larger than necessary
718      if (dimension == null) {
719        int rows = calculateNumberOfRows(sourceCodeWords, errorCorrectionCodeWords, minCols);
720        if (rows < minRows) {
721          dimension = new int[]{minCols, minRows};
722        }
723      }
724 
725     if (dimension == null) {
726       throw new WriterException("Unable to fit message in columns");
727     }
728 
729     return dimension;
730   }
