130   /**
131    * Determines the number of error correction codewords for a specified error correction
132    * level.
133    *
134    * @param errorCorrectionLevel the error correction level (0-8)
135    * @return the number of codewords generated for error correction
136    */
137   static int getErrorCorrectionCodewordCount(int errorCorrectionLevel) {
138     if (errorCorrectionLevel < 0 || errorCorrectionLevel > 8) {
139       throw new IllegalArgumentException("Error correction level must be between 0 and 8!");
140     }
141     return 1 << (errorCorrectionLevel + 1);
142   }
143 
144   /**
145    * Returns the recommended minimum error correction level as described in annex E of
146    * ISO/IEC 15438:2001(E).
147    *
148    * @param n the number of data codewords
149    * @return the recommended minimum error correction level
150    */
151   static int getRecommendedMinimumErrorCorrectionLevel(int n) throws WriterException {
152     if (n <= 0) {
153       throw new IllegalArgumentException("n must be > 0");
154     }
155     if (n <= 40) {
156       return 2;
157     }
158     if (n <= 160) {
159       return 3;
160     }
161     if (n <= 320) {
162       return 4;
163     }
164     if (n <= 863) {
165       return 5;
166     }
167     throw new WriterException("No recommendation possible");
168   }

170   /**
171    * Generates the error correction codewords according to 4.10 in ISO/IEC 15438:2001(E).
172    *
173    * @param dataCodewords        the data codewords
174    * @param errorCorrectionLevel the error correction level (0-8)
175    * @return the String representing the error correction codewords
176    */
177   static String generateErrorCorrection(CharSequence dataCodewords, int errorCorrectionLevel) {
178     int k = getErrorCorrectionCodewordCount(errorCorrectionLevel);
179     char[] e = new char[k];
180     int sld = dataCodewords.length();
181     for (int i = 0; i < sld; i++) {
182       int t1 = (dataCodewords.charAt(i) + e[e.length - 1]) % 929;
183       int t2;
184       int t3;
185       for (int j = k - 1; j >= 1; j--) {
186         t2 = (t1 * EC_COEFFICIENTS[errorCorrectionLevel][j]) % 929;
187         t3 = 929 - t2;
188         e[j] = (char) ((e[j - 1] + t3) % 929);
189       }
190       t2 = (t1 * EC_COEFFICIENTS[errorCorrectionLevel][0]) % 929;
191       t3 = 929 - t2;
192       e[0] = (char) (t3 % 929);
193     }
194     StringBuilder sb = new StringBuilder(k);
195     for (int j = k - 1; j >= 0; j--) {
196       if (e[j] != 0) {
197         e[j] = (char) (929 - e[j]);
198       }
199       sb.append(e[j]);
200     }
201     return sb.toString();
202   }
