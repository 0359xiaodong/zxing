17  package com.google.zxing.pdf417.decoder.ec;
18  
19  import com.google.zxing.ChecksumException;
20  
21  /**
22   * <p>PDF417 error correction implementation.</p>
23   *
24   * <p>This <a href="http://en.wikipedia.org/wiki/Reed%E2%80%93Solomon_error_correction#Example">example</a>
25   * is quite useful in understanding the algorithm.</p>
26   *
27   * @author Sean Owen
28   * @see com.google.zxing.common.reedsolomon.ReedSolomonDecoder
29   */
30  public final class ErrorCorrection {
31  
32    private final ModulusGF field;
33  
34    public ErrorCorrection() {
35      this.field = ModulusGF.PDF417_GF;
36    }
37  
38    /**
39     * @param received received codewords
40     * @param numECCodewords number of those codewords used for EC
41     * @param erasures location of erasures
42     * @return number of errors
43     * @throws ChecksumException if errors cannot be corrected, maybe because of too many errors
44     */
45    public int decode(int[] received,
46                      int numECCodewords,
47                      int[] erasures) throws ChecksumException {
48  
49      ModulusPoly poly = new ModulusPoly(field, received);
50      int[] S = new int[numECCodewords];
51      boolean error = false;
52      for (int i = numECCodewords; i > 0; i--) {
53        int eval = poly.evaluateAt(field.exp(i));
54        S[numECCodewords - i] = eval;
55        if (eval != 0) {
56          error = true;
57        }
58      }
59  
60      if (!error) {
61        return 0;
62      }
63  
64      ModulusPoly knownErrors = field.getOne();
65      for (int erasure : erasures) {
66        int b = field.exp(received.length - 1 - erasure);
67        // Add (1 - bx) term:
68        ModulusPoly term = new ModulusPoly(field, new int[] { field.subtract(0, b), 1 });
69        knownErrors = knownErrors.multiply(term);
70      }
71  
72      ModulusPoly syndrome = new ModulusPoly(field, S);
73      //syndrome = syndrome.multiply(knownErrors);
74  
75      ModulusPoly[] sigmaOmega =
76          runEuclideanAlgorithm(field.buildMonomial(numECCodewords, 1), syndrome, numECCodewords);
77      ModulusPoly sigma = sigmaOmega[0];
78      ModulusPoly omega = sigmaOmega[1];
79  
80      //sigma = sigma.multiply(knownErrors);
81  
82      int[] errorLocations = findErrorLocations(sigma);
83      int[] errorMagnitudes = findErrorMagnitudes(omega, sigma, errorLocations);
84  
85      for (int i = 0; i < errorLocations.length; i++) {
86        int position = received.length - 1 - field.log(errorLocations[i]);
87        if (position < 0) {
88          throw ChecksumException.getChecksumInstance();
89        }
90        received[position] = field.subtract(received[position], errorMagnitudes[i]);
91      }
92      return errorLocations.length;
93    }
94  
95    private ModulusPoly[] runEuclideanAlgorithm(ModulusPoly a, ModulusPoly b, int R)
96        throws ChecksumException {
97      // Assume a's degree is >= b's
98      if (a.getDegree() < b.getDegree()) {
99        ModulusPoly temp = a;
100       a = b;
101       b = temp;
102     }
103 
104     ModulusPoly rLast = a;
105     ModulusPoly r = b;
106     ModulusPoly tLast = field.getZero();
107     ModulusPoly t = field.getOne();
108 
109     // Run Euclidean algorithm until r's degree is less than R/2
110     while (r.getDegree() >= R / 2) {
111       ModulusPoly rLastLast = rLast;
112       ModulusPoly tLastLast = tLast;
113       rLast = r;
114       tLast = t;
115 
116       // Divide rLastLast by rLast, with quotient in q and remainder in r
117       if (rLast.isZero()) {
118         // Oops, Euclidean algorithm already terminated?
119         throw ChecksumException.getChecksumInstance();
120       }
121       r = rLastLast;
122       ModulusPoly q = field.getZero();
123       int denominatorLeadingTerm = rLast.getCoefficient(rLast.getDegree());
124       int dltInverse = field.inverse(denominatorLeadingTerm);
125       while (r.getDegree() >= rLast.getDegree() && !r.isZero()) {
126         int degreeDiff = r.getDegree() - rLast.getDegree();
127         int scale = field.multiply(r.getCoefficient(r.getDegree()), dltInverse);
128         q = q.add(field.buildMonomial(degreeDiff, scale));
129         r = r.subtract(rLast.multiplyByMonomial(degreeDiff, scale));
130       }
131 
132       t = q.multiply(tLast).subtract(tLastLast).negative();
133     }
134 
135     int sigmaTildeAtZero = t.getCoefficient(0);
136     if (sigmaTildeAtZero == 0) {
137       throw ChecksumException.getChecksumInstance();
138     }
139 
140     int inverse = field.inverse(sigmaTildeAtZero);
141     ModulusPoly sigma = t.multiply(inverse);
142     ModulusPoly omega = r.multiply(inverse);
143     return new ModulusPoly[]{sigma, omega};
144   }
145 
146   private int[] findErrorLocations(ModulusPoly errorLocator) throws ChecksumException {
147     // This is a direct application of Chien's search
148     int numErrors = errorLocator.getDegree();
149     int[] result = new int[numErrors];
150     int e = 0;
151     for (int i = 1; i < field.getSize() && e < numErrors; i++) {
152       if (errorLocator.evaluateAt(i) == 0) {
153         result[e] = field.inverse(i);
154         e++;
155       }
156     }
157     if (e != numErrors) {
158       throw ChecksumException.getChecksumInstance();
159     }
160     return result;
161   }
162 
163   private int[] findErrorMagnitudes(ModulusPoly errorEvaluator,
164                                     ModulusPoly errorLocator,
165                                     int[] errorLocations) {
166     int errorLocatorDegree = errorLocator.getDegree();
167     int[] formalDerivativeCoefficients = new int[errorLocatorDegree];
168     for (int i = 1; i <= errorLocatorDegree; i++) {
169       formalDerivativeCoefficients[errorLocatorDegree - i] =
170           field.multiply(i, errorLocator.getCoefficient(i));
171     }
172     ModulusPoly formalDerivative = new ModulusPoly(field, formalDerivativeCoefficients);
173 
174     // This is directly applying Forney's Formula
175     int s = errorLocations.length;
176     int[] result = new int[s];
177     for (int i = 0; i < s; i++) {
178       int xiInverse = field.inverse(errorLocations[i]);
179       int numerator = field.subtract(0, errorEvaluator.evaluateAt(xiInverse));
180       int denominator = field.inverse(formalDerivative.evaluateAt(xiInverse));
181       result[i] = field.multiply(numerator, denominator);
182     }
183     return result;
184   }
