47    // TODO don't pass in minCodewordWidth and maxCodewordWidth, pass in barcode columns for start and stop pattern
48    // columns. That way width can be deducted from the pattern column.
49    // This approach also allows to detect more details about the barcode, e.g. if a bar type (white or black) is wider 
50    // than it should be. This can happen if the scanner used a bad blackpoint.
51    public static DecoderResult decode(BitMatrix image,
52                                       ResultPoint imageTopLeft,
53                                       ResultPoint imageBottomLeft,
54                                       ResultPoint imageTopRight,
55                                       ResultPoint imageBottomRight,
56                                       int minCodewordWidth,
57                                       int maxCodewordWidth) throws NotFoundException, FormatException, ChecksumException {
58      BoundingBox boundingBox = new BoundingBox(image, imageTopLeft, imageBottomLeft, imageTopRight, imageBottomRight);
59      DetectionResultRowIndicatorColumn leftRowIndicatorColumn = null;
60      DetectionResultRowIndicatorColumn rightRowIndicatorColumn = null;
61      DetectionResult detectionResult = null;
62      for (int i = 0; i < 2; i++) {
63        if (imageTopLeft != null) {
64          leftRowIndicatorColumn = getRowIndicatorColumn(image, boundingBox, imageTopLeft, true, minCodewordWidth,
65              maxCodewordWidth);
66        }
67        if (imageTopRight != null) {
68          rightRowIndicatorColumn = getRowIndicatorColumn(image, boundingBox, imageTopRight, false, minCodewordWidth,
69              maxCodewordWidth);
70        }
71        detectionResult = merge(leftRowIndicatorColumn, rightRowIndicatorColumn);
72        if (detectionResult == null) {
73          throw NotFoundException.getNotFoundInstance();
74        }
75        if (i == 0 && detectionResult.getBoundingBox() != null &&
76            (detectionResult.getBoundingBox().getMinY() < boundingBox.getMinY() || detectionResult.getBoundingBox()
77                .getMaxY() > boundingBox.getMaxY())) {
78          boundingBox = detectionResult.getBoundingBox();
79        } else {
80          detectionResult.setBoundingBox(boundingBox);
81          break;
82        }
83      }
84      int maxBarcodeColumn = detectionResult.getBarcodeColumnCount() + 1;
85      detectionResult.setDetectionResultColumn(0, leftRowIndicatorColumn);
86      detectionResult.setDetectionResultColumn(maxBarcodeColumn, rightRowIndicatorColumn);
87  
88      boolean leftToRight = leftRowIndicatorColumn != null;
89      for (int barcodeColumnCount = 1; barcodeColumnCount <= maxBarcodeColumn; barcodeColumnCount++) {
90        int barcodeColumn = leftToRight ? barcodeColumnCount : maxBarcodeColumn - barcodeColumnCount;
91        if (detectionResult.getDetectionResultColumn(barcodeColumn) != null) {
92          // This will be the case for the opposite row indicator column, which doesn't need to be decoded again.
93          continue;
94        }
95        DetectionResultColumn detectionResultColumn;
96        if (barcodeColumn == 0 || barcodeColumn == maxBarcodeColumn) {
97          detectionResultColumn = new DetectionResultRowIndicatorColumn(boundingBox, barcodeColumn == 0);
98        } else {
99          detectionResultColumn = new DetectionResultColumn(boundingBox);
100       }
101       detectionResult.setDetectionResultColumn(barcodeColumn, detectionResultColumn);
102       int startColumn = -1;
103       int previousStartColumn = startColumn;
104       // TODO start at a row for which we know the start position, then detect upwards and downwards from there.
105       for (int imageRow = boundingBox.getMinY(); imageRow <= boundingBox.getMaxY(); imageRow++) {
106         startColumn = getStartColumn(detectionResult, barcodeColumn, imageRow, leftToRight);
107         if (startColumn < 0 || startColumn > boundingBox.getMaxX()) {
108           if (previousStartColumn == -1) {
109             continue;
110           }
111           startColumn = previousStartColumn;
112         }
113         Codeword codeword = detectCodeword(image, boundingBox.getMinX(), boundingBox.getMaxX(), leftToRight,
114             startColumn, imageRow, minCodewordWidth, maxCodewordWidth);
115         if (codeword != null) {
116           detectionResultColumn.setCodeword(imageRow, codeword);
117           previousStartColumn = startColumn;
118           minCodewordWidth = Math.min(minCodewordWidth, codeword.getWidth());
119           maxCodewordWidth = Math.max(maxCodewordWidth, codeword.getWidth());
120         }
121       }
122     }
123     return createDecoderResult(detectionResult);
124   }

126   private static DetectionResult merge(DetectionResultRowIndicatorColumn leftRowIndicatorColumn,
127                                        DetectionResultRowIndicatorColumn rightRowIndicatorColumn)
128       throws NotFoundException, FormatException {
129     if (leftRowIndicatorColumn == null && rightRowIndicatorColumn == null) {
130       return null;
131     }
132     BarcodeMetadata barcodeMetadata = getBarcodeMetadata(leftRowIndicatorColumn, rightRowIndicatorColumn);
133     if (barcodeMetadata == null) {
134       return null;
135     }
136     BoundingBox boundingBox = BoundingBox.merge(adjustBoundingBox(leftRowIndicatorColumn),
137         adjustBoundingBox(rightRowIndicatorColumn));
138     return new DetectionResult(barcodeMetadata, boundingBox);
139   }

141   private static BoundingBox adjustBoundingBox(DetectionResultRowIndicatorColumn rowIndicatorColumn)
142       throws NotFoundException, FormatException {
143     if (rowIndicatorColumn == null) {
144       return null;
145     }
146     int[] rowHeights = rowIndicatorColumn.getRowHeights();
147     if (rowHeights == null) {
148       return null;
149     }
150     int maxRowHeight = getMax(rowHeights);
151     int missingStartRows = 0;
152     for (int rowHeight : rowHeights) {
153       missingStartRows += maxRowHeight - rowHeight;
154       if (rowHeight > 0) {
155         break;
156       }
157     }
158     Codeword[] codewords = rowIndicatorColumn.getCodewords();
159     for (int row = 0; missingStartRows > 0 && codewords[row] == null; row++) {
160       missingStartRows--;
161     }
162     int missingEndRows = 0;
163     for (int row = rowHeights.length - 1; row >= 0; row--) {
164       missingEndRows += maxRowHeight - rowHeights[row];
165       if (rowHeights[row] > 0) {
166         break;
167       }
168     }
169     for (int row = codewords.length - 1; missingEndRows > 0 && codewords[row] == null; row--) {
170       missingEndRows--;
171     }
172     return rowIndicatorColumn.getBoundingBox().addMissingRows(missingStartRows, missingEndRows,
173         rowIndicatorColumn.isLeft());
174   }
175 
176   private static int getMax(int[] values) {
177     int maxValue = -1;
178     for (int value : values) {
179       maxValue = Math.max(maxValue, value);
180     }
181     return maxValue;
182   }

184   private static BarcodeMetadata getBarcodeMetadata(DetectionResultRowIndicatorColumn leftRowIndicatorColumn,
185                                                     DetectionResultRowIndicatorColumn rightRowIndicatorColumn) {
186     BarcodeMetadata leftBarcodeMetadata;
187     if (leftRowIndicatorColumn == null ||
188         (leftBarcodeMetadata = leftRowIndicatorColumn.getBarcodeMetadata()) == null) {
189       return rightRowIndicatorColumn == null ? null : rightRowIndicatorColumn.getBarcodeMetadata();
190     }
191     BarcodeMetadata rightBarcodeMetadata;
192     if (rightRowIndicatorColumn == null ||
193         (rightBarcodeMetadata = rightRowIndicatorColumn.getBarcodeMetadata()) == null) {
194       return leftBarcodeMetadata;
195     }
196 
197     if (leftBarcodeMetadata.getColumnCount() != rightBarcodeMetadata.getColumnCount() &&
198         leftBarcodeMetadata.getErrorCorrectionLevel() != rightBarcodeMetadata.getErrorCorrectionLevel() &&
199         leftBarcodeMetadata.getRowCount() != rightBarcodeMetadata.getRowCount()) {
200       return null;
201     }
202     return leftBarcodeMetadata;
203   }

205   private static DetectionResultRowIndicatorColumn getRowIndicatorColumn(BitMatrix image,
206                                                                          BoundingBox boundingBox,
207                                                                          ResultPoint startPoint,
208                                                                          boolean leftToRight,
209                                                                          int minCodewordWidth,
210                                                                          int maxCodewordWidth) {
211     DetectionResultRowIndicatorColumn rowIndicatorColumn = new DetectionResultRowIndicatorColumn(boundingBox,
212         leftToRight);
213     for (int i = 0; i < 2; i++) {
214       int increment = i == 0 ? 1 : -1;
215       int startColumn = (int) startPoint.getX();
216       for (int imageRow = (int) startPoint.getY(); imageRow <= boundingBox.getMaxY() &&
217           imageRow >= boundingBox.getMinY(); imageRow += increment) {
218         Codeword codeword = detectCodeword(image, 0, image.getWidth(), leftToRight, startColumn, imageRow,
219             minCodewordWidth, maxCodewordWidth);
220         if (codeword != null) {
221           rowIndicatorColumn.setCodeword(imageRow, codeword);
222           if (leftToRight) {
223             startColumn = codeword.getStartX();
224           } else {
225             startColumn = codeword.getEndX();
226           }
227         }
228       }
229     }
230     return rowIndicatorColumn;
231   }

233   private static void adjustCodewordCount(DetectionResult detectionResult, BarcodeValue[][] barcodeMatrix)
234       throws NotFoundException {
235     int[] numberOfCodewords = barcodeMatrix[0][1].getValue();
236     int calculatedNumberOfCodewords = detectionResult.getBarcodeColumnCount() *
237         detectionResult.getBarcodeRowCount() -
238         getNumberOfECCodeWords(detectionResult.getBarcodeECLevel());
239     if (numberOfCodewords.length == 0) {
240       if (calculatedNumberOfCodewords < 1 || calculatedNumberOfCodewords > PDF417Common.MAX_CODEWORDS_IN_BARCODE) {
241         throw NotFoundException.getNotFoundInstance();
242       }
243       barcodeMatrix[0][1].setValue(calculatedNumberOfCodewords);
244     } else if (numberOfCodewords[0] != calculatedNumberOfCodewords) {
245       // The calculated one is more reliable as it is derived from the row indicator columns
246       barcodeMatrix[0][1].setValue(calculatedNumberOfCodewords);
247     }
248   }

250   private static DecoderResult createDecoderResult(DetectionResult detectionResult) throws FormatException,
251       ChecksumException, NotFoundException {
252     BarcodeValue[][] barcodeMatrix = createBarcodeMatrix(detectionResult);
253     adjustCodewordCount(detectionResult, barcodeMatrix);
254     Collection<Integer> erasures = new ArrayList<>();
255     int[] codewords = new int[detectionResult.getBarcodeRowCount() * detectionResult.getBarcodeColumnCount()];
256     List<int[]> ambiguousIndexValuesList = new ArrayList<>();
257     List<Integer> ambiguousIndexesList = new ArrayList<>();
258     for (int row = 0; row < detectionResult.getBarcodeRowCount(); row++) {
259       for (int column = 0; column < detectionResult.getBarcodeColumnCount(); column++) {
260         int[] values = barcodeMatrix[row][column + 1].getValue();
261         int codewordIndex = row * detectionResult.getBarcodeColumnCount() + column;
262         if (values.length == 0) {
263           erasures.add(codewordIndex);
264         } else if (values.length == 1) {
265           codewords[codewordIndex] = values[0];
266         } else {
267           ambiguousIndexesList.add(codewordIndex);
268           ambiguousIndexValuesList.add(values);
269         }
270       }
271     }
272     int[][] ambiguousIndexValues = new int[ambiguousIndexValuesList.size()][];
273     for (int i = 0; i < ambiguousIndexValues.length; i++) {
274       ambiguousIndexValues[i] = ambiguousIndexValuesList.get(i);
275     }
276     return createDecoderResultFromAmbiguousValues(detectionResult.getBarcodeECLevel(), codewords,
277         PDF417Common.toIntArray(erasures), PDF417Common.toIntArray(ambiguousIndexesList), ambiguousIndexValues);
278   }

280   /**
281    * This method deals with the fact, that the decoding process doesn't always yield a single most likely value. The
282    * current error correction implementation doesn't deal with erasures very well, so it's better to provide a value
283    * for these ambiguous codewords instead of treating it as an erasure. The problem is that we don't know which of
284    * the ambiguous values to choose. We try decode using the first value, and if that fails, we use another of the
285    * ambiguous values and try to decode again. This usually only happens on very hard to read and decode barcodes,
286    * so decoding the normal barcodes is not affected by this. 
287    *
288    * @param erasureArray contains the indexes of erasures
289    * @param ambiguousIndexes array with the indexes that have more than one most likely value
290    * @param ambiguousIndexValues two dimensional array that contains the ambiguous values. The first dimension must
291    * be the same length as the ambiguousIndexes array
292    */
293   private static DecoderResult createDecoderResultFromAmbiguousValues(int ecLevel,
294                                                                       int[] codewords,
295                                                                       int[] erasureArray,
296                                                                       int[] ambiguousIndexes,
297                                                                       int[][] ambiguousIndexValues)
298       throws FormatException, ChecksumException {
299     int[] ambiguousIndexCount = new int[ambiguousIndexes.length];
300 
301     int tries = 100;
302     while (tries-- > 0) {
303       for (int i = 0; i < ambiguousIndexCount.length; i++) {
304         codewords[ambiguousIndexes[i]] = ambiguousIndexValues[i][ambiguousIndexCount[i]];
305       }
306       try {
307         return decodeCodewords(codewords, ecLevel, erasureArray);
308       } catch (ChecksumException ignored) {
309         //
310       }
311       if (ambiguousIndexCount.length == 0) {
312         throw ChecksumException.getChecksumInstance();
313       }
314       for (int i = 0; i < ambiguousIndexCount.length; i++) {
315         if (ambiguousIndexCount[i] < ambiguousIndexValues[i].length - 1) {
316           ambiguousIndexCount[i]++;
317           break;
318         } else {
319           ambiguousIndexCount[i] = 0;
320           if (i == ambiguousIndexCount.length - 1) {
321             throw ChecksumException.getChecksumInstance();
322           }
323         }
324       }
325     }
326     throw ChecksumException.getChecksumInstance();
327   }

329   private static BarcodeValue[][] createBarcodeMatrix(DetectionResult detectionResult) throws FormatException {
330     BarcodeValue[][] barcodeMatrix =
331         new BarcodeValue[detectionResult.getBarcodeRowCount()][detectionResult.getBarcodeColumnCount() + 2];
332     for (int row = 0; row < barcodeMatrix.length; row++) {
333       for (int column = 0; column < barcodeMatrix[row].length; column++) {
334         barcodeMatrix[row][column] = new BarcodeValue();
335       }
336     }
337 
338     int column = 0;
339     for (DetectionResultColumn detectionResultColumn : detectionResult.getDetectionResultColumns()) {
340       if (detectionResultColumn != null) {
341         for (Codeword codeword : detectionResultColumn.getCodewords()) {
342           if (codeword != null) {
343             int rowNumber = codeword.getRowNumber();
344             if (rowNumber >= 0) {
345               if (rowNumber >= barcodeMatrix.length) {
346                 throw FormatException.getFormatInstance();
347               }
348               barcodeMatrix[rowNumber][column].setValue(codeword.getValue());
349             }
350           }
351         }
352       }
353       column++;
354     }
355     return barcodeMatrix;
356   }

362   private static int getStartColumn(DetectionResult detectionResult,
363                                     int barcodeColumn,
364                                     int imageRow,
365                                     boolean leftToRight) {
366     int offset = leftToRight ? 1 : -1;
367     Codeword codeword = null;
368     if (isValidBarcodeColumn(detectionResult, barcodeColumn - offset)) {
369       codeword = detectionResult.getDetectionResultColumn(barcodeColumn - offset).getCodeword(imageRow);
370     }
371     if (codeword != null) {
372       return leftToRight ? codeword.getEndX() : codeword.getStartX();
373     }
374     codeword = detectionResult.getDetectionResultColumn(barcodeColumn).getCodewordNearby(imageRow);
375     if (codeword != null) {
376       return leftToRight ? codeword.getStartX() : codeword.getEndX();
377     }
378     if (isValidBarcodeColumn(detectionResult, barcodeColumn - offset)) {
379       codeword = detectionResult.getDetectionResultColumn(barcodeColumn - offset).getCodewordNearby(imageRow);
380     }
381     if (codeword != null) {
382       return leftToRight ? codeword.getEndX() : codeword.getStartX();
383     }
384     int skippedColumns = 0;
385 
386     while (isValidBarcodeColumn(detectionResult, barcodeColumn - offset)) {
387       barcodeColumn -= offset;
388       for (Codeword previousRowCodeword : detectionResult.getDetectionResultColumn(barcodeColumn).getCodewords()) {
389         if (previousRowCodeword != null) {
390           return (leftToRight ? previousRowCodeword.getEndX() : previousRowCodeword.getStartX()) +
391               offset *
392               skippedColumns *
393               (previousRowCodeword.getEndX() - previousRowCodeword.getStartX());
394         }
395       }
396       skippedColumns++;
397     }
398     return leftToRight ? detectionResult.getBoundingBox().getMinX() : detectionResult.getBoundingBox().getMaxX();
399   }

401   private static Codeword detectCodeword(BitMatrix image,
402                                          int minColumn,
403                                          int maxColumn,
404                                          boolean leftToRight,
405                                          int startColumn,
406                                          int imageRow,
407                                          int minCodewordWidth,
408                                          int maxCodewordWidth) {
409     startColumn = adjustCodewordStartColumn(image, minColumn, maxColumn, leftToRight, startColumn, imageRow);
410     // we usually know fairly exact now how long a codeword is. We should provide minimum and maximum expected length
411     // and try to adjust the read pixels, e.g. remove single pixel errors or try to cut off exceeding pixels.
412     // min and maxCodewordWidth should not be used as they are calculated for the whole barcode an can be inaccurate
413     // for the current position
414     int[] moduleBitCount = getModuleBitCount(image, minColumn, maxColumn, leftToRight, startColumn, imageRow);
415     if (moduleBitCount == null) {
416       return null;
417     }
418     int endColumn;
419     int codewordBitCount = PDF417Common.getBitCountSum(moduleBitCount);
420     if (leftToRight) {
421       endColumn = startColumn + codewordBitCount;
422     } else {
423       for (int i = 0; i < moduleBitCount.length / 2; i++) {
424         int tmpCount = moduleBitCount[i];
425         moduleBitCount[i] = moduleBitCount[moduleBitCount.length - 1 - i];
426         moduleBitCount[moduleBitCount.length - 1 - i] = tmpCount;
427       }
428       endColumn = startColumn;
429       startColumn = endColumn - codewordBitCount;
430     }
431     // TODO implement check for width and correction of black and white bars
432     // use start (and maybe stop pattern) to determine if blackbars are wider than white bars. If so, adjust.
433     // should probably done only for codewords with a lot more than 17 bits. 
434     // The following fixes 10-1.png, which has wide black bars and small white bars
435     //    for (int i = 0; i < moduleBitCount.length; i++) {
436     //      if (i % 2 == 0) {
437     //        moduleBitCount[i]--;
438     //      } else {
439     //        moduleBitCount[i]++;
440     //      }
441     //    }
442 
443     // We could also use the width of surrounding codewords for more accurate results, but this seems
444     // sufficient for now
445     if (!checkCodewordSkew(codewordBitCount, minCodewordWidth, maxCodewordWidth)) {
446       // We could try to use the startX and endX position of the codeword in the same column in the previous row,
447       // create the bit count from it and normalize it to 8. This would help with single pixel errors.
448       return null;
449     }
450 
451     int decodedValue = PDF417CodewordDecoder.getDecodedValue(moduleBitCount);
452     int codeword = PDF417Common.getCodeword(decodedValue);
453     if (codeword == -1) {
454       return null;
455     }
456     return new Codeword(startColumn, endColumn, getCodewordBucketNumber(decodedValue), codeword);
457   }

459   private static int[] getModuleBitCount(BitMatrix image,
460                                          int minColumn,
461                                          int maxColumn,
462                                          boolean leftToRight,
463                                          int startColumn,
464                                          int imageRow) {
465     int imageColumn = startColumn;
466     int[] moduleBitCount = new int[8];
467     int moduleNumber = 0;
468     int increment = leftToRight ? 1 : -1;
469     boolean previousPixelValue = leftToRight;
470     while (((leftToRight && imageColumn < maxColumn) || (!leftToRight && imageColumn >= minColumn)) &&
471         moduleNumber < moduleBitCount.length) {
472       if (image.get(imageColumn, imageRow) == previousPixelValue) {
473         moduleBitCount[moduleNumber]++;
474         imageColumn += increment;
475       } else {
476         moduleNumber++;
477         previousPixelValue = !previousPixelValue;
478       }
479     }
480     if (moduleNumber == moduleBitCount.length ||
481         (((leftToRight && imageColumn == maxColumn) || (!leftToRight && imageColumn == minColumn)) && moduleNumber == moduleBitCount.length - 1)) {
482       return moduleBitCount;
483     }
484     return null;
485   }

487   private static int getNumberOfECCodeWords(int barcodeECLevel) {
488     return 2 << barcodeECLevel;

491   private static int adjustCodewordStartColumn(BitMatrix image,
492                                                int minColumn,
493                                                int maxColumn,
494                                                boolean leftToRight,
495                                                int codewordStartColumn,
496                                                int imageRow) {
497     int correctedStartColumn = codewordStartColumn;
498     int increment = leftToRight ? -1 : 1;
499     // there should be no black pixels before the start column. If there are, then we need to start earlier.
500     for (int i = 0; i < 2; i++) {
501       while (((leftToRight && correctedStartColumn >= minColumn) || (!leftToRight && correctedStartColumn < maxColumn)) &&
502           leftToRight == image.get(correctedStartColumn, imageRow)) {
503         if (Math.abs(codewordStartColumn - correctedStartColumn) > CODEWORD_SKEW_SIZE) {
504           return codewordStartColumn;
505         }
506         correctedStartColumn += increment;
507       }
508       increment = -increment;
509       leftToRight = !leftToRight;
510     }
511     return correctedStartColumn;
512   }

514   private static boolean checkCodewordSkew(int codewordSize, int minCodewordWidth, int maxCodewordWidth) {
515     return minCodewordWidth - CODEWORD_SKEW_SIZE <= codewordSize &&
516         codewordSize <= maxCodewordWidth + CODEWORD_SKEW_SIZE;
517   }

519   private static DecoderResult decodeCodewords(int[] codewords, int ecLevel, int[] erasures) throws FormatException,
520       ChecksumException {
521     if (codewords.length == 0) {
522       throw FormatException.getFormatInstance();
523     }
524 
525     int numECCodewords = 1 << (ecLevel + 1);
526     int correctedErrorsCount = correctErrors(codewords, erasures, numECCodewords);
527     verifyCodewordCount(codewords, numECCodewords);
528 
529     // Decode the codewords
530     DecoderResult decoderResult = DecodedBitStreamParser.decode(codewords, String.valueOf(ecLevel));
531     decoderResult.setErrorsCorrected(correctedErrorsCount);
532     decoderResult.setErasures(erasures.length);
533     return decoderResult;
534   }

536   /**
537    * <p>Given data and error-correction codewords received, possibly corrupted by errors, attempts to
538    * correct the errors in-place.</p>
539    *
540    * @param codewords   data and error correction codewords
541    * @param erasures positions of any known erasures
542    * @param numECCodewords number of error correction codewords that are available in codewords
543    * @throws ChecksumException if error correction fails
544    */
545   private static int correctErrors(int[] codewords, int[] erasures, int numECCodewords) throws ChecksumException {
546     if (erasures != null &&
547         erasures.length > numECCodewords / 2 + MAX_ERRORS ||
548         numECCodewords < 0 ||
549         numECCodewords > MAX_EC_CODEWORDS) {
550       // Too many errors or EC Codewords is corrupted
551       throw ChecksumException.getChecksumInstance();
552     }
553     return errorCorrection.decode(codewords, numECCodewords, erasures);
554   }

556   /**
557    * Verify that all is OK with the codeword array.
558    */
559   private static void verifyCodewordCount(int[] codewords, int numECCodewords) throws FormatException {
560     if (codewords.length < 4) {
561       // Codeword array size should be at least 4 allowing for
562       // Count CW, At least one Data CW, Error Correction CW, Error Correction CW
563       throw FormatException.getFormatInstance();
564     }
565     // The first codeword, the Symbol Length Descriptor, shall always encode the total number of data
566     // codewords in the symbol, including the Symbol Length Descriptor itself, data codewords and pad
567     // codewords, but excluding the number of error correction codewords.
568     int numberOfCodewords = codewords[0];
569     if (numberOfCodewords > codewords.length) {
570       throw FormatException.getFormatInstance();
571     }
572     if (numberOfCodewords == 0) {
573       // Reset to the length of the array - 8 (Allow for at least level 3 Error Correction (8 Error Codewords)
574       if (numECCodewords < codewords.length) {
575         codewords[0] = codewords.length - numECCodewords;
576       } else {
577         throw FormatException.getFormatInstance();
578       }
579     }
580   }

582   private static int[] getBitCountForCodeword(int codeword) {
583     int[] result = new int[8];
584     int previousValue = 0;
585     int i = result.length - 1;
586     while (true) {
587       if ((codeword & 0x1) != previousValue) {
588         previousValue = codeword & 0x1;
589         i--;
590         if (i < 0) {
591           break;
592         }
593       }
594       result[i]++;
595       codeword >>= 1;
596     }
597     return result;
598   }

600   private static int getCodewordBucketNumber(int codeword) {
601     return getCodewordBucketNumber(getBitCountForCodeword(codeword));
602   }

604   private static int getCodewordBucketNumber(int[] moduleBitCount) {
605     return (moduleBitCount[0] - moduleBitCount[2] + moduleBitCount[4] - moduleBitCount[6] + 9) % 9;
606   }

608   public static String toString(BarcodeValue[][] barcodeMatrix) {
609     Formatter formatter = new Formatter();
610     for (int row = 0; row < barcodeMatrix.length; row++) {
611       formatter.format("Row %2d: ", row);
612       for (int column = 0; column < barcodeMatrix[row].length; column++) {
613         BarcodeValue barcodeValue = barcodeMatrix[row][column];
614         if (barcodeValue.getValue().length == 0) {
615           formatter.format("        ", (Object[]) null);
616         } else {
617           formatter.format("%4d(%2d)", barcodeValue.getValue()[0],
618               barcodeValue.getConfidence(barcodeValue.getValue()[0]));
619         }
620       }
621       formatter.format("%n");
622     }
623     String result = formatter.toString();
624     formatter.close();
625     return result;
626   }
