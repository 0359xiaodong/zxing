130   // Build 2D matrix of QR Code from "dataBits" with "ecLevel", "version" and "getMaskPattern". On
131   // success, store the result in "matrix" and return true.
132   static void buildMatrix(BitArray dataBits,
133                           ErrorCorrectionLevel ecLevel,
134                           Version version,
135                           int maskPattern,
136                           ByteMatrix matrix) throws WriterException {
137     clearMatrix(matrix);
138     embedBasicPatterns(version, matrix);
139     // Type information appear with any version.
140     embedTypeInfo(ecLevel, maskPattern, matrix);
141     // Version info appear if version >= 7.
142     maybeEmbedVersionInfo(version, matrix);
143     // Data should be embedded at end.
144     embedDataBits(dataBits, maskPattern, matrix);
145   }

147   // Embed basic patterns. On success, modify the matrix and return true.
148   // The basic patterns are:
149   // - Position detection patterns
150   // - Timing patterns
151   // - Dark dot at the left bottom corner
152   // - Position adjustment patterns, if need be
153   static void embedBasicPatterns(Version version, ByteMatrix matrix) throws WriterException {
154     // Let's get started with embedding big squares at corners.
155     embedPositionDetectionPatternsAndSeparators(matrix);
156     // Then, embed the dark dot at the left bottom corner.
157     embedDarkDotAtLeftBottomCorner(matrix);
158 
159     // Position adjustment patterns appear if version >= 2.
160     maybeEmbedPositionAdjustmentPatterns(version, matrix);
161     // Timing patterns should be embedded after position adj. patterns.
162     embedTimingPatterns(matrix);
163   }

165   // Embed type information. On success, modify the matrix.
166   static void embedTypeInfo(ErrorCorrectionLevel ecLevel, int maskPattern, ByteMatrix matrix)
167       throws WriterException {
168     BitArray typeInfoBits = new BitArray();
169     makeTypeInfoBits(ecLevel, maskPattern, typeInfoBits);
170 
171     for (int i = 0; i < typeInfoBits.getSize(); ++i) {
172       // Place bits in LSB to MSB order.  LSB (least significant bit) is the last value in
173       // "typeInfoBits".
174       boolean bit = typeInfoBits.get(typeInfoBits.getSize() - 1 - i);
175 
176       // Type info bits at the left top corner. See 8.9 of JISX0510:2004 (p.46).
177       int x1 = TYPE_INFO_COORDINATES[i][0];
178       int y1 = TYPE_INFO_COORDINATES[i][1];
179       matrix.set(x1, y1, bit);
180 
181       if (i < 8) {
182         // Right top corner.
183         int x2 = matrix.getWidth() - i - 1;
184         int y2 = 8;
185         matrix.set(x2, y2, bit);
186       } else {
187         // Left bottom corner.
188         int x2 = 8;
189         int y2 = matrix.getHeight() - 7 + (i - 8);
190         matrix.set(x2, y2, bit);
191       }
192     }
193   }

195   // Embed version information if need be. On success, modify the matrix and return true.
196   // See 8.10 of JISX0510:2004 (p.47) for how to embed version information.
197   static void maybeEmbedVersionInfo(Version version, ByteMatrix matrix) throws WriterException {
198     if (version.getVersionNumber() < 7) {  // Version info is necessary if version >= 7.
199       return;  // Don't need version info.
200     }
201     BitArray versionInfoBits = new BitArray();
202     makeVersionInfoBits(version, versionInfoBits);
203 
204     int bitIndex = 6 * 3 - 1;  // It will decrease from 17 to 0.
205     for (int i = 0; i < 6; ++i) {
206       for (int j = 0; j < 3; ++j) {
207         // Place bits in LSB (least significant bit) to MSB order.
208         boolean bit = versionInfoBits.get(bitIndex);
209         bitIndex--;
210         // Left bottom corner.
211         matrix.set(i, matrix.getHeight() - 11 + j, bit);
212         // Right bottom corner.
213         matrix.set(matrix.getHeight() - 11 + j, i, bit);
214       }
215     }
216   }

218   // Embed "dataBits" using "getMaskPattern". On success, modify the matrix and return true.
219   // For debugging purposes, it skips masking process if "getMaskPattern" is -1.
220   // See 8.7 of JISX0510:2004 (p.38) for how to embed data bits.
221   static void embedDataBits(BitArray dataBits, int maskPattern, ByteMatrix matrix)
222       throws WriterException {
223     int bitIndex = 0;
224     int direction = -1;
225     // Start from the right bottom cell.
226     int x = matrix.getWidth() - 1;
227     int y = matrix.getHeight() - 1;
228     while (x > 0) {
229       // Skip the vertical timing pattern.
230       if (x == 6) {
231         x -= 1;
232       }
233       while (y >= 0 && y < matrix.getHeight()) {
234         for (int i = 0; i < 2; ++i) {
235           int xx = x - i;
236           // Skip the cell if it's not empty.
237           if (!isEmpty(matrix.get(xx, y))) {
238             continue;
239           }
240           boolean bit;
241           if (bitIndex < dataBits.getSize()) {
242             bit = dataBits.get(bitIndex);
243             ++bitIndex;
244           } else {
245             // Padding bit. If there is no bit left, we'll fill the left cells with 0, as described
246             // in 8.4.9 of JISX0510:2004 (p. 24).
247             bit = false;
248           }
249 
250           // Skip masking if mask_pattern is -1.
251           if (maskPattern != -1 && MaskUtil.getDataMaskBit(maskPattern, xx, y)) {
252             bit = !bit;
253           }
254           matrix.set(xx, y, bit);
255         }
256         y += direction;
257       }
258       direction = -direction;  // Reverse the direction.
259       y += direction;
260       x -= 2;  // Move to the left.
261     }
262     // All bits should be consumed.
263     if (bitIndex != dataBits.getSize()) {
264       throw new WriterException("Not all bits consumed: " + bitIndex + '/' + dataBits.getSize());
265     }
266   }

268   // Return the position of the most significant bit set (to one) in the "value". The most
269   // significant bit is position 32. If there is no bit set, return 0. Examples:
270   // - findMSBSet(0) => 0
271   // - findMSBSet(1) => 1
272   // - findMSBSet(255) => 8
273   static int findMSBSet(int value) {
274     int numDigits = 0;
275     while (value != 0) {
276       value >>>= 1;
277       ++numDigits;
278     }
279     return numDigits;
280   }

282   // Calculate BCH (Bose-Chaudhuri-Hocquenghem) code for "value" using polynomial "poly". The BCH
283   // code is used for encoding type information and version information.
284   // Example: Calculation of version information of 7.
285   // f(x) is created from 7.
286   //   - 7 = 000111 in 6 bits
287   //   - f(x) = x^2 + x^1 + x^0
288   // g(x) is given by the standard (p. 67)
289   //   - g(x) = x^12 + x^11 + x^10 + x^9 + x^8 + x^5 + x^2 + 1
290   // Multiply f(x) by x^(18 - 6)
291   //   - f'(x) = f(x) * x^(18 - 6)
292   //   - f'(x) = x^14 + x^13 + x^12
293   // Calculate the remainder of f'(x) / g(x)
294   //         x^2
295   //         __________________________________________________
296   //   g(x) )x^14 + x^13 + x^12
297   //         x^14 + x^13 + x^12 + x^11 + x^10 + x^7 + x^4 + x^2
298   //         --------------------------------------------------
299   //                              x^11 + x^10 + x^7 + x^4 + x^2
300   //
301   // The remainder is x^11 + x^10 + x^7 + x^4 + x^2
302   // Encode it in binary: 110010010100
303   // The return value is 0xc94 (1100 1001 0100)
304   //
305   // Since all coefficients in the polynomials are 1 or 0, we can do the calculation by bit
306   // operations. We don't care if cofficients are positive or negative.
307   static int calculateBCHCode(int value, int poly) {
308     // If poly is "1 1111 0010 0101" (version info poly), msbSetInPoly is 13. We'll subtract 1
309     // from 13 to make it 12.
310     int msbSetInPoly = findMSBSet(poly);
311     value <<= msbSetInPoly - 1;
312     // Do the division business using exclusive-or operations.
313     while (findMSBSet(value) >= msbSetInPoly) {
314       value ^= poly << (findMSBSet(value) - msbSetInPoly);
315     }
316     // Now the "value" is the remainder (i.e. the BCH code)
317     return value;
318   }

320   // Make bit vector of type information. On success, store the result in "bits" and return true.
321   // Encode error correction level and mask pattern. See 8.9 of
322   // JISX0510:2004 (p.45) for details.
323   static void makeTypeInfoBits(ErrorCorrectionLevel ecLevel, int maskPattern, BitArray bits)
324       throws WriterException {
325     if (!QRCode.isValidMaskPattern(maskPattern)) {
326       throw new WriterException("Invalid mask pattern");
327     }
328     int typeInfo = (ecLevel.getBits() << 3) | maskPattern;
329     bits.appendBits(typeInfo, 5);
330 
331     int bchCode = calculateBCHCode(typeInfo, TYPE_INFO_POLY);
332     bits.appendBits(bchCode, 10);
333 
334     BitArray maskBits = new BitArray();
335     maskBits.appendBits(TYPE_INFO_MASK_PATTERN, 15);
336     bits.xor(maskBits);
337 
338     if (bits.getSize() != 15) {  // Just in case.
339       throw new WriterException("should not happen but we got: " + bits.getSize());
340     }
341   }

343   // Make bit vector of version information. On success, store the result in "bits" and return true.
344   // See 8.10 of JISX0510:2004 (p.45) for details.
345   static void makeVersionInfoBits(Version version, BitArray bits) throws WriterException {
346     bits.appendBits(version.getVersionNumber(), 6);
347     int bchCode = calculateBCHCode(version.getVersionNumber(), VERSION_INFO_POLY);
348     bits.appendBits(bchCode, 12);
349 
350     if (bits.getSize() != 18) {  // Just in case.
351       throw new WriterException("should not happen but we got: " + bits.getSize());
352     }
353   }

360   private static void embedTimingPatterns(ByteMatrix matrix) {
361     // -8 is for skipping position detection patterns (size 7), and two horizontal/vertical
362     // separation patterns (size 1). Thus, 8 = 7 + 1.
363     for (int i = 8; i < matrix.getWidth() - 8; ++i) {
364       int bit = (i + 1) % 2;
365       // Horizontal line.
366       if (isEmpty(matrix.get(i, 6))) {
367         matrix.set(i, 6, bit);
368       }
369       // Vertical line.
370       if (isEmpty(matrix.get(6, i))) {
371         matrix.set(6, i, bit);
372       }
373     }
374   }

376   // Embed the lonely dark dot at left bottom corner. JISX0510:2004 (p.46)
377   private static void embedDarkDotAtLeftBottomCorner(ByteMatrix matrix) throws WriterException {
378     if (matrix.get(8, matrix.getHeight() - 8) == 0) {
379       throw new WriterException();
380     }
381     matrix.set(8, matrix.getHeight() - 8, 1);
382   }

384   private static void embedHorizontalSeparationPattern(int xStart,
385                                                        int yStart,
386                                                        ByteMatrix matrix) throws WriterException {
387     for (int x = 0; x < 8; ++x) {
388       if (!isEmpty(matrix.get(xStart + x, yStart))) {
389         throw new WriterException();
390       }
391       matrix.set(xStart + x, yStart, 0);
392     }
393   }

395   private static void embedVerticalSeparationPattern(int xStart,
396                                                      int yStart,
397                                                      ByteMatrix matrix) throws WriterException {
398     for (int y = 0; y < 7; ++y) {
399       if (!isEmpty(matrix.get(xStart, yStart + y))) {
400         throw new WriterException();
401       }
402       matrix.set(xStart, yStart + y, 0);
403     }
404   }

406   // Note that we cannot unify the function with embedPositionDetectionPattern() despite they are
407   // almost identical, since we cannot write a function that takes 2D arrays in different sizes in

409   private static void embedPositionAdjustmentPattern(int xStart, int yStart, ByteMatrix matrix) {
410     for (int y = 0; y < 5; ++y) {
411       for (int x = 0; x < 5; ++x) {
412         matrix.set(xStart + x, yStart + y, POSITION_ADJUSTMENT_PATTERN[y][x]);
413       }
414     }
415   }

418     for (int y = 0; y < 7; ++y) {

420         matrix.set(xStart + x, yStart + y, POSITION_DETECTION_PATTERN[y][x]);

425   // Embed position detection patterns and surrounding vertical/horizontal separators.
426   private static void embedPositionDetectionPatternsAndSeparators(ByteMatrix matrix) throws WriterException {
427     // Embed three big squares at corners.
428     int pdpWidth = POSITION_DETECTION_PATTERN[0].length;
429     // Left top corner.
430     embedPositionDetectionPattern(0, 0, matrix);
431     // Right top corner.
432     embedPositionDetectionPattern(matrix.getWidth() - pdpWidth, 0, matrix);
433     // Left bottom corner.
434     embedPositionDetectionPattern(0, matrix.getWidth() - pdpWidth, matrix);
435 
436     // Embed horizontal separation patterns around the squares.
437     int hspWidth = 8;
438     // Left top corner.
439     embedHorizontalSeparationPattern(0, hspWidth - 1, matrix);
440     // Right top corner.
441     embedHorizontalSeparationPattern(matrix.getWidth() - hspWidth,
442         hspWidth - 1, matrix);
443     // Left bottom corner.
444     embedHorizontalSeparationPattern(0, matrix.getWidth() - hspWidth, matrix);
445 
446     // Embed vertical separation patterns around the squares.
447     int vspSize = 7;
448     // Left top corner.
449     embedVerticalSeparationPattern(vspSize, 0, matrix);
450     // Right top corner.
451     embedVerticalSeparationPattern(matrix.getHeight() - vspSize - 1, 0, matrix);
452     // Left bottom corner.
453     embedVerticalSeparationPattern(vspSize, matrix.getHeight() - vspSize,
454         matrix);
455   }

457   // Embed position adjustment patterns if need be.
458   private static void maybeEmbedPositionAdjustmentPatterns(Version version, ByteMatrix matrix) {
459     if (version.getVersionNumber() < 2) {  // The patterns appear if version >= 2
460       return;
461     }
462     int index = version.getVersionNumber() - 1;
463     int[] coordinates = POSITION_ADJUSTMENT_PATTERN_COORDINATE_TABLE[index];
464     int numCoordinates = POSITION_ADJUSTMENT_PATTERN_COORDINATE_TABLE[index].length;
465     for (int i = 0; i < numCoordinates; ++i) {
466       for (int j = 0; j < numCoordinates; ++j) {
467         int y = coordinates[i];
468         int x = coordinates[j];
469         if (x == -1 || y == -1) {
470           continue;
471         }
472         // If the cell is unset, we embed the position adjustment pattern here.
473         if (isEmpty(matrix.get(x, y))) {
474           // -2 is necessary since the x/y coordinates point to the center of the pattern, not the
475           // left top corner.
476           embedPositionAdjustmentPattern(x - 2, y - 2, matrix);
477         }
478       }
479     }
