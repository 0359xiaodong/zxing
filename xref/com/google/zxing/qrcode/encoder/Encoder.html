55    // The mask penalty calculation is complicated.  See Table 21 of JISX0510:2004 (p.45) for details.
56    // Basically it applies four rules and summate all penalties.
57    private static int calculateMaskPenalty(ByteMatrix matrix) {
58      return MaskUtil.applyMaskPenaltyRule1(matrix)
59          + MaskUtil.applyMaskPenaltyRule2(matrix)
60          + MaskUtil.applyMaskPenaltyRule3(matrix)
61          + MaskUtil.applyMaskPenaltyRule4(matrix);
62    }

64    /**
65     *  Encode "bytes" with the error correction level "ecLevel". The encoding mode will be chosen
66     * internally by chooseMode(). On success, store the result in "qrCode".
67     *
68     * We recommend you to use QRCode.EC_LEVEL_L (the lowest level) for
69     * "getECLevel" since our primary use is to show QR code on desktop screens. We don't need very
70     * strong error correction for this purpose.
71     *
72     * Note that there is no way to encode bytes in MODE_KANJI. We might want to add EncodeWithMode()
73     * with which clients can specify the encoding mode. For now, we don't need the functionality.
74     */
75    public static QRCode encode(String content, ErrorCorrectionLevel ecLevel) throws WriterException {
76      return encode(content, ecLevel, null);
77    }

79    public static QRCode encode(String content,
80                                ErrorCorrectionLevel ecLevel,
81                                Map<EncodeHintType,?> hints) throws WriterException {
82  
83      // Determine what character encoding has been specified by the caller, if any
84      String encoding = hints == null ? null : (String) hints.get(EncodeHintType.CHARACTER_SET);
85      if (encoding == null) {
86        encoding = DEFAULT_BYTE_MODE_ENCODING;
87      }
88  
89      // Pick an encoding mode appropriate for the content. Note that this will not attempt to use
90      // multiple modes / segments even if that were more efficient. Twould be nice.
91      Mode mode = chooseMode(content, encoding);
92  
93      // This will store the header information, like mode and
94      // length, as well as "header" segments like an ECI segment.
95      BitArray headerBits = new BitArray();
96  
97      // Append ECI segment if applicable
98      if (mode == Mode.BYTE && !DEFAULT_BYTE_MODE_ENCODING.equals(encoding)) {
99        CharacterSetECI eci = CharacterSetECI.getCharacterSetECIByName(encoding);
100       if (eci != null) {
101         appendECI(eci, headerBits);
102       }
103     }
104 
105     // (With ECI in place,) Write the mode marker
106     appendModeInfo(mode, headerBits);
107 
108     // Collect data within the main segment, separately, to count its size if needed. Don't add it to
109     // main payload yet.
110     BitArray dataBits = new BitArray();
111     appendBytes(content, mode, dataBits, encoding);
112 
113     // Hard part: need to know version to know how many bits length takes. But need to know how many
114     // bits it takes to know version. First we take a guess at version by assuming version will be
115     // the minimum, 1:
116 
117     int provisionalBitsNeeded = headerBits.getSize()
118         + mode.getCharacterCountBits(Version.getVersionForNumber(1))
119         + dataBits.getSize();
120     Version provisionalVersion = chooseVersion(provisionalBitsNeeded, ecLevel);
121 
122     // Use that guess to calculate the right version. I am still not sure this works in 100% of cases.
123 
124     int bitsNeeded = headerBits.getSize()
125         + mode.getCharacterCountBits(provisionalVersion)
126         + dataBits.getSize();
127     Version version = chooseVersion(bitsNeeded, ecLevel);
128 
129     BitArray headerAndDataBits = new BitArray();
130     headerAndDataBits.appendBitArray(headerBits);
131     // Find "length" of main segment and write it
132     int numLetters = mode == Mode.BYTE ? dataBits.getSizeInBytes() : content.length();
133     appendLengthInfo(numLetters, version, mode, headerAndDataBits);
134     // Put data together into the overall payload
135     headerAndDataBits.appendBitArray(dataBits);
136 
137     Version.ECBlocks ecBlocks = version.getECBlocksForLevel(ecLevel);
138     int numDataBytes = version.getTotalCodewords() - ecBlocks.getTotalECCodewords();
139 
140     // Terminate the bits properly.
141     terminateBits(numDataBytes, headerAndDataBits);
142 
143     // Interleave data bits with error correction code.
144     BitArray finalBits = interleaveWithECBytes(headerAndDataBits,
145                                                version.getTotalCodewords(),
146                                                numDataBytes,
147                                                ecBlocks.getNumBlocks());
148 
149     QRCode qrCode = new QRCode();
150 
151     qrCode.setECLevel(ecLevel);
152     qrCode.setMode(mode);
153     qrCode.setVersion(version);
154 
155     //  Choose the mask pattern and set to "qrCode".
156     int dimension = version.getDimensionForVersion();
157     ByteMatrix matrix = new ByteMatrix(dimension, dimension);
158     int maskPattern = chooseMaskPattern(finalBits, ecLevel, version, matrix);
159     qrCode.setMaskPattern(maskPattern);
160 
161     // Build the matrix and set it to "qrCode".
162     MatrixUtil.buildMatrix(finalBits, ecLevel, version, maskPattern, matrix);
163     qrCode.setMatrix(matrix);
164 
165     return qrCode;
166   }

169    * @return the code point of the table used in alphanumeric mode or
170    *  -1 if there is no corresponding code in the table.
171    */
172   static int getAlphanumericCode(int code) {
173     if (code < ALPHANUMERIC_TABLE.length) {
174       return ALPHANUMERIC_TABLE[code];
175     }
176     return -1;
177   }

183   /**
184    * Choose the best mode by examining the content. Note that 'encoding' is used as a hint;
185    * if it is Shift_JIS, and the input is only double-byte Kanji, then we return {@link Mode#KANJI}.
186    */
187   private static Mode chooseMode(String content, String encoding) {
188     if ("Shift_JIS".equals(encoding)) {
189       // Choose Kanji mode if all input are double-byte characters
190       return isOnlyDoubleByteKanji(content) ? Mode.KANJI : Mode.BYTE;
191     }
192     boolean hasNumeric = false;
193     boolean hasAlphanumeric = false;
194     for (int i = 0; i < content.length(); ++i) {
195       char c = content.charAt(i);
196       if (c >= '0' && c <= '9') {
197         hasNumeric = true;
198       } else if (getAlphanumericCode(c) != -1) {
199         hasAlphanumeric = true;
200       } else {
201         return Mode.BYTE;
202       }
203     }
204     if (hasAlphanumeric) {
205       return Mode.ALPHANUMERIC;
206     }
207     if (hasNumeric) {
208       return Mode.NUMERIC;
209     }
210     return Mode.BYTE;
211   }

213   private static boolean isOnlyDoubleByteKanji(String content) {
214     byte[] bytes;
215     try {
216       bytes = content.getBytes("Shift_JIS");
217     } catch (UnsupportedEncodingException ignored) {
218       return false;
219     }
220     int length = bytes.length;
221     if (length % 2 != 0) {
222       return false;
223     }
224     for (int i = 0; i < length; i += 2) {
225       int byte1 = bytes[i] & 0xFF;
226       if ((byte1 < 0x81 || byte1 > 0x9F) && (byte1 < 0xE0 || byte1 > 0xEB)) {
227         return false;
228       }
229     }
230     return true;
231   }

233   private static int chooseMaskPattern(BitArray bits,
234                                        ErrorCorrectionLevel ecLevel,
235                                        Version version,
236                                        ByteMatrix matrix) throws WriterException {
237 
238     int minPenalty = Integer.MAX_VALUE;  // Lower penalty is better.
239     int bestMaskPattern = -1;
240     // We try all mask patterns to choose the best one.
241     for (int maskPattern = 0; maskPattern < QRCode.NUM_MASK_PATTERNS; maskPattern++) {
242       MatrixUtil.buildMatrix(bits, ecLevel, version, maskPattern, matrix);
243       int penalty = calculateMaskPenalty(matrix);
244       if (penalty < minPenalty) {
245         minPenalty = penalty;
246         bestMaskPattern = maskPattern;
247       }
248     }
249     return bestMaskPattern;
250   }

252   private static Version chooseVersion(int numInputBits, ErrorCorrectionLevel ecLevel) throws WriterException {
253     // In the following comments, we use numbers of Version 7-H.
254     for (int versionNum = 1; versionNum <= 40; versionNum++) {
255       Version version = Version.getVersionForNumber(versionNum);
256       // numBytes = 196
257       int numBytes = version.getTotalCodewords();
258       // getNumECBytes = 130
259       Version.ECBlocks ecBlocks = version.getECBlocksForLevel(ecLevel);
260       int numEcBytes = ecBlocks.getTotalECCodewords();
261       // getNumDataBytes = 196 - 130 = 66
262       int numDataBytes = numBytes - numEcBytes;
263       int totalInputBytes = (numInputBits + 7) / 8;
264       if (numDataBytes >= totalInputBytes) {
265         return version;
266       }
267     }
268     throw new WriterException("Data too big");
269   }

271   /**
272    * Terminate bits as described in 8.4.8 and 8.4.9 of JISX0510:2004 (p.24).
273    */
274   static void terminateBits(int numDataBytes, BitArray bits) throws WriterException {
275     int capacity = numDataBytes << 3;
276     if (bits.getSize() > capacity) {
277       throw new WriterException("data bits cannot fit in the QR Code" + bits.getSize() + " > " +
278           capacity);
279     }
280     for (int i = 0; i < 4 && bits.getSize() < capacity; ++i) {
281       bits.appendBit(false);
282     }
283     // Append termination bits. See 8.4.8 of JISX0510:2004 (p.24) for details.
284     // If the last byte isn't 8-bit aligned, we'll add padding bits.
285     int numBitsInLastByte = bits.getSize() & 0x07;    
286     if (numBitsInLastByte > 0) {
287       for (int i = numBitsInLastByte; i < 8; i++) {
288         bits.appendBit(false);
289       }
290     }
291     // If we have more space, we'll fill the space with padding patterns defined in 8.4.9 (p.24).
292     int numPaddingBytes = numDataBytes - bits.getSizeInBytes();
293     for (int i = 0; i < numPaddingBytes; ++i) {
294       bits.appendBits((i & 0x01) == 0 ? 0xEC : 0x11, 8);
295     }
296     if (bits.getSize() != capacity) {
297       throw new WriterException("Bits size does not equal capacity");
298     }
299   }

301   /**
302    * Get number of data bytes and number of error correction bytes for block id "blockID". Store
303    * the result in "numDataBytesInBlock", and "numECBytesInBlock". See table 12 in 8.5.1 of
304    * JISX0510:2004 (p.30)
305    */
306   static void getNumDataBytesAndNumECBytesForBlockID(int numTotalBytes,
307                                                      int numDataBytes,
308                                                      int numRSBlocks,
309                                                      int blockID,
310                                                      int[] numDataBytesInBlock,
311                                                      int[] numECBytesInBlock) throws WriterException {
312     if (blockID >= numRSBlocks) {
313       throw new WriterException("Block ID too large");
314     }
315     // numRsBlocksInGroup2 = 196 % 5 = 1
316     int numRsBlocksInGroup2 = numTotalBytes % numRSBlocks;
317     // numRsBlocksInGroup1 = 5 - 1 = 4
318     int numRsBlocksInGroup1 = numRSBlocks - numRsBlocksInGroup2;
319     // numTotalBytesInGroup1 = 196 / 5 = 39
320     int numTotalBytesInGroup1 = numTotalBytes / numRSBlocks;
321     // numTotalBytesInGroup2 = 39 + 1 = 40
322     int numTotalBytesInGroup2 = numTotalBytesInGroup1 + 1;
323     // numDataBytesInGroup1 = 66 / 5 = 13
324     int numDataBytesInGroup1 = numDataBytes / numRSBlocks;
325     // numDataBytesInGroup2 = 13 + 1 = 14
326     int numDataBytesInGroup2 = numDataBytesInGroup1 + 1;
327     // numEcBytesInGroup1 = 39 - 13 = 26
328     int numEcBytesInGroup1 = numTotalBytesInGroup1 - numDataBytesInGroup1;
329     // numEcBytesInGroup2 = 40 - 14 = 26
330     int numEcBytesInGroup2 = numTotalBytesInGroup2 - numDataBytesInGroup2;
331     // Sanity checks.
332     // 26 = 26
333     if (numEcBytesInGroup1 != numEcBytesInGroup2) {
334       throw new WriterException("EC bytes mismatch");
335     }
336     // 5 = 4 + 1.
337     if (numRSBlocks != numRsBlocksInGroup1 + numRsBlocksInGroup2) {
338       throw new WriterException("RS blocks mismatch");
339     }
340     // 196 = (13 + 26) * 4 + (14 + 26) * 1
341     if (numTotalBytes !=
342         ((numDataBytesInGroup1 + numEcBytesInGroup1) *
343             numRsBlocksInGroup1) +
344             ((numDataBytesInGroup2 + numEcBytesInGroup2) *
345                 numRsBlocksInGroup2)) {
346       throw new WriterException("Total bytes mismatch");
347     }
348 
349     if (blockID < numRsBlocksInGroup1) {
350       numDataBytesInBlock[0] = numDataBytesInGroup1;
351       numECBytesInBlock[0] = numEcBytesInGroup1;
352     } else {
353       numDataBytesInBlock[0] = numDataBytesInGroup2;
354       numECBytesInBlock[0] = numEcBytesInGroup2;
355     }
356   }

358   /**
359    * Interleave "bits" with corresponding error correction bytes. On success, store the result in
360    * "result". The interleave rule is complicated. See 8.6 of JISX0510:2004 (p.37) for details.
361    */
362   static BitArray interleaveWithECBytes(BitArray bits,
363                                         int numTotalBytes,
364                                         int numDataBytes,
365                                         int numRSBlocks) throws WriterException {
366 
367     // "bits" must have "getNumDataBytes" bytes of data.
368     if (bits.getSizeInBytes() != numDataBytes) {
369       throw new WriterException("Number of bits and data bytes does not match");
370     }
371 
372     // Step 1.  Divide data bytes into blocks and generate error correction bytes for them. We'll
373     // store the divided data bytes blocks and error correction bytes blocks into "blocks".
374     int dataBytesOffset = 0;
375     int maxNumDataBytes = 0;
376     int maxNumEcBytes = 0;
377 
378     // Since, we know the number of reedsolmon blocks, we can initialize the vector with the number.
379     Collection<BlockPair> blocks = new ArrayList<>(numRSBlocks);
380 
381     for (int i = 0; i < numRSBlocks; ++i) {
382       int[] numDataBytesInBlock = new int[1];
383       int[] numEcBytesInBlock = new int[1];
384       getNumDataBytesAndNumECBytesForBlockID(
385           numTotalBytes, numDataBytes, numRSBlocks, i,
386           numDataBytesInBlock, numEcBytesInBlock);
387 
388       int size = numDataBytesInBlock[0];
389       byte[] dataBytes = new byte[size];
390       bits.toBytes(8*dataBytesOffset, dataBytes, 0, size);
391       byte[] ecBytes = generateECBytes(dataBytes, numEcBytesInBlock[0]);
392       blocks.add(new BlockPair(dataBytes, ecBytes));
393 
394       maxNumDataBytes = Math.max(maxNumDataBytes, size);
395       maxNumEcBytes = Math.max(maxNumEcBytes, ecBytes.length);
396       dataBytesOffset += numDataBytesInBlock[0];
397     }
398     if (numDataBytes != dataBytesOffset) {
399       throw new WriterException("Data bytes does not match offset");
400     }
401 
402     BitArray result = new BitArray();
403 
404     // First, place data blocks.
405     for (int i = 0; i < maxNumDataBytes; ++i) {
406       for (BlockPair block : blocks) {
407         byte[] dataBytes = block.getDataBytes();
408         if (i < dataBytes.length) {
409           result.appendBits(dataBytes[i], 8);
410         }
411       }
412     }
413     // Then, place error correction blocks.
414     for (int i = 0; i < maxNumEcBytes; ++i) {
415       for (BlockPair block : blocks) {
416         byte[] ecBytes = block.getErrorCorrectionBytes();
417         if (i < ecBytes.length) {
418           result.appendBits(ecBytes[i], 8);
419         }
420       }
421     }
422     if (numTotalBytes != result.getSizeInBytes()) {  // Should be same.
423       throw new WriterException("Interleaving error: " + numTotalBytes + " and " +
424           result.getSizeInBytes() + " differ.");
425     }
426 
427     return result;
428   }

430   static byte[] generateECBytes(byte[] dataBytes, int numEcBytesInBlock) {
431     int numDataBytes = dataBytes.length;
432     int[] toEncode = new int[numDataBytes + numEcBytesInBlock];
433     for (int i = 0; i < numDataBytes; i++) {
434       toEncode[i] = dataBytes[i] & 0xFF;
435     }
436     new ReedSolomonEncoder(GenericGF.QR_CODE_FIELD_256).encode(toEncode, numEcBytesInBlock);
437 
438     byte[] ecBytes = new byte[numEcBytesInBlock];
439     for (int i = 0; i < numEcBytesInBlock; i++) {
440       ecBytes[i] = (byte) toEncode[numDataBytes + i];
441     }
442     return ecBytes;
443   }

453   /**
454    * Append length info. On success, store the result in "bits".
455    */
456   static void appendLengthInfo(int numLetters, Version version, Mode mode, BitArray bits) throws WriterException {
457     int numBits = mode.getCharacterCountBits(version);
458     if (numLetters >= (1 << numBits)) {
459       throw new WriterException(numLetters + " is bigger than " + ((1 << numBits) - 1));
460     }
461     bits.appendBits(numLetters, numBits);
462   }

464   /**
465    * Append "bytes" in "mode" mode (encoding) into "bits". On success, store the result in "bits".
466    */
467   static void appendBytes(String content,
468                           Mode mode,
469                           BitArray bits,
470                           String encoding) throws WriterException {
471     switch (mode) {
472       case NUMERIC:
473         appendNumericBytes(content, bits);
474         break;
475       case ALPHANUMERIC:
476         appendAlphanumericBytes(content, bits);
477         break;
478       case BYTE:
479         append8BitBytes(content, bits, encoding);
480         break;
481       case KANJI:
482         appendKanjiBytes(content, bits);
483         break;
484       default:
485         throw new WriterException("Invalid mode: " + mode);
486     }
487   }

489   static void appendNumericBytes(CharSequence content, BitArray bits) {
490     int length = content.length();
491     int i = 0;
492     while (i < length) {
493       int num1 = content.charAt(i) - '0';
494       if (i + 2 < length) {
495         // Encode three numeric letters in ten bits.
496         int num2 = content.charAt(i + 1) - '0';
497         int num3 = content.charAt(i + 2) - '0';
498         bits.appendBits(num1 * 100 + num2 * 10 + num3, 10);
499         i += 3;
500       } else if (i + 1 < length) {
501         // Encode two numeric letters in seven bits.
502         int num2 = content.charAt(i + 1) - '0';
503         bits.appendBits(num1 * 10 + num2, 7);
504         i += 2;
505       } else {
506         // Encode one numeric letter in four bits.
507         bits.appendBits(num1, 4);
508         i++;
509       }
510     }
511   }
512 
513   static void appendAlphanumericBytes(CharSequence content, BitArray bits) throws WriterException {
514     int length = content.length();
515     int i = 0;
516     while (i < length) {
517       int code1 = getAlphanumericCode(content.charAt(i));
518       if (code1 == -1) {
519         throw new WriterException();
520       }
521       if (i + 1 < length) {
522         int code2 = getAlphanumericCode(content.charAt(i + 1));
523         if (code2 == -1) {
524           throw new WriterException();
525         }
526         // Encode two alphanumeric letters in 11 bits.
527         bits.appendBits(code1 * 45 + code2, 11);
528         i += 2;
529       } else {
530         // Encode one alphanumeric letter in six bits.
531         bits.appendBits(code1, 6);
532         i++;
533       }
534     }
535   }

537   static void append8BitBytes(String content, BitArray bits, String encoding)
538       throws WriterException {
539     byte[] bytes;
540     try {
541       bytes = content.getBytes(encoding);
542     } catch (UnsupportedEncodingException uee) {
543       throw new WriterException(uee);
544     }
545     for (byte b : bytes) {
546       bits.appendBits(b, 8);
547     }
548   }

550   static void appendKanjiBytes(String content, BitArray bits) throws WriterException {
551     byte[] bytes;
552     try {
553       bytes = content.getBytes("Shift_JIS");
554     } catch (UnsupportedEncodingException uee) {
555       throw new WriterException(uee);
556     }
557     int length = bytes.length;
558     for (int i = 0; i < length; i += 2) {
559       int byte1 = bytes[i] & 0xFF;
560       int byte2 = bytes[i + 1] & 0xFF;
561       int code = (byte1 << 8) | byte2;
562       int subtracted = -1;
563       if (code >= 0x8140 && code <= 0x9ffc) {
564         subtracted = code - 0x8140;
565       } else if (code >= 0xe040 && code <= 0xebbf) {
566         subtracted = code - 0xc140;
567       }
568       if (subtracted == -1) {
569         throw new WriterException("Invalid byte sequence");
570       }
571       int encoded = ((subtracted >> 8) * 0xc0) + (subtracted & 0xff);
572       bits.appendBits(encoded, 13);
573     }
574   }

576   private static void appendECI(CharacterSetECI eci, BitArray bits) {
577     bits.appendBits(Mode.ECI.getBits(), 4);
578     // This is correct for values up to 127, which is all we need now.
579     bits.appendBits(eci.getValue(), 8);
580   }
