17  package com.google.zxing.web.generator.client;
18  
19  /**
20   * Helpers methods to check for phone numbers, email addresses, and URL. Other
21   * general purpose check methods should go here as well.
22   * 
23   * @author Yohann Coppel
24   */
25  final class Validators {
26  
27    private Validators() {
28    }
29  
30    static String filterNumber(String number) {
31      return number.replaceAll("[ \\.,\\-\\(\\)]", "");
32    }
33    
34    static void validateNumber(String number) throws GeneratorException {
35      if (!number.matches("\\+?[0-9]+")) {
36        throw new GeneratorException("Phone number must be digits only.");
37      }
38    }
39    
40    static void validateUrl(String url) throws GeneratorException {
41      if (!isBasicallyValidURI(url)) {
42        throw new GeneratorException("URL is not valid.");
43      }
44    }
45  
46    private static boolean isBasicallyValidURI(String uri) {
47      if (uri == null || uri.indexOf(' ') >= 0 || uri.indexOf('\n') >= 0) {
48        return false;
49      }
50      int period = uri.indexOf('.');
51      // Look for period in a domain but followed by at least a two-char TLD
52      return period < uri.length() - 2 && (period >= 0 || uri.indexOf(':') >= 0);
53    }
54    
55    static void validateEmail(String email) throws GeneratorException {
56      //FIXME: we can have a better check for email here.
57      if (!email.matches("^[A-Za-z0-9._%+-]+@[A-Za-z0-9.-]+\\.[A-Za-z]{2,6}$")) {
58        throw new GeneratorException("Email is not valid.");
59      }
60    }
61  
62  }
