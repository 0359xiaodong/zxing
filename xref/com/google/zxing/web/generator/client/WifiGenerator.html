17  package com.google.zxing.web.generator.client;
18  
19  import com.google.gwt.event.dom.client.ChangeHandler;
20  import com.google.gwt.event.dom.client.KeyPressHandler;
21  import com.google.gwt.user.client.ui.CheckBox;
22  import com.google.gwt.user.client.ui.Grid;
23  import com.google.gwt.user.client.ui.HasText;
24  import com.google.gwt.user.client.ui.TextBox;
25  import com.google.gwt.user.client.ui.ListBox;
26  import com.google.gwt.user.client.ui.Widget;
27  
28  /**
29   * A Generator for Wifi networks.
30   * 
31   * @author Vikram Aggarwal
32   * @author Sean Owen
33   */
34  public final class WifiGenerator implements GeneratorSource {
35  
36    private Grid table;
37    private final TextBox ssid = new TextBox();
38    private final TextBox password = new TextBox();
39    private final ListBox networkType = new ListBox(false);
40    private final CheckBox hidden = new CheckBox();
41  
42    public WifiGenerator(ChangeHandler handler, KeyPressHandler keyListener) {
43      networkType.addItem("WEP", "WEP");
44      networkType.addItem("WPA/WPA2", "WPA");
45      networkType.addItem("No encryption", "nopass");
46      TextBox[] widgets = {ssid, password};
47      for (TextBox w: widgets) {
48        w.addChangeHandler(handler);
49        w.addKeyPressHandler(keyListener);
50      }
51    }
52    
53    @Override
54    public String getName() {
55      return "Wifi network";
56    }
57  
58    @Override
59    public String getText() throws GeneratorException {
60      String ssid = getSsidField();
61      String password = getPasswordField();
62      String networkType = getNetworkTypeField();
63      boolean hidden = getHiddenField();
64      // Build the output with obtained data.
65      return getWifiString(ssid, password, networkType, hidden);
66    }
67  
68    private static String getWifiString(String ssid, String password, String type, boolean hidden) {
69      StringBuilder output = new StringBuilder(100);
70      output.append("WIFI:");
71      output.append("S:").append(ssid).append(';');
72      if (type != null && !type.isEmpty() && !"nopass".equals(type)) {
73        maybeAppend(output, "T:", type);
74      }
75      maybeAppend(output, "P:", password);
76      if (hidden) {
77        maybeAppend(output, "H:", "true");
78      }
79      output.append(';');
80      return output.toString();
81    }
82  
83    private static void maybeAppend(StringBuilder output, String prefix, String value) {
84      if (value != null && !value.isEmpty()) {
85        output.append(prefix).append(value).append(';');
86      }
87    }
88  
89    private static String parseTextField(String name, HasText textBox) throws GeneratorException {
90      String input = textBox.getText();
91      if (input.isEmpty()) {
92        return "";
93      }
94      if (input.contains("\n")) {
95        throw new GeneratorException(name + " field must not contain \\n characters.");
96      }
97      return input.replaceAll("([\\\\:;])", "\\\\$1");
98    }
99    
100   private String getSsidField() throws GeneratorException {
101     String input = ssid.getText();
102     if (input.isEmpty()) {
103       throw new GeneratorException("SSID must be at least 1 character.");
104     }
105     String parsed = parseTextField("SSID", ssid);
106     return quoteHex(parsed); // Android needs hex-like SSIDs quoted or will be read as hex
107   }
108   
109   private String getPasswordField() throws GeneratorException {
110   return parseTextField("Password", password);
111   }
112   
113   private String getNetworkTypeField() {
114     return networkType.getValue(networkType.getSelectedIndex());
115   }
116 
117   private boolean getHiddenField() {
118     Boolean value = hidden.getValue();
119     return value != null && value;
120   }
121   
122   @Override
123   public Grid getWidget() {
124     if (table != null) {
125       // early termination if the table has already been constructed
126       return table;
127     }
128     table = new Grid(4, 2);
129     
130     table.setText(0, 0, "SSID");
131     table.setWidget(0, 1, ssid);
132     table.setText(1, 0, "Password");
133     table.setWidget(1, 1, password);
134     table.setText(2, 0, "Network Type");
135     table.setWidget(2, 1, networkType);
136     table.setText(3, 0, "Hidden?");
137     table.setWidget(3, 1, hidden);
138 
139     ssid.addStyleName(StylesDefs.INPUT_FIELD_REQUIRED);
140     return table;
141   }
142 
143   @Override
144   public void validate(Widget widget) throws GeneratorException {
145     if (widget == ssid) {
146       getSsidField();
147     }
148     if (widget == password) {
149       getPasswordField();
150     }
151     if (widget == networkType) {
152       getNetworkTypeField();
153     }
154     if (widget == hidden) {
155       getHiddenField();
156     }
157   }
158 
159   @Override
160   public void setFocus() {
161     ssid.setFocus(true);
162   }
163 
164   private static String quoteHex(String value) {
165     if (value != null && value.matches("[0-9A-Fa-f]+")) {
166       if (value.charAt(0) == '"' && value.charAt(value.length() - 1) == '"') {
167         return value;
168       }
169       return '\"' + value + '\"';
170     }
171     return value;
172   }
