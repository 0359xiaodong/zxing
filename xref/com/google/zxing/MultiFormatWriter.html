34  /**
35   * This is a factory class which finds the appropriate Writer subclass for the BarcodeFormat
36   * requested and encodes the barcode with the supplied contents.
37   *
38   * @author dswitkin@google.com (Daniel Switkin)
39   */
40  public final class MultiFormatWriter implements Writer {
41  
42    @Override
43    public BitMatrix encode(String contents,
44                            BarcodeFormat format,
45                            int width,
46                            int height) throws WriterException {
47      return encode(contents, format, width, height, null);
48    }
49  
50    @Override
51    public BitMatrix encode(String contents,
52                            BarcodeFormat format,
53                            int width, int height,
54                            Map<EncodeHintType,?> hints) throws WriterException {
55  
56      Writer writer;
57      switch (format) {
58        case EAN_8:
59          writer = new EAN8Writer();
60          break;
61        case EAN_13:
62          writer = new EAN13Writer();
63          break;
64        case UPC_A:
65          writer = new UPCAWriter();
66          break;
67        case QR_CODE:
68          writer = new QRCodeWriter();
69          break;
70        case CODE_39:
71          writer = new Code39Writer();
72          break;
73        case CODE_128:
74          writer = new Code128Writer();
75          break;
76        case ITF:
77          writer = new ITFWriter();
78          break;
79        case PDF_417:
80          writer = new PDF417Writer();
81          break;
82        case CODABAR:
83          writer = new CodaBarWriter();
84          break;
85        case DATA_MATRIX:
86          writer = new DataMatrixWriter();
87          break;
88        case AZTEC:
89          writer = new AztecWriter();
90          break;
91        default:
92          throw new IllegalArgumentException("No encoder available for format " + format);
93      }
94      return writer.encode(contents, format, width, height, hints);
95    }
96  
97  }
