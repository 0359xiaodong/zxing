63    private final BitMatrix bitMatrix;
64  
65    /**
66     * @param bitMatrix {@link BitMatrix} to parse
67     */
68    BitMatrixParser(BitMatrix bitMatrix) {
69      this.bitMatrix = bitMatrix;
70    }
71  
72    byte[] readCodewords() {
73      byte[] result = new byte[144];
74      int height = bitMatrix.getHeight();
75      int width = bitMatrix.getWidth();
76      for (int y = 0; y < height; y++) {
77        int[] bitnrRow = BITNR[y];
78        for (int x = 0; x < width; x++) {
79          int bit = bitnrRow[x];
80          if (bit >= 0 && bitMatrix.get(x, y)) {
81            result[bit / 6] |= (byte) (1 << (5 - (bit % 6)));
82          }
83        }
84      }
85      return result;
86    }
