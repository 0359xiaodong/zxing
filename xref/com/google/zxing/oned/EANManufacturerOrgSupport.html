22  /**
23   * Records EAN prefix to GS1 Member Organization, where the member organization
24   * correlates strongly with a country. This is an imperfect means of identifying
25   * a country of origin by EAN-13 barcode value. See
26   * <a href="http://en.wikipedia.org/wiki/List_of_GS1_country_codes">
27   * http://en.wikipedia.org/wiki/List_of_GS1_country_codes</a>.
28   *
29   * @author Sean Owen
30   */
31  final class EANManufacturerOrgSupport {
32  
33    private final List<int[]> ranges = new ArrayList<>();
34    private final List<String> countryIdentifiers = new ArrayList<>();
35  
36    String lookupCountryIdentifier(String productCode) {
37      initIfNeeded();
38      int prefix = Integer.parseInt(productCode.substring(0, 3));
39      int max = ranges.size();
40      for (int i = 0; i < max; i++) {
41        int[] range = ranges.get(i);
42        int start = range[0];
43        if (prefix < start) {
44          return null;
45        }
46        int end = range.length == 1 ? start : range[1];
47        if (prefix <= end) {
48          return countryIdentifiers.get(i);
49        }
50      }
51      return null;
52    }
53    
54    private void add(int[] range, String id) {
55      ranges.add(range);
56      countryIdentifiers.add(id);
57    }
