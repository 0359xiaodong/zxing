17  package com.google.zxing.oned;
18  
19  import com.google.zxing.BarcodeFormat;
20  import com.google.zxing.FormatException;
21  import com.google.zxing.NotFoundException;
22  import com.google.zxing.common.BitArray;
23  
24  /**
25   * <p>Implements decoding of the UPC-E format.</p>
26   * <p><a href="http://www.barcodeisland.com/upce.phtml">This</a> is a great reference for
27   * UPC-E information.</p>
28   *
29   * @author Sean Owen
30   */
31  public final class UPCEReader extends UPCEANReader {
32  
33    /**
34     * The pattern that marks the middle, and end, of a UPC-E pattern.
35     * There is no "second half" to a UPC-E barcode.
36     */
37    private static final int[] MIDDLE_END_PATTERN = {1, 1, 1, 1, 1, 1};
38  
39    /**
40     * See {@link #L_AND_G_PATTERNS}; these values similarly represent patterns of
41     * even-odd parity encodings of digits that imply both the number system (0 or 1)
42     * used, and the check digit.
43     */
44    private static final int[][] NUMSYS_AND_CHECK_DIGIT_PATTERNS = {
45        {0x38, 0x34, 0x32, 0x31, 0x2C, 0x26, 0x23, 0x2A, 0x29, 0x25},
46        {0x07, 0x0B, 0x0D, 0x0E, 0x13, 0x19, 0x1C, 0x15, 0x16, 0x1A}
47    };
48  
49    private final int[] decodeMiddleCounters;
50  
51    public UPCEReader() {
52      decodeMiddleCounters = new int[4];
53    }
54  
55    @Override
56    protected int decodeMiddle(BitArray row, int[] startRange, StringBuilder result)
57        throws NotFoundException {
58      int[] counters = decodeMiddleCounters;
59      counters[0] = 0;
60      counters[1] = 0;
61      counters[2] = 0;
62      counters[3] = 0;
63      int end = row.getSize();
64      int rowOffset = startRange[1];
65  
66      int lgPatternFound = 0;
67  
68      for (int x = 0; x < 6 && rowOffset < end; x++) {
69        int bestMatch = decodeDigit(row, counters, rowOffset, L_AND_G_PATTERNS);
70        result.append((char) ('0' + bestMatch % 10));
71        for (int counter : counters) {
72          rowOffset += counter;
73        }
74        if (bestMatch >= 10) {
75          lgPatternFound |= 1 << (5 - x);
76        }
77      }
78  
79      determineNumSysAndCheckDigit(result, lgPatternFound);
80  
81      return rowOffset;
82    }
83  
84    @Override
85    protected int[] decodeEnd(BitArray row, int endStart) throws NotFoundException {
86      return findGuardPattern(row, endStart, true, MIDDLE_END_PATTERN);
87    }
88  
89    @Override
90    protected boolean checkChecksum(String s) throws FormatException {
91      return super.checkChecksum(convertUPCEtoUPCA(s));
92    }
93  
94    private static void determineNumSysAndCheckDigit(StringBuilder resultString, int lgPatternFound)
95        throws NotFoundException {
96  
97      for (int numSys = 0; numSys <= 1; numSys++) {
98        for (int d = 0; d < 10; d++) {
99          if (lgPatternFound == NUMSYS_AND_CHECK_DIGIT_PATTERNS[numSys][d]) {
100           resultString.insert(0, (char) ('0' + numSys));
101           resultString.append((char) ('0' + d));
102           return;
103         }
104       }
105     }
106     throw NotFoundException.getNotFoundInstance();
107   }
108 
109   @Override
110   BarcodeFormat getBarcodeFormat() {
111     return BarcodeFormat.UPC_E;
112   }
113 
114   /**
115    * Expands a UPC-E value back into its full, equivalent UPC-A code value.
116    *
117    * @param upce UPC-E code as string of digits
118    * @return equivalent UPC-A code as string of digits
119    */
120   public static String convertUPCEtoUPCA(String upce) {
121     char[] upceChars = new char[6];
122     upce.getChars(1, 7, upceChars, 0);
123     StringBuilder result = new StringBuilder(12);
124     result.append(upce.charAt(0));
125     char lastChar = upceChars[5];
126     switch (lastChar) {
127       case '0':
128       case '1':
129       case '2':
130         result.append(upceChars, 0, 2);
131         result.append(lastChar);
132         result.append("0000");
133         result.append(upceChars, 2, 3);
134         break;
135       case '3':
136         result.append(upceChars, 0, 3);
137         result.append("00000");
138         result.append(upceChars, 3, 2);
139         break;
140       case '4':
141         result.append(upceChars, 0, 4);
142         result.append("00000");
143         result.append(upceChars[4]);
144         break;
145       default:
146         result.append(upceChars, 0, 5);
147         result.append("0000");
148         result.append(lastChar);
149         break;
150     }
151     result.append(upce.charAt(7));
152     return result.toString();
153   }
