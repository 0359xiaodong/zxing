17  package com.google.zxing.oned;
18  
19  import com.google.zxing.BarcodeFormat;
20  import com.google.zxing.EncodeHintType;
21  import com.google.zxing.WriterException;
22  import com.google.zxing.common.BitMatrix;
23  
24  import java.util.ArrayList;
25  import java.util.Collection;
26  import java.util.Map;

28  /**
29   * This object renders a CODE128 code as a {@link BitMatrix}.
30   * 
31   * @author erik.barbara@gmail.com (Erik Barbara)
32   */
33  public final class Code128Writer extends OneDimensionalCodeWriter {
34  
35    private static final int CODE_START_B = 104;
36    private static final int CODE_START_C = 105;
37    private static final int CODE_CODE_B = 100;
38    private static final int CODE_CODE_C = 99;
39    private static final int CODE_STOP = 106;
40  
41    // Dummy characters used to specify control characters in input
42    private static final char ESCAPE_FNC_1 = '\u00f1';
43    private static final char ESCAPE_FNC_2 = '\u00f2';
44    private static final char ESCAPE_FNC_3 = '\u00f3';
45    private static final char ESCAPE_FNC_4 = '\u00f4';
46  
47    private static final int CODE_FNC_1 = 102;   // Code A, Code B, Code C
48    private static final int CODE_FNC_2 = 97;    // Code A, Code B
49    private static final int CODE_FNC_3 = 96;    // Code A, Code B
50    private static final int CODE_FNC_4_B = 100; // Code B
51  
52    @Override
53    public BitMatrix encode(String contents,
54                            BarcodeFormat format,
55                            int width,
56                            int height,
57                            Map<EncodeHintType,?> hints) throws WriterException {
58      if (format != BarcodeFormat.CODE_128) {
59        throw new IllegalArgumentException("Can only encode CODE_128, but got " + format);
60      }
61      return super.encode(contents, format, width, height, hints);
62    }
63  
64    @Override
65    public boolean[] encode(String contents) {
66      int length = contents.length();
67      // Check length
68      if (length < 1 || length > 80) {
69        throw new IllegalArgumentException(
70            "Contents length should be between 1 and 80 characters, but got " + length);
71      }
72      // Check content
73      for (int i = 0; i < length; i++) {
74        char c = contents.charAt(i);
75        if (c < ' ' || c > '~') {
76          switch (c) {
77            case ESCAPE_FNC_1:
78            case ESCAPE_FNC_2:
79            case ESCAPE_FNC_3:
80            case ESCAPE_FNC_4:
81              break;
82            default:
83              throw new IllegalArgumentException("Bad character in input: " + c);
84          }
85        }
86      }
87      
88      Collection<int[]> patterns = new ArrayList<>(); // temporary storage for patterns
89      int checkSum = 0;
90      int checkWeight = 1;
91      int codeSet = 0; // selected code (CODE_CODE_B or CODE_CODE_C)
92      int position = 0; // position in contents
93      
94      while (position < length) {
95        //Select code to use
96        int requiredDigitCount = codeSet == CODE_CODE_C ? 2 : 4;
97        int newCodeSet;
98        if (isDigits(contents, position, requiredDigitCount)) {
99          newCodeSet = CODE_CODE_C;
100       } else {
101         newCodeSet = CODE_CODE_B;
102       }
103       
104       //Get the pattern index
105       int patternIndex;
106       if (newCodeSet == codeSet) {
107         // Encode the current character
108         // First handle escapes
109         switch (contents.charAt(position)) {
110           case ESCAPE_FNC_1:
111             patternIndex = CODE_FNC_1;
112             break;
113           case ESCAPE_FNC_2:
114             patternIndex = CODE_FNC_2;
115             break;
116           case ESCAPE_FNC_3:
117             patternIndex = CODE_FNC_3;
118             break;
119           case ESCAPE_FNC_4:
120             patternIndex = CODE_FNC_4_B; // FIXME if this ever outputs Code A
121             break;
122           default:
123             // Then handle normal characters otherwise
124             if (codeSet == CODE_CODE_B) {
125               patternIndex = contents.charAt(position) - ' ';
126             } else { // CODE_CODE_C
127               patternIndex = Integer.parseInt(contents.substring(position, position + 2));
128               position++; // Also incremented below
129             }
130         }
131         position++;
132       } else {
133         // Should we change the current code?
134         // Do we have a code set?
135         if (codeSet == 0) {
136           // No, we don't have a code set
137           if (newCodeSet == CODE_CODE_B) {
138             patternIndex = CODE_START_B;
139           } else {
140             // CODE_CODE_C
141             patternIndex = CODE_START_C;
142           }
143         } else {
144           // Yes, we have a code set
145           patternIndex = newCodeSet;
146         }
147         codeSet = newCodeSet;
148       }
149       
150       // Get the pattern
151       patterns.add(Code128Reader.CODE_PATTERNS[patternIndex]);
152       
153       // Compute checksum
154       checkSum += patternIndex * checkWeight;
155       if (position != 0) {
156         checkWeight++;
157       }
158     }
159     
160     // Compute and append checksum
161     checkSum %= 103;
162     patterns.add(Code128Reader.CODE_PATTERNS[checkSum]);
163     
164     // Append stop code
165     patterns.add(Code128Reader.CODE_PATTERNS[CODE_STOP]);
166     
167     // Compute code width
168     int codeWidth = 0;
169     for (int[] pattern : patterns) {
170       for (int width : pattern) {
171         codeWidth += width;
172       }
173     }
174     
175     // Compute result
176     boolean[] result = new boolean[codeWidth];
177     int pos = 0;
178     for (int[] pattern : patterns) {
179       pos += appendPattern(result, pos, pattern, true);
180     }
181     
182     return result;
183   }
184 
185   private static boolean isDigits(CharSequence value, int start, int length) {
186     int end = start + length;
187     int last = value.length();
188     for (int i = start; i < end && i < last; i++) {
189       char c = value.charAt(i);
190       if (c < '0' || c > '9') {
191         if (c != ESCAPE_FNC_1) {
192           return false;
193         }
194         end++; // ignore FNC_1
195       }
196     }
197     return end <= last; // end > last if we've run out of string
198   }
