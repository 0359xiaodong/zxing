44    // These two values are critical for determining how permissive the decoding will be.
45    // We've arrived at these values through a lot of trial and error. Setting them any higher
46    // lets false positives creep in quickly.
47    private static final int MAX_AVG_VARIANCE = (int) (PATTERN_MATCH_RESULT_SCALE_FACTOR * 0.48f);
48    private static final int MAX_INDIVIDUAL_VARIANCE = (int) (PATTERN_MATCH_RESULT_SCALE_FACTOR * 0.7f);
49  
50    /**
51     * Start/end guard pattern.
52     */
53    static final int[] START_END_PATTERN = {1, 1, 1,};
54  
55    /**
56     * Pattern marking the middle of a UPC/EAN pattern, separating the two halves.
57     */
58    static final int[] MIDDLE_PATTERN = {1, 1, 1, 1, 1};
59  
60    /**
61     * "Odd", or "L" patterns used to encode UPC/EAN digits.
62     */
63    static final int[][] L_PATTERNS = {
64        {3, 2, 1, 1}, // 0
65        {2, 2, 2, 1}, // 1
66        {2, 1, 2, 2}, // 2
67        {1, 4, 1, 1}, // 3
68        {1, 1, 3, 2}, // 4
69        {1, 2, 3, 1}, // 5
70        {1, 1, 1, 4}, // 6
71        {1, 3, 1, 2}, // 7
72        {1, 2, 1, 3}, // 8
73        {3, 1, 1, 2}  // 9
74    };
75  
76    /**
77     * As above but also including the "even", or "G" patterns used to encode UPC/EAN digits.
78     */
79    static final int[][] L_AND_G_PATTERNS;
80  
81    static {
82      L_AND_G_PATTERNS = new int[20][];
83      System.arraycopy(L_PATTERNS, 0, L_AND_G_PATTERNS, 0, 10);
84      for (int i = 10; i < 20; i++) {
85        int[] widths = L_PATTERNS[i - 10];
86        int[] reversedWidths = new int[widths.length];
87        for (int j = 0; j < widths.length; j++) {
88          reversedWidths[j] = widths[widths.length - j - 1];
89        }
90        L_AND_G_PATTERNS[i] = reversedWidths;
91      }
92    }

96    private final EANManufacturerOrgSupport eanManSupport;

98    protected UPCEANReader() {
99      decodeRowStringBuffer = new StringBuilder(20);

101     eanManSupport = new EANManufacturerOrgSupport();

104   static int[] findStartGuardPattern(BitArray row) throws NotFoundException {
105     boolean foundStart = false;
106     int[] startRange = null;
107     int nextStart = 0;
108     int[] counters = new int[START_END_PATTERN.length];
109     while (!foundStart) {
110       Arrays.fill(counters, 0, START_END_PATTERN.length, 0);
111       startRange = findGuardPattern(row, nextStart, false, START_END_PATTERN, counters);
112       int start = startRange[0];
113       nextStart = startRange[1];
114       // Make sure there is a quiet zone at least as big as the start pattern before the barcode.
115       // If this check would run off the left edge of the image, do not accept this barcode,
116       // as it is very likely to be a false positive.
117       int quietStart = start - (nextStart - start);
118       if (quietStart >= 0) {
119         foundStart = row.isRange(quietStart, start, false);
120       }
121     }
122     return startRange;
123   }

125   @Override
126   public Result decodeRow(int rowNumber, BitArray row, Map<DecodeHintType,?> hints)
127       throws NotFoundException, ChecksumException, FormatException {
128     return decodeRow(rowNumber, row, findStartGuardPattern(row), hints);
129   }

131   /**
132    * <p>Like {@link #decodeRow(int, BitArray, java.util.Map)}, but
133    * allows caller to inform method about where the UPC/EAN start pattern is
134    * found. This allows this to be computed once and reused across many implementations.</p>
135    */
136   public Result decodeRow(int rowNumber,
137                           BitArray row,
138                           int[] startGuardRange,
139                           Map<DecodeHintType,?> hints)
140       throws NotFoundException, ChecksumException, FormatException {
141 
142     ResultPointCallback resultPointCallback = hints == null ? null :
143         (ResultPointCallback) hints.get(DecodeHintType.NEED_RESULT_POINT_CALLBACK);
144 
145     if (resultPointCallback != null) {
146       resultPointCallback.foundPossibleResultPoint(new ResultPoint(
147           (startGuardRange[0] + startGuardRange[1]) / 2.0f, rowNumber
148       ));
149     }
150 
151     StringBuilder result = decodeRowStringBuffer;
152     result.setLength(0);
153     int endStart = decodeMiddle(row, startGuardRange, result);
154 
155     if (resultPointCallback != null) {
156       resultPointCallback.foundPossibleResultPoint(new ResultPoint(
157           endStart, rowNumber
158       ));
159     }
160 
161     int[] endRange = decodeEnd(row, endStart);
162 
163     if (resultPointCallback != null) {
164       resultPointCallback.foundPossibleResultPoint(new ResultPoint(
165           (endRange[0] + endRange[1]) / 2.0f, rowNumber
166       ));
167     }
168 
169 
170     // Make sure there is a quiet zone at least as big as the end pattern after the barcode. The
171     // spec might want more whitespace, but in practice this is the maximum we can count on.
172     int end = endRange[1];
173     int quietEnd = end + (end - endRange[0]);
174     if (quietEnd >= row.getSize() || !row.isRange(end, quietEnd, false)) {
175       throw NotFoundException.getNotFoundInstance();
176     }
177 
178     String resultString = result.toString();
179     // UPC/EAN should never be less than 8 chars anyway
180     if (resultString.length() < 8) {
181       throw FormatException.getFormatInstance();
182     }
183     if (!checkChecksum(resultString)) {
184       throw ChecksumException.getChecksumInstance();
185     }
186 
187     float left = (float) (startGuardRange[1] + startGuardRange[0]) / 2.0f;
188     float right = (float) (endRange[1] + endRange[0]) / 2.0f;
189     BarcodeFormat format = getBarcodeFormat();
190     Result decodeResult = new Result(resultString,
191         null, // no natural byte representation for these barcodes
192         new ResultPoint[]{
193             new ResultPoint(left, (float) rowNumber),
194             new ResultPoint(right, (float) rowNumber)},
195         format);
196 
197     try {
198       Result extensionResult = extensionReader.decodeRow(rowNumber, row, endRange[1]);
199       decodeResult.putMetadata(ResultMetadataType.UPC_EAN_EXTENSION, extensionResult.getText());
200       decodeResult.putAllMetadata(extensionResult.getResultMetadata());
201       decodeResult.addResultPoints(extensionResult.getResultPoints());
202     } catch (ReaderException re) {
203       // continue
204     }
205 
206     if (format == BarcodeFormat.EAN_13 || format == BarcodeFormat.UPC_A) {
207       String countryID = eanManSupport.lookupCountryIdentifier(resultString);
208       if (countryID != null) {
209         decodeResult.putMetadata(ResultMetadataType.POSSIBLE_COUNTRY, countryID);
210       }
211     }
212 
213     return decodeResult;
214   }

216   /**
217    * @return {@link #checkStandardUPCEANChecksum(CharSequence)}
218    */
219   boolean checkChecksum(String s) throws ChecksumException, FormatException {
220     return checkStandardUPCEANChecksum(s);
221   }

223   /**
224    * Computes the UPC/EAN checksum on a string of digits, and reports
225    * whether the checksum is correct or not.
226    *
227    * @param s string of digits to check
228    * @return true iff string of digits passes the UPC/EAN checksum algorithm
229    * @throws FormatException if the string does not contain only digits
230    */
231   static boolean checkStandardUPCEANChecksum(CharSequence s) throws FormatException {
232     int length = s.length();
233     if (length == 0) {
234       return false;
235     }
236 
237     int sum = 0;
238     for (int i = length - 2; i >= 0; i -= 2) {
239       int digit = (int) s.charAt(i) - (int) '0';
240       if (digit < 0 || digit > 9) {
241         throw FormatException.getFormatInstance();
242       }
243       sum += digit;
244     }
245     sum *= 3;
246     for (int i = length - 1; i >= 0; i -= 2) {
247       int digit = (int) s.charAt(i) - (int) '0';
248       if (digit < 0 || digit > 9) {
249         throw FormatException.getFormatInstance();
250       }
251       sum += digit;
252     }
253     return sum % 10 == 0;
254   }

260   static int[] findGuardPattern(BitArray row,
261                                 int rowOffset,
262                                 boolean whiteFirst,
263                                 int[] pattern) throws NotFoundException {
264     return findGuardPattern(row, rowOffset, whiteFirst, pattern, new int[pattern.length]);
265   }

267   /**
268    * @param row row of black/white values to search
269    * @param rowOffset position to start search
270    * @param whiteFirst if true, indicates that the pattern specifies white/black/white/...
271    * pixel counts, otherwise, it is interpreted as black/white/black/...
272    * @param pattern pattern of counts of number of black and white pixels that are being
273    * searched for as a pattern
274    * @param counters array of counters, as long as pattern, to re-use
275    * @return start/end horizontal offset of guard pattern, as an array of two ints
276    * @throws NotFoundException if pattern is not found
277    */
278   private static int[] findGuardPattern(BitArray row,
279                                         int rowOffset,
280                                         boolean whiteFirst,
281                                         int[] pattern,
282                                         int[] counters) throws NotFoundException {
283     int patternLength = pattern.length;
284     int width = row.getSize();
285     boolean isWhite = whiteFirst;
286     rowOffset = whiteFirst ? row.getNextUnset(rowOffset) : row.getNextSet(rowOffset);
287     int counterPosition = 0;
288     int patternStart = rowOffset;
289     for (int x = rowOffset; x < width; x++) {
290       if (row.get(x) ^ isWhite) {
291         counters[counterPosition]++;
292       } else {
293         if (counterPosition == patternLength - 1) {
294           if (patternMatchVariance(counters, pattern, MAX_INDIVIDUAL_VARIANCE) < MAX_AVG_VARIANCE) {
295             return new int[]{patternStart, x};
296           }
297           patternStart += counters[0] + counters[1];
298           System.arraycopy(counters, 2, counters, 0, patternLength - 2);
299           counters[patternLength - 2] = 0;
300           counters[patternLength - 1] = 0;
301           counterPosition--;
302         } else {
303           counterPosition++;
304         }
305         counters[counterPosition] = 1;
306         isWhite = !isWhite;
307       }
308     }
309     throw NotFoundException.getNotFoundInstance();
310   }

312   /**
313    * Attempts to decode a single UPC/EAN-encoded digit.
314    *
315    * @param row row of black/white values to decode
316    * @param counters the counts of runs of observed black/white/black/... values
317    * @param rowOffset horizontal offset to start decoding from
318    * @param patterns the set of patterns to use to decode -- sometimes different encodings
319    * for the digits 0-9 are used, and this indicates the encodings for 0 to 9 that should
320    * be used
321    * @return horizontal offset of first pixel beyond the decoded digit
322    * @throws NotFoundException if digit cannot be decoded
323    */
324   static int decodeDigit(BitArray row, int[] counters, int rowOffset, int[][] patterns)
325       throws NotFoundException {
326     recordPattern(row, rowOffset, counters);
327     int bestVariance = MAX_AVG_VARIANCE; // worst variance we'll accept
328     int bestMatch = -1;
329     int max = patterns.length;
330     for (int i = 0; i < max; i++) {
331       int[] pattern = patterns[i];
332       int variance = patternMatchVariance(counters, pattern, MAX_INDIVIDUAL_VARIANCE);
333       if (variance < bestVariance) {
334         bestVariance = variance;
335         bestMatch = i;
336       }
337     }
338     if (bestMatch >= 0) {
339       return bestMatch;
340     } else {
341       throw NotFoundException.getNotFoundInstance();
342     }
343   }
