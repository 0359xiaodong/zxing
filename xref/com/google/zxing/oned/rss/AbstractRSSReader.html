17  package com.google.zxing.oned.rss;
18  
19  import com.google.zxing.NotFoundException;
20  import com.google.zxing.oned.OneDReader;
21  
22  public abstract class AbstractRSSReader extends OneDReader {
23  
24    private static final int MAX_AVG_VARIANCE = (int) (PATTERN_MATCH_RESULT_SCALE_FACTOR * 0.2f);
25    private static final int MAX_INDIVIDUAL_VARIANCE = (int) (PATTERN_MATCH_RESULT_SCALE_FACTOR * 0.45f);
26  
27    private static final float MIN_FINDER_PATTERN_RATIO = 9.5f / 12.0f;
28    private static final float MAX_FINDER_PATTERN_RATIO = 12.5f / 14.0f;
29  
30    private final int[] decodeFinderCounters;
31    private final int[] dataCharacterCounters;
32    private final float[] oddRoundingErrors;
33    private final float[] evenRoundingErrors;
34    private final int[] oddCounts;
35    private final int[] evenCounts;
36  
37    protected AbstractRSSReader(){
38      decodeFinderCounters = new int[4];
39      dataCharacterCounters = new int[8];
40      oddRoundingErrors = new float[4];
41      evenRoundingErrors = new float[4];
42      oddCounts = new int[dataCharacterCounters.length / 2];
43      evenCounts = new int[dataCharacterCounters.length / 2];
44    }
45  
46    protected final int[] getDecodeFinderCounters() {
47      return decodeFinderCounters;
48    }
49  
50    protected final int[] getDataCharacterCounters() {
51      return dataCharacterCounters;
52    }
53  
54    protected final float[] getOddRoundingErrors() {
55      return oddRoundingErrors;
56    }
57  
58    protected final float[] getEvenRoundingErrors() {
59      return evenRoundingErrors;
60    }
61  
62    protected final int[] getOddCounts() {
63      return oddCounts;
64    }
65  
66    protected final int[] getEvenCounts() {
67      return evenCounts;
68    }
69  
70    protected static int parseFinderValue(int[] counters,
71                                          int[][] finderPatterns) throws NotFoundException {
72      for (int value = 0; value < finderPatterns.length; value++) {
73        if (patternMatchVariance(counters, finderPatterns[value], MAX_INDIVIDUAL_VARIANCE) <
74            MAX_AVG_VARIANCE) {
75          return value;
76        }
77      }
78      throw NotFoundException.getNotFoundInstance();
79    }
80  
81    protected static int count(int[] array) {
82      int count = 0;
83      for (int a : array) {
84        count += a;
85      }
86      return count;
87    }
88  
89    protected static void increment(int[] array, float[] errors) {
90      int index = 0;
91      float biggestError = errors[0];
92      for (int i = 1; i < array.length; i++) {
93        if (errors[i] > biggestError) {
94          biggestError = errors[i];
95          index = i;
96        }
97      }
98      array[index]++;
99    }
100 
101   protected static void decrement(int[] array, float[] errors) {
102     int index = 0;
103     float biggestError = errors[0];
104     for (int i = 1; i < array.length; i++) {
105       if (errors[i] < biggestError) {
106         biggestError = errors[i];
107         index = i;
108       }
109     }
110     array[index]--;
111   }
112 
113   protected static boolean isFinderPattern(int[] counters) {
114     int firstTwoSum = counters[0] + counters[1];
115     int sum = firstTwoSum + counters[2] + counters[3];
116     float ratio = (float) firstTwoSum / (float) sum;
117     if (ratio >= MIN_FINDER_PATTERN_RATIO && ratio <= MAX_FINDER_PATTERN_RATIO) {
118       // passes ratio test in spec, but see if the counts are unreasonable
119       int minCounter = Integer.MAX_VALUE;
120       int maxCounter = Integer.MIN_VALUE;
121       for (int counter : counters) {
122         if (counter > maxCounter) {
123           maxCounter = counter;
124         }
125         if (counter < minCounter) {
126           minCounter = counter;
127         }
128       }
129       return maxCounter < 10 * minCounter;
130     }
131     return false;
132   }
133 }
