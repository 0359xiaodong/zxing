70    private static final int MODULO_VALUE = 0x12D;
71  
72    private static final int[] LOG;
73    private static final int[] ALOG;
74  
75    static {
76      //Create log and antilog table
77      LOG = new int[256];
78      ALOG = new int[255];
79  
80      int p = 1;
81      for (int i = 0; i < 255; i++) {
82        ALOG[i] = p;
83        LOG[p] = i;
84        p *= 2;
85        if (p >= 256) {
86          p ^= MODULO_VALUE;
87        }
88      }
89    }

94    /**
95     * Creates the ECC200 error correction for an encoded message.
96     *
97     * @param codewords  the codewords
98     * @param symbolInfo information about the symbol to be encoded
99     * @return the codewords with interleaved error correction.
100    */
101   public static String encodeECC200(String codewords, SymbolInfo symbolInfo) {
102     if (codewords.length() != symbolInfo.getDataCapacity()) {
103       throw new IllegalArgumentException(
104           "The number of codewords does not match the selected symbol");
105     }
106     StringBuilder sb = new StringBuilder(symbolInfo.getDataCapacity() + symbolInfo.getErrorCodewords());
107     sb.append(codewords);
108     int blockCount = symbolInfo.getInterleavedBlockCount();
109     if (blockCount == 1) {
110       String ecc = createECCBlock(codewords, symbolInfo.getErrorCodewords());
111       sb.append(ecc);
112     } else {
113       sb.setLength(sb.capacity());
114       int[] dataSizes = new int[blockCount];
115       int[] errorSizes = new int[blockCount];
116       int[] startPos = new int[blockCount];
117       for (int i = 0; i < blockCount; i++) {
118         dataSizes[i] = symbolInfo.getDataLengthForInterleavedBlock(i + 1);
119         errorSizes[i] = symbolInfo.getErrorLengthForInterleavedBlock(i + 1);
120         startPos[i] = 0;
121         if (i > 0) {
122           startPos[i] = startPos[i - 1] + dataSizes[i];
123         }
124       }
125       for (int block = 0; block < blockCount; block++) {
126         StringBuilder temp = new StringBuilder(dataSizes[block]);
127         for (int d = block; d < symbolInfo.getDataCapacity(); d += blockCount) {
128           temp.append(codewords.charAt(d));
129         }
130         String ecc = createECCBlock(temp.toString(), errorSizes[block]);
131         int pos = 0;
132         for (int e = block; e < errorSizes[block] * blockCount; e += blockCount) {
133           sb.setCharAt(symbolInfo.getDataCapacity() + e, ecc.charAt(pos++));
134         }
135       }
136     }
137     return sb.toString();
138 
139   }

141   private static String createECCBlock(CharSequence codewords, int numECWords) {
142     return createECCBlock(codewords, 0, codewords.length(), numECWords);
143   }
144 
145   private static String createECCBlock(CharSequence codewords, int start, int len, int numECWords) {
146     int table = -1;
147     for (int i = 0; i < FACTOR_SETS.length; i++) {
148       if (FACTOR_SETS[i] == numECWords) {
149         table = i;
150         break;
151       }
152     }
153     if (table < 0) {
154       throw new IllegalArgumentException(
155           "Illegal number of error correction codewords specified: " + numECWords);
156     }
157     int[] poly = FACTORS[table];
158     char[] ecc = new char[numECWords];
159     for (int i = 0; i < numECWords; i++) {
160       ecc[i] = 0;
161     }
162     for (int i = start; i < start + len; i++) {
163       int m = ecc[numECWords - 1] ^ codewords.charAt(i);
164       for (int k = numECWords - 1; k > 0; k--) {
165         if (m != 0 && poly[k] != 0) {
166           ecc[k] = (char) (ecc[k - 1] ^ ALOG[(LOG[m] + LOG[poly[k]]) % 255]);
167         } else {
168           ecc[k] = ecc[k - 1];
169         }
170       }
171       if (m != 0 && poly[0] != 0) {
172         ecc[0] = (char) ALOG[(LOG[m] + LOG[poly[0]]) % 255];
173       } else {
174         ecc[0] = 0;
175       }
176     }
177     char[] eccReversed = new char[numECWords];
178     for (int i = 0; i < numECWords; i++) {
179       eccReversed[i] = ecc[numECWords - i - 1];
180     }
181     return String.valueOf(eccReversed);
182   }
