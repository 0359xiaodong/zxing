111   public static SymbolInfo lookup(int dataCodewords, boolean allowRectangular, boolean fail) {
112     SymbolShapeHint shape = allowRectangular
113         ? SymbolShapeHint.FORCE_NONE : SymbolShapeHint.FORCE_SQUARE;
114     return lookup(dataCodewords, shape, fail);
115   }

117   private static SymbolInfo lookup(int dataCodewords, SymbolShapeHint shape, boolean fail) {
118     return lookup(dataCodewords, shape, null, null, fail);
119   }
120 
121   public static SymbolInfo lookup(int dataCodewords,
122                                   SymbolShapeHint shape, 
123                                   Dimension minSize, 
124                                   Dimension maxSize, 
125                                   boolean fail) {
126     for (SymbolInfo symbol : symbols) {
127       if (shape == SymbolShapeHint.FORCE_SQUARE && symbol.rectangular) {
128         continue;
129       }
130       if (shape == SymbolShapeHint.FORCE_RECTANGLE && !symbol.rectangular) {
131         continue;
132       }
133       if (minSize != null
134           && (symbol.getSymbolWidth() < minSize.getWidth()
135           || symbol.getSymbolHeight() < minSize.getHeight())) {
136         continue;
137       }
138       if (maxSize != null
139           && (symbol.getSymbolWidth() > maxSize.getWidth()
140           || symbol.getSymbolHeight() > maxSize.getHeight())) {
141         continue;
142       }
143       if (dataCodewords <= symbol.dataCapacity) {
144         return symbol;
145       }
146     }
147     if (fail) {
148       throw new IllegalArgumentException(
149           "Can't find a symbol arrangement that matches the message. Data codewords: "
150               + dataCodewords);
151     }
152     return null;
153   }

155   final int getHorizontalDataRegions() {
156     switch (dataRegions) {
157       case 1:
158         return 1;
159       case 2:
160         return 2;
161       case 4:
162         return 2;
163       case 16:
164         return 4;
165       case 36:
166         return 6;
167       default:
168         throw new IllegalStateException("Cannot handle this number of data regions");
169     }
170   }
171 
172   final int getVerticalDataRegions() {
173     switch (dataRegions) {
174       case 1:
175         return 1;
176       case 2:
177         return 1;
178       case 4:
179         return 2;
180       case 16:
181         return 4;
182       case 36:
183         return 6;
184       default:
185         throw new IllegalStateException("Cannot handle this number of data regions");
186     }
187   }

189   public final int getSymbolDataWidth() {
190     return getHorizontalDataRegions() * matrixWidth;

193   public final int getSymbolDataHeight() {
194     return getVerticalDataRegions() * matrixHeight;

197   public final int getSymbolWidth() {
198     return getSymbolDataWidth() + (getHorizontalDataRegions() * 2);

229   @Override
230   public final String toString() {
231     StringBuilder sb = new StringBuilder();
232     sb.append(rectangular ? "Rectangular Symbol:" : "Square Symbol:");
233     sb.append(" data region ").append(matrixWidth).append('x').append(matrixHeight);
234     sb.append(", symbol size ").append(getSymbolWidth()).append('x').append(getSymbolHeight());
235     sb.append(", symbol data size ").append(getSymbolDataWidth()).append('x').append(getSymbolDataHeight());
236     sb.append(", codewords ").append(dataCapacity).append('+').append(errorCodewords);
237     return sb.toString();
238   }
