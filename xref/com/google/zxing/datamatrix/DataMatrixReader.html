38  /**
39   * This implementation can detect and decode Data Matrix codes in an image.
40   *
41   * @author bbrown@google.com (Brian Brown)
42   */
43  public final class DataMatrixReader implements Reader {
44  
45    private static final ResultPoint[] NO_POINTS = new ResultPoint[0];
46  
47    private final Decoder decoder = new Decoder();
48  
49    /**
50     * Locates and decodes a Data Matrix code in an image.
51     *
52     * @return a String representing the content encoded by the Data Matrix code
53     * @throws NotFoundException if a Data Matrix code cannot be found
54     * @throws FormatException if a Data Matrix code cannot be decoded
55     * @throws ChecksumException if error correction fails
56     */
57    @Override
58    public Result decode(BinaryBitmap image) throws NotFoundException, ChecksumException, FormatException {
59      return decode(image, null);
60    }
61  
62    @Override
63    public Result decode(BinaryBitmap image, Map<DecodeHintType,?> hints)
64        throws NotFoundException, ChecksumException, FormatException {
65      DecoderResult decoderResult;
66      ResultPoint[] points;
67      if (hints != null && hints.containsKey(DecodeHintType.PURE_BARCODE)) {
68        BitMatrix bits = extractPureBits(image.getBlackMatrix());
69        decoderResult = decoder.decode(bits);
70        points = NO_POINTS;
71      } else {
72        DetectorResult detectorResult = new Detector(image.getBlackMatrix()).detect();
73        decoderResult = decoder.decode(detectorResult.getBits());
74        points = detectorResult.getPoints();
75      }
76      Result result = new Result(decoderResult.getText(), decoderResult.getRawBytes(), points,
77          BarcodeFormat.DATA_MATRIX);
78      List<byte[]> byteSegments = decoderResult.getByteSegments();
79      if (byteSegments != null) {
80        result.putMetadata(ResultMetadataType.BYTE_SEGMENTS, byteSegments);
81      }
82      String ecLevel = decoderResult.getECLevel();
83      if (ecLevel != null) {
84        result.putMetadata(ResultMetadataType.ERROR_CORRECTION_LEVEL, ecLevel);
85      }
86      return result;
87    }
88  
89    @Override
90    public void reset() {
91      // do nothing
92    }
93  
94    /**
95     * This method detects a code in a "pure" image -- that is, pure monochrome image
96     * which contains only an unrotated, unskewed, image of a code, with some white border
97     * around it. This is a specialized method that works exceptionally fast in this special
98     * case.
99     *
100    * @see com.google.zxing.qrcode.QRCodeReader#extractPureBits(BitMatrix)
101    */
102   private static BitMatrix extractPureBits(BitMatrix image) throws NotFoundException {
103 
104     int[] leftTopBlack = image.getTopLeftOnBit();
105     int[] rightBottomBlack = image.getBottomRightOnBit();
106     if (leftTopBlack == null || rightBottomBlack == null) {
107       throw NotFoundException.getNotFoundInstance();
108     }
109 
110     int moduleSize = moduleSize(leftTopBlack, image);
111 
112     int top = leftTopBlack[1];
113     int bottom = rightBottomBlack[1];
114     int left = leftTopBlack[0];
115     int right = rightBottomBlack[0];
116 
117     int matrixWidth = (right - left + 1) / moduleSize;
118     int matrixHeight = (bottom - top + 1) / moduleSize;
119     if (matrixWidth <= 0 || matrixHeight <= 0) {
120       throw NotFoundException.getNotFoundInstance();
121     }
122 
123     // Push in the "border" by half the module width so that we start
124     // sampling in the middle of the module. Just in case the image is a
125     // little off, this will help recover.
126     int nudge = moduleSize / 2;
127     top += nudge;
128     left += nudge;
129 
130     // Now just read off the bits
131     BitMatrix bits = new BitMatrix(matrixWidth, matrixHeight);
132     for (int y = 0; y < matrixHeight; y++) {
133       int iOffset = top + y * moduleSize;
134       for (int x = 0; x < matrixWidth; x++) {
135         if (image.get(left + x * moduleSize, iOffset)) {
136           bits.set(x, y);
137         }
138       }
139     }
140     return bits;
141   }
142 
143   private static int moduleSize(int[] leftTopBlack, BitMatrix image) throws NotFoundException {
144     int width = image.getWidth();
145     int x = leftTopBlack[0];
146     int y = leftTopBlack[1];
147     while (x < width && image.get(x, y)) {
148       x++;
149     }
150     if (x == width) {
151       throw NotFoundException.getNotFoundInstance();
152     }
153 
154     int moduleSize = x - leftTopBlack[0];
155     if (moduleSize == 0) {
156       throw NotFoundException.getNotFoundInstance();
157     }
158     return moduleSize;
159   }
