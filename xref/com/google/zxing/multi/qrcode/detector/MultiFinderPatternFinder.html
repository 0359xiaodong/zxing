35  /**
36   * <p>This class attempts to find finder patterns in a QR Code. Finder patterns are the square
37   * markers at three corners of a QR Code.</p>
38   *
39   * <p>This class is thread-safe but not reentrant. Each thread must allocate its own object.
40   *
41   * <p>In contrast to {@link FinderPatternFinder}, this class will return an array of all possible
42   * QR code locations in the image.</p>
43   *
44   * <p>Use the TRY_HARDER hint to ask for a more thorough detection.</p>
45   *
46   * @author Sean Owen
47   * @author Hannes Erven
48   */
49  final class MultiFinderPatternFinder extends FinderPatternFinder {
50  
51    private static final FinderPatternInfo[] EMPTY_RESULT_ARRAY = new FinderPatternInfo[0];
52  
53    // TODO MIN_MODULE_COUNT and MAX_MODULE_COUNT would be great hints to ask the user for
54    // since it limits the number of regions to decode
55  
56    // max. legal count of modules per QR code edge (177)
57    private static final float MAX_MODULE_COUNT_PER_EDGE = 180;
58    // min. legal count per modules per QR code edge (11)
59    private static final float MIN_MODULE_COUNT_PER_EDGE = 9;
60  
61    /**
62     * More or less arbitrary cutoff point for determining if two finder patterns might belong
63     * to the same code if they differ less than DIFF_MODSIZE_CUTOFF_PERCENT percent in their
64     * estimated modules sizes.
65     */
66    private static final float DIFF_MODSIZE_CUTOFF_PERCENT = 0.05f;
67  
68    /**
69     * More or less arbitrary cutoff point for determining if two finder patterns might belong
70     * to the same code if they differ less than DIFF_MODSIZE_CUTOFF pixels/module in their
71     * estimated modules sizes.
72     */
73    private static final float DIFF_MODSIZE_CUTOFF = 0.5f;
74  
75  
76    /**
77     * A comparator that orders FinderPatterns by their estimated module size.
78     */
79    private static final class ModuleSizeComparator implements Comparator<FinderPattern>, Serializable {
80      @Override
81      public int compare(FinderPattern center1, FinderPattern center2) {
82        float value = center2.getEstimatedModuleSize() - center1.getEstimatedModuleSize();
83        return value < 0.0 ? -1 : value > 0.0 ? 1 : 0;
84      }
85    }

87    /**
88     * <p>Creates a finder that will search the image for three finder patterns.</p>
89     *
90     * @param image image to search
91     */
92    MultiFinderPatternFinder(BitMatrix image) {
93      super(image);
94    }
95  
96    MultiFinderPatternFinder(BitMatrix image, ResultPointCallback resultPointCallback) {
97      super(image, resultPointCallback);
98    }

100   /**
101    * @return the 3 best {@link FinderPattern}s from our list of candidates. The "best" are
102    *         those that have been detected at least {@link #CENTER_QUORUM} times, and whose module
103    *         size differs from the average among those patterns the least
104    * @throws NotFoundException if 3 such finder patterns do not exist
105    */
106   private FinderPattern[][] selectMutipleBestPatterns() throws NotFoundException {
107     List<FinderPattern> possibleCenters = getPossibleCenters();
108     int size = possibleCenters.size();
109 
110     if (size < 3) {
111       // Couldn't find enough finder patterns
112       throw NotFoundException.getNotFoundInstance();
113     }
114 
115     /*
116      * Begin HE modifications to safely detect multiple codes of equal size
117      */
118     if (size == 3) {
119       return new FinderPattern[][]{
120           new FinderPattern[]{
121               possibleCenters.get(0),
122               possibleCenters.get(1),
123               possibleCenters.get(2)
124           }
125       };
126     }
127 
128     // Sort by estimated module size to speed up the upcoming checks
129     Collections.sort(possibleCenters, new ModuleSizeComparator());
130 
131     /*
132      * Now lets start: build a list of tuples of three finder locations that
133      *  - feature similar module sizes
134      *  - are placed in a distance so the estimated module count is within the QR specification
135      *  - have similar distance between upper left/right and left top/bottom finder patterns
136      *  - form a triangle with 90° angle (checked by comparing top right/bottom left distance
137      *    with pythagoras)
138      *
139      * Note: we allow each point to be used for more than one code region: this might seem
140      * counterintuitive at first, but the performance penalty is not that big. At this point,
141      * we cannot make a good quality decision whether the three finders actually represent
142      * a QR code, or are just by chance layouted so it looks like there might be a QR code there.
143      * So, if the layout seems right, lets have the decoder try to decode.     
144      */
145 
146      List<FinderPattern[]> results = new ArrayList<>(); // holder for the results
147 
148     for (int i1 = 0; i1 < (size - 2); i1++) {
149       FinderPattern p1 = possibleCenters.get(i1);
150       if (p1 == null) {
151         continue;
152       }
153 
154       for (int i2 = i1 + 1; i2 < (size - 1); i2++) {
155         FinderPattern p2 = possibleCenters.get(i2);
156         if (p2 == null) {
157           continue;
158         }
159 
160         // Compare the expected module sizes; if they are really off, skip
161         float vModSize12 = (p1.getEstimatedModuleSize() - p2.getEstimatedModuleSize()) /
162             Math.min(p1.getEstimatedModuleSize(), p2.getEstimatedModuleSize());
163         float vModSize12A = Math.abs(p1.getEstimatedModuleSize() - p2.getEstimatedModuleSize());
164         if (vModSize12A > DIFF_MODSIZE_CUTOFF && vModSize12 >= DIFF_MODSIZE_CUTOFF_PERCENT) {
165           // break, since elements are ordered by the module size deviation there cannot be
166           // any more interesting elements for the given p1.
167           break;
168         }
169 
170         for (int i3 = i2 + 1; i3 < size; i3++) {
171           FinderPattern p3 = possibleCenters.get(i3);
172           if (p3 == null) {
173             continue;
174           }
175 
176           // Compare the expected module sizes; if they are really off, skip
177           float vModSize23 = (p2.getEstimatedModuleSize() - p3.getEstimatedModuleSize()) /
178               Math.min(p2.getEstimatedModuleSize(), p3.getEstimatedModuleSize());
179           float vModSize23A = Math.abs(p2.getEstimatedModuleSize() - p3.getEstimatedModuleSize());
180           if (vModSize23A > DIFF_MODSIZE_CUTOFF && vModSize23 >= DIFF_MODSIZE_CUTOFF_PERCENT) {
181             // break, since elements are ordered by the module size deviation there cannot be
182             // any more interesting elements for the given p1.
183             break;
184           }
185 
186           FinderPattern[] test = {p1, p2, p3};
187           ResultPoint.orderBestPatterns(test);
188 
189           // Calculate the distances: a = topleft-bottomleft, b=topleft-topright, c = diagonal
190           FinderPatternInfo info = new FinderPatternInfo(test);
191           float dA = ResultPoint.distance(info.getTopLeft(), info.getBottomLeft());
192           float dC = ResultPoint.distance(info.getTopRight(), info.getBottomLeft());
193           float dB = ResultPoint.distance(info.getTopLeft(), info.getTopRight());
194 
195           // Check the sizes
196           float estimatedModuleCount = (dA + dB) / (p1.getEstimatedModuleSize() * 2.0f);
197           if (estimatedModuleCount > MAX_MODULE_COUNT_PER_EDGE ||
198               estimatedModuleCount < MIN_MODULE_COUNT_PER_EDGE) {
199             continue;
200           }
201 
202           // Calculate the difference of the edge lengths in percent
203           float vABBC = Math.abs((dA - dB) / Math.min(dA, dB));
204           if (vABBC >= 0.1f) {
205             continue;
206           }
207 
208           // Calculate the diagonal length by assuming a 90° angle at topleft
209           float dCpy = (float) Math.sqrt(dA * dA + dB * dB);
210           // Compare to the real distance in %
211           float vPyC = Math.abs((dC - dCpy) / Math.min(dC, dCpy));
212 
213           if (vPyC >= 0.1f) {
214             continue;
215           }
216 
217           // All tests passed!
218           results.add(test);
219         } // end iterate p3
220       } // end iterate p2
221     } // end iterate p1
222 
223     if (!results.isEmpty()) {
224       return results.toArray(new FinderPattern[results.size()][]);
225     }
226 
227     // Nothing found!
228     throw NotFoundException.getNotFoundInstance();
229   }

231   public FinderPatternInfo[] findMulti(Map<DecodeHintType,?> hints) throws NotFoundException {
232     boolean tryHarder = hints != null && hints.containsKey(DecodeHintType.TRY_HARDER);
233     boolean pureBarcode = hints != null && hints.containsKey(DecodeHintType.PURE_BARCODE);
234     BitMatrix image = getImage();
235     int maxI = image.getHeight();
236     int maxJ = image.getWidth();
237     // We are looking for black/white/black/white/black modules in
238     // 1:1:3:1:1 ratio; this tracks the number of such modules seen so far
239 
240     // Let's assume that the maximum version QR Code we support takes up 1/4 the height of the
241     // image, and then account for the center being 3 modules in size. This gives the smallest
242     // number of pixels the center could be, so skip this often. When trying harder, look for all
243     // QR versions regardless of how dense they are.
244     int iSkip = (int) (maxI / (MAX_MODULES * 4.0f) * 3);
245     if (iSkip < MIN_SKIP || tryHarder) {
246       iSkip = MIN_SKIP;
247     }
248 
249     int[] stateCount = new int[5];
250     for (int i = iSkip - 1; i < maxI; i += iSkip) {
251       // Get a row of black/white values
252       stateCount[0] = 0;
253       stateCount[1] = 0;
254       stateCount[2] = 0;
255       stateCount[3] = 0;
256       stateCount[4] = 0;
257       int currentState = 0;
258       for (int j = 0; j < maxJ; j++) {
259         if (image.get(j, i)) {
260           // Black pixel
261           if ((currentState & 1) == 1) { // Counting white pixels
262             currentState++;
263           }
264           stateCount[currentState]++;
265         } else { // White pixel
266           if ((currentState & 1) == 0) { // Counting black pixels
267             if (currentState == 4) { // A winner?
268               if (foundPatternCross(stateCount) && handlePossibleCenter(stateCount, i, j, pureBarcode)) { // Yes
269                 // Clear state to start looking again
270                 currentState = 0;
271                 stateCount[0] = 0;
272                 stateCount[1] = 0;
273                 stateCount[2] = 0;
274                 stateCount[3] = 0;
275                 stateCount[4] = 0;
276               } else { // No, shift counts back by two
277                 stateCount[0] = stateCount[2];
278                 stateCount[1] = stateCount[3];
279                 stateCount[2] = stateCount[4];
280                 stateCount[3] = 1;
281                 stateCount[4] = 0;
282                 currentState = 3;
283               }
284             } else {
285               stateCount[++currentState]++;
286             }
287           } else { // Counting white pixels
288             stateCount[currentState]++;
289           }
290         }
291       } // for j=...
292 
293       if (foundPatternCross(stateCount)) {
294         handlePossibleCenter(stateCount, i, maxJ, pureBarcode);
295       } // end if foundPatternCross
296     } // for i=iSkip-1 ...
297     FinderPattern[][] patternInfo = selectMutipleBestPatterns();
298     List<FinderPatternInfo> result = new ArrayList<>();
299     for (FinderPattern[] pattern : patternInfo) {
300       ResultPoint.orderBestPatterns(pattern);
301       result.add(new FinderPatternInfo(pattern));
302     }
303 
304     if (result.isEmpty()) {
305       return EMPTY_RESULT_ARRAY;
306     } else {
307       return result.toArray(new FinderPatternInfo[result.size()]);
308     }
309   }
