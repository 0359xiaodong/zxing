17  package com.google.zxing.multi;
18  
19  import com.google.zxing.BinaryBitmap;
20  import com.google.zxing.DecodeHintType;
21  import com.google.zxing.NotFoundException;
22  import com.google.zxing.Reader;
23  import com.google.zxing.ReaderException;
24  import com.google.zxing.Result;
25  import com.google.zxing.ResultPoint;
26  
27  import java.util.ArrayList;
28  import java.util.List;
29  import java.util.Map;
30  
31  /**
32   * <p>Attempts to locate multiple barcodes in an image by repeatedly decoding portion of the image.
33   * After one barcode is found, the areas left, above, right and below the barcode's
34   * {@link ResultPoint}s are scanned, recursively.</p>
35   *
36   * <p>A caller may want to also employ {@link ByQuadrantReader} when attempting to find multiple
37   * 2D barcodes, like QR Codes, in an image, where the presence of multiple barcodes might prevent
38   * detecting any one of them.</p>
39   *
40   * <p>That is, instead of passing a {@link Reader} a caller might pass
41   * {@code new ByQuadrantReader(reader)}.</p>
42   *
43   * @author Sean Owen
44   */
45  public final class GenericMultipleBarcodeReader implements MultipleBarcodeReader {
46  
47    private static final int MIN_DIMENSION_TO_RECUR = 100;
48    private static final int MAX_DEPTH = 4;
49  
50    private final Reader delegate;
51  
52    public GenericMultipleBarcodeReader(Reader delegate) {
53      this.delegate = delegate;
54    }
55  
56    @Override
57    public Result[] decodeMultiple(BinaryBitmap image) throws NotFoundException {
58      return decodeMultiple(image, null);
59    }
60  
61    @Override
62    public Result[] decodeMultiple(BinaryBitmap image, Map<DecodeHintType,?> hints)
63        throws NotFoundException {
64      List<Result> results = new ArrayList<>();
65      doDecodeMultiple(image, hints, results, 0, 0, 0);
66      if (results.isEmpty()) {
67        throw NotFoundException.getNotFoundInstance();
68      }
69      return results.toArray(new Result[results.size()]);
70    }
71  
72    private void doDecodeMultiple(BinaryBitmap image,
73                                  Map<DecodeHintType,?> hints,
74                                  List<Result> results,
75                                  int xOffset,
76                                  int yOffset,
77                                  int currentDepth) {
78      if (currentDepth > MAX_DEPTH) {
79        return;
80      }
81      
82      Result result;
83      try {
84        result = delegate.decode(image, hints);
85      } catch (ReaderException ignored) {
86        return;
87      }
88      boolean alreadyFound = false;
89      for (Result existingResult : results) {
90        if (existingResult.getText().equals(result.getText())) {
91          alreadyFound = true;
92          break;
93        }
94      }
95      if (!alreadyFound) {
96        results.add(translateResultPoints(result, xOffset, yOffset));
97      }
98      ResultPoint[] resultPoints = result.getResultPoints();
99      if (resultPoints == null || resultPoints.length == 0) {
100       return;
101     }
102     int width = image.getWidth();
103     int height = image.getHeight();
104     float minX = width;
105     float minY = height;
106     float maxX = 0.0f;
107     float maxY = 0.0f;
108     for (ResultPoint point : resultPoints) {
109       float x = point.getX();
110       float y = point.getY();
111       if (x < minX) {
112         minX = x;
113       }
114       if (y < minY) {
115         minY = y;
116       }
117       if (x > maxX) {
118         maxX = x;
119       }
120       if (y > maxY) {
121         maxY = y;
122       }
123     }
124 
125     // Decode left of barcode
126     if (minX > MIN_DIMENSION_TO_RECUR) {
127       doDecodeMultiple(image.crop(0, 0, (int) minX, height),
128                        hints, results, 
129                        xOffset, yOffset, 
130                        currentDepth + 1);
131     }
132     // Decode above barcode
133     if (minY > MIN_DIMENSION_TO_RECUR) {
134       doDecodeMultiple(image.crop(0, 0, width, (int) minY),
135                        hints, results, 
136                        xOffset, yOffset, 
137                        currentDepth + 1);
138     }
139     // Decode right of barcode
140     if (maxX < width - MIN_DIMENSION_TO_RECUR) {
141       doDecodeMultiple(image.crop((int) maxX, 0, width - (int) maxX, height),
142                        hints, results, 
143                        xOffset + (int) maxX, yOffset, 
144                        currentDepth + 1);
145     }
146     // Decode below barcode
147     if (maxY < height - MIN_DIMENSION_TO_RECUR) {
148       doDecodeMultiple(image.crop(0, (int) maxY, width, height - (int) maxY),
149                        hints, results, 
150                        xOffset, yOffset + (int) maxY, 
151                        currentDepth + 1);
152     }
153   }
154 
155   private static Result translateResultPoints(Result result, int xOffset, int yOffset) {
156     ResultPoint[] oldResultPoints = result.getResultPoints();
157     if (oldResultPoints == null) {
158       return result;
159     }
160     ResultPoint[] newResultPoints = new ResultPoint[oldResultPoints.length];
161     for (int i = 0; i < oldResultPoints.length; i++) {
162       ResultPoint oldPoint = oldResultPoints[i];
163       newResultPoints[i] = new ResultPoint(oldPoint.getX() + xOffset, oldPoint.getY() + yOffset);
164     }
165     Result newResult = new Result(result.getText(), result.getRawBytes(), newResultPoints, result.getBarcodeFormat());
166     newResult.putAllMetadata(result.getResultMetadata());
167     return newResult;
168   }
