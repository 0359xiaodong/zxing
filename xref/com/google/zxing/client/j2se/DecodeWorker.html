50  /**
51   * One of a pool of threads which pulls images off the Inputs queue and decodes them in parallel.
52   *
53   * @see CommandLineRunner
54   */
55  final class DecodeWorker implements Callable<Integer> {
56  
57    private static final int RED = 0xFFFF0000;
58    private static final int BLACK = 0xFF000000;
59    private static final int WHITE = 0xFFFFFFFF;
60  
61    private final Config config;
62    private final Queue<Path> inputs;
63  
64    DecodeWorker(Config config, Queue<Path> inputs) {
65      this.config = config;
66      this.inputs = inputs;
67    }
68  
69    @Override
70    public Integer call() throws IOException {
71      int successful = 0;
72      Path input;
73      while ((input = inputs.poll()) != null) {
74        if (Files.exists(input)) {
75          if (config.isMulti()) {
76            Result[] results = decodeMulti(input.toUri(), config.getHints());
77            if (results != null) {
78              successful++;
79              if (config.isDumpResults()) {
80                dumpResultMulti(input, results);
81              }
82            }
83          } else {
84            Result result = decode(input.toUri(), config.getHints());
85            if (result != null) {
86              successful++;
87              if (config.isDumpResults()) {
88                dumpResult(input, result);
89              }
90            }
91          }
92        } else {
93          if (decode(input.toUri(), config.getHints()) != null) {
94            successful++;
95          }
96        }
97      }
98      return successful;
99    }
100 
101   private static void dumpResult(Path input, Result result) throws IOException {
102     String name = input.getFileName().toString();
103     int pos = name.lastIndexOf('.');
104     if (pos > 0) {
105       name = name.substring(0, pos) + ".txt";
106     }
107     Path dumpFile = input.getParent().resolve(name);
108     Files.write(dumpFile, Collections.singleton(result.getText()), StandardCharsets.UTF_8);
109   }
110 
111   private static void dumpResultMulti(Path input, Result[] results) throws IOException {
112     String name = input.getFileName().toString();
113     int pos = name.lastIndexOf('.');
114     if (pos > 0) {
115       name = name.substring(0, pos) + ".txt";
116     }
117     Path dumpFile = input.getParent().resolve(name);
118     Collection<String> resultTexts = new ArrayList<>();
119     for (Result result : results) {
120       resultTexts.add(result.getText());
121     }
122     Files.write(dumpFile, resultTexts, StandardCharsets.UTF_8);
123   }

125   private Result decode(URI uri, Map<DecodeHintType,?> hints) throws IOException {
126     BufferedImage image = ImageReader.readImage(uri);
127     try {
128       LuminanceSource source;
129       if (config.getCrop() == null) {
130         source = new BufferedImageLuminanceSource(image);
131       } else {
132         int[] crop = config.getCrop();
133         source = new BufferedImageLuminanceSource(image, crop[0], crop[1], crop[2], crop[3]);
134       }
135       BinaryBitmap bitmap = new BinaryBitmap(new HybridBinarizer(source));
136       if (config.isDumpBlackPoint()) {
137         dumpBlackPoint(uri, image, bitmap);
138       }
139       Result result = new MultiFormatReader().decode(bitmap, hints);
140       if (config.isBrief()) {
141         System.out.println(uri + ": Success");
142       } else {
143         ParsedResult parsedResult = ResultParser.parseResult(result);
144         System.out.println(uri + " (format: " + result.getBarcodeFormat() + ", type: " +
145             parsedResult.getType() + "):\nRaw result:\n" + result.getText() + "\nParsed result:\n" +
146             parsedResult.getDisplayResult());
147 
148         System.out.println("Found " + result.getResultPoints().length + " result points.");
149         for (int i = 0; i < result.getResultPoints().length; i++) {
150           ResultPoint rp = result.getResultPoints()[i];
151           if (rp != null) {
152             System.out.println("  Point " + i + ": (" + rp.getX() + ',' + rp.getY() + ')');
153           }
154         }
155       }
156 
157       return result;
158     } catch (NotFoundException ignored) {
159       System.out.println(uri + ": No barcode found");
160       return null;
161     }
162   }

164   private Result[] decodeMulti(URI uri, Map<DecodeHintType,?> hints) throws IOException {
165     BufferedImage image = ImageReader.readImage(uri);
166     try {
167       LuminanceSource source;
168       if (config.getCrop() == null) {
169         source = new BufferedImageLuminanceSource(image);
170       } else {
171         int[] crop = config.getCrop();
172         source = new BufferedImageLuminanceSource(image, crop[0], crop[1], crop[2], crop[3]);
173       }
174       BinaryBitmap bitmap = new BinaryBitmap(new HybridBinarizer(source));
175       if (config.isDumpBlackPoint()) {
176         dumpBlackPoint(uri, image, bitmap);
177       }
178 
179       MultiFormatReader multiFormatReader = new MultiFormatReader();
180       MultipleBarcodeReader reader = new GenericMultipleBarcodeReader(multiFormatReader);
181       Result[] results = reader.decodeMultiple(bitmap, hints);
182 
183       if (config.isBrief()) {
184         System.out.println(uri + ": Success");
185       } else {
186         for (Result result : results) {
187           ParsedResult parsedResult = ResultParser.parseResult(result);
188           System.out.println(uri + " (format: "
189               + result.getBarcodeFormat() + ", type: "
190               + parsedResult.getType() + "):\nRaw result:\n"
191               + result.getText() + "\nParsed result:\n"
192               + parsedResult.getDisplayResult());
193           System.out.println("Found " + result.getResultPoints().length + " result points.");
194           for (int i = 0; i < result.getResultPoints().length; i++) {
195             ResultPoint rp = result.getResultPoints()[i];
196             System.out.println("  Point " + i + ": (" + rp.getX() + ',' + rp.getY() + ')');
197           }
198         }
199       }
200       return results;
201     } catch (NotFoundException ignored) {
202       System.out.println(uri + ": No barcode found");
203       return null;
204     }
205   }

207   /**
208    * Writes out a single PNG which is three times the width of the input image, containing from left
209    * to right: the original image, the row sampling monochrome version, and the 2D sampling
210    * monochrome version.
211    */
212   private static void dumpBlackPoint(URI uri, BufferedImage image, BinaryBitmap bitmap) {
213     if (uri.getPath().contains(".mono.png")) {
214       return;
215     }
216 
217     int width = bitmap.getWidth();
218     int height = bitmap.getHeight();
219     int stride = width * 3;
220     int[] pixels = new int[stride * height];
221 
222     // The original image
223     int[] argb = new int[width];
224     for (int y = 0; y < height; y++) {
225       image.getRGB(0, y, width, 1, argb, 0, width);
226       System.arraycopy(argb, 0, pixels, y * stride, width);
227     }
228 
229     // Row sampling
230     BitArray row = new BitArray(width);
231     for (int y = 0; y < height; y++) {
232       try {
233         row = bitmap.getBlackRow(y, row);
234       } catch (NotFoundException nfe) {
235         // If fetching the row failed, draw a red line and keep going.
236         int offset = y * stride + width;
237         Arrays.fill(pixels, offset, offset + width, RED);
238         continue;
239       }
240 
241       int offset = y * stride + width;
242       for (int x = 0; x < width; x++) {
243         pixels[offset + x] = row.get(x) ? BLACK : WHITE;
244       }
245     }
246 
247     // 2D sampling
248     try {
249       for (int y = 0; y < height; y++) {
250         BitMatrix matrix = bitmap.getBlackMatrix();
251         int offset = y * stride + width * 2;
252         for (int x = 0; x < width; x++) {
253           pixels[offset + x] = matrix.get(x, y) ? BLACK : WHITE;
254         }
255       }
256     } catch (NotFoundException ignored) {
257       // continue
258     }
259 
260     writeResultImage(stride, height, pixels, uri, ".mono.png");
261   }
262 
263   private static void writeResultImage(int stride,
264                                        int height,
265                                        int[] pixels,
266                                        URI uri,
267                                        String suffix) {
268     BufferedImage result = new BufferedImage(stride, height, BufferedImage.TYPE_INT_ARGB);
269     result.setRGB(0, 0, stride, height, pixels, 0, stride);
270 
271     // Use the current working directory for URLs
272     String resultName = uri.getPath();
273     if ("http".equals(uri.getScheme())) {
274       int pos = resultName.lastIndexOf('/');
275       if (pos > 0) {
276         resultName = '.' + resultName.substring(pos);
277       }
278     }
279     int pos = resultName.lastIndexOf('.');
280     if (pos > 0) {
281       resultName = resultName.substring(0, pos);
282     }
283     resultName += suffix;
284     try {
285       if (!ImageIO.write(result, "png", Paths.get(resultName).toFile())) {
286         System.err.println("Could not encode an image to " + resultName);
287       }
288     } catch (IOException ignored) {
289       System.err.println("Could not write to " + resultName);
290     }
291   }
