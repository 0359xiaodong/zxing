27  package com.google.zxing.client.result;

29  import java.util.HashMap;
30  import java.util.Map;

32  import com.google.zxing.BarcodeFormat;
33  import com.google.zxing.Result;

35  /**
36   * Parses strings of digits that represent a RSS Extended code.
37   * 
38   * @author Antonio Manuel Benjumea Conde, Servinform, S.A.
39   * @author Agust√≠n Delgado, Servinform, S.A.
40   */
41  public final class ExpandedProductResultParser extends ResultParser {
42  
43    @Override
44    public ExpandedProductParsedResult parse(Result result) {
45      BarcodeFormat format = result.getBarcodeFormat();
46      if (format != BarcodeFormat.RSS_EXPANDED) {
47        // ExtendedProductParsedResult NOT created. Not a RSS Expanded barcode
48        return null;
49      }
50      String rawText = getMassagedText(result);
51  
52      String productID = null;
53      String sscc = null;
54      String lotNumber = null;
55      String productionDate = null;
56      String packagingDate = null;
57      String bestBeforeDate = null;
58      String expirationDate = null;
59      String weight = null;
60      String weightType = null;
61      String weightIncrement = null;
62      String price = null;
63      String priceIncrement = null;
64      String priceCurrency = null;
65      Map<String,String> uncommonAIs = new HashMap<>();
66  
67      int i = 0;
68  
69      while (i < rawText.length()) {
70        String ai = findAIvalue(i, rawText);
71        if (ai == null) {
72          // Error. Code doesn't match with RSS expanded pattern
73          // ExtendedProductParsedResult NOT created. Not match with RSS Expanded pattern
74          return null;
75        }
76        i += ai.length() + 2;
77        String value = findValue(i, rawText);
78        i += value.length();
79  
80        switch (ai) {
81          case "00":
82            sscc = value;
83            break;
84          case "01":
85            productID = value;
86            break;
87          case "10":
88            lotNumber = value;
89            break;
90          case "11":
91            productionDate = value;
92            break;
93          case "13":
94            packagingDate = value;
95            break;
96          case "15":
97            bestBeforeDate = value;
98            break;
99          case "17":
100           expirationDate = value;
101           break;
102         case "3100":
103         case "3101":
104         case "3102":
105         case "3103":
106         case "3104":
107         case "3105":
108         case "3106":
109         case "3107":
110         case "3108":
111         case "3109":
112           weight = value;
113           weightType = ExpandedProductParsedResult.KILOGRAM;
114           weightIncrement = ai.substring(3);
115           break;
116         case "3200":
117         case "3201":
118         case "3202":
119         case "3203":
120         case "3204":
121         case "3205":
122         case "3206":
123         case "3207":
124         case "3208":
125         case "3209":
126           weight = value;
127           weightType = ExpandedProductParsedResult.POUND;
128           weightIncrement = ai.substring(3);
129           break;
130         case "3920":
131         case "3921":
132         case "3922":
133         case "3923":
134           price = value;
135           priceIncrement = ai.substring(3);
136           break;
137         case "3930":
138         case "3931":
139         case "3932":
140         case "3933":
141           if (value.length() < 4) {
142             // The value must have more of 3 symbols (3 for currency and
143             // 1 at least for the price)
144             // ExtendedProductParsedResult NOT created. Not match with RSS Expanded pattern
145             return null;
146           }
147           price = value.substring(3);
148           priceCurrency = value.substring(0, 3);
149           priceIncrement = ai.substring(3);
150           break;
151         default:
152           // No match with common AIs
153           uncommonAIs.put(ai, value);
154           break;
155       }
156     }
157 
158     return new ExpandedProductParsedResult(rawText,
159                                            productID,
160                                            sscc,
161                                            lotNumber,
162                                            productionDate,
163                                            packagingDate,
164                                            bestBeforeDate,
165                                            expirationDate,
166                                            weight,
167                                            weightType,
168                                            weightIncrement,
169                                            price,
170                                            priceIncrement,
171                                            priceCurrency,
172                                            uncommonAIs);
173   }
174 
175   private static String findAIvalue(int i, String rawText) {
176     char c = rawText.charAt(i);
177     // First character must be a open parenthesis.If not, ERROR
178     if (c != '(') {
179       return null;
180     }
181 
182     CharSequence rawTextAux = rawText.substring(i + 1);
183 
184     StringBuilder buf = new StringBuilder();
185     for (int index = 0; index < rawTextAux.length(); index++) {
186       char currentChar = rawTextAux.charAt(index);
187       if (currentChar == ')') {
188         return buf.toString();
189       } else if (currentChar >= '0' && currentChar <= '9') {
190         buf.append(currentChar);
191       } else {
192         return null;
193       }
194     }
195     return buf.toString();
196   }
197 
198   private static String findValue(int i, String rawText) {
199     StringBuilder buf = new StringBuilder();
200     String rawTextAux = rawText.substring(i);
201 
202     for (int index = 0; index < rawTextAux.length(); index++) {
203       char c = rawTextAux.charAt(index);
204       if (c == '(') {
205         // We look for a new AI. If it doesn't exist (ERROR), we coninue
206         // with the iteration
207         if (findAIvalue(index, rawTextAux) == null) {
208           buf.append('(');
209         } else {
210           break;
211         }
212       } else {
213         buf.append(c);
214       }
215     }
216     return buf.toString();
217   }
