40    private final String rawText;
41    private final String productID;
42    private final String sscc;
43    private final String lotNumber;

45    private final String packagingDate;

53    private final String priceCurrency;
54    // For AIS that not exist in this object
55    private final Map<String,String> uncommonAIs;

57    public ExpandedProductParsedResult(String rawText,
58                                       String productID,
59                                       String sscc,
60                                       String lotNumber,
61                                       String productionDate,
62                                       String packagingDate,
63                                       String bestBeforeDate,
64                                       String expirationDate,
65                                       String weight,
66                                       String weightType,
67                                       String weightIncrement,
68                                       String price,
69                                       String priceIncrement,
70                                       String priceCurrency,
71                                       Map<String,String> uncommonAIs) {
72      super(ParsedResultType.PRODUCT);
73      this.rawText = rawText;
74      this.productID = productID;
75      this.sscc = sscc;
76      this.lotNumber = lotNumber;
77      this.productionDate = productionDate;
78      this.packagingDate = packagingDate;
79      this.bestBeforeDate = bestBeforeDate;
80      this.expirationDate = expirationDate;
81      this.weight = weight;
82      this.weightType = weightType;
83      this.weightIncrement = weightIncrement;
84      this.price = price;
85      this.priceIncrement = priceIncrement;
86      this.priceCurrency = priceCurrency;
87      this.uncommonAIs = uncommonAIs;
88    }
89  
90    @Override
91    public boolean equals(Object o){
92      if (!(o instanceof ExpandedProductParsedResult)) {
93        return false;
94      }
95  
96      ExpandedProductParsedResult other = (ExpandedProductParsedResult)o;
97  
98      return equalsOrNull(productID, other.productID)
99          && equalsOrNull(sscc, other.sscc)
100         && equalsOrNull(lotNumber, other.lotNumber)
101         && equalsOrNull(productionDate, other.productionDate)
102         && equalsOrNull(bestBeforeDate, other.bestBeforeDate)
103         && equalsOrNull(expirationDate, other.expirationDate)
104         && equalsOrNull(weight, other.weight)
105         && equalsOrNull(weightType, other.weightType)
106         && equalsOrNull(weightIncrement, other.weightIncrement)
107         && equalsOrNull(price, other.price)
108         && equalsOrNull(priceIncrement, other.priceIncrement)
109         && equalsOrNull(priceCurrency, other.priceCurrency)
110         && equalsOrNull(uncommonAIs, other.uncommonAIs);
111   }
112 
113   private static boolean equalsOrNull(Object o1, Object o2) {
114     return o1 == null ? o2 == null : o1.equals(o2);
115   }
116 
117   @Override
118   public int hashCode(){
119     int hash = 0;
120     hash ^= hashNotNull(productID);
121     hash ^= hashNotNull(sscc);
122     hash ^= hashNotNull(lotNumber);
123     hash ^= hashNotNull(productionDate);
124     hash ^= hashNotNull(bestBeforeDate);
125     hash ^= hashNotNull(expirationDate);
126     hash ^= hashNotNull(weight);
127     hash ^= hashNotNull(weightType);
128     hash ^= hashNotNull(weightIncrement);
129     hash ^= hashNotNull(price);
130     hash ^= hashNotNull(priceIncrement);
131     hash ^= hashNotNull(priceCurrency);
132     hash ^= hashNotNull(uncommonAIs);
133     return hash;
134   }
135 
136   private static int hashNotNull(Object o) {
137     return o == null ? 0 : o.hashCode();
138   }
