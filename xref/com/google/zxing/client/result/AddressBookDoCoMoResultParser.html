17  package com.google.zxing.client.result;
18  
19  import com.google.zxing.Result;
20  
21  /**
22   * Implements the "MECARD" address book entry format.
23   *
24   * Supported keys: N, SOUND, TEL, EMAIL, NOTE, ADR, BDAY, URL, plus ORG
25   * Unsupported keys: TEL-AV, NICKNAME
26   *
27   * Except for TEL, multiple values for keys are also not supported;
28   * the first one found takes precedence.
29   *
30   * Our understanding of the MECARD format is based on this document:
31   *
32   * http://www.mobicode.org.tw/files/OMIA%20Mobile%20Bar%20Code%20Standard%20v3.2.1.doc 
33   *
34   * @author Sean Owen
35   */
36  public final class AddressBookDoCoMoResultParser extends AbstractDoCoMoResultParser {
37  
38    @Override
39    public AddressBookParsedResult parse(Result result) {
40      String rawText = getMassagedText(result);
41      if (!rawText.startsWith("MECARD:")) {
42        return null;
43      }
44      String[] rawName = matchDoCoMoPrefixedField("N:", rawText, true);
45      if (rawName == null) {
46        return null;
47      }
48      String name = parseName(rawName[0]);
49      String pronunciation = matchSingleDoCoMoPrefixedField("SOUND:", rawText, true);
50      String[] phoneNumbers = matchDoCoMoPrefixedField("TEL:", rawText, true);
51      String[] emails = matchDoCoMoPrefixedField("EMAIL:", rawText, true);
52      String note = matchSingleDoCoMoPrefixedField("NOTE:", rawText, false);
53      String[] addresses = matchDoCoMoPrefixedField("ADR:", rawText, true);
54      String birthday = matchSingleDoCoMoPrefixedField("BDAY:", rawText, true);
55      if (!isStringOfDigits(birthday, 8)) {
56        // No reason to throw out the whole card because the birthday is formatted wrong.
57        birthday = null;
58      }
59      String[] urls = matchDoCoMoPrefixedField("URL:", rawText, true);
60  
61      // Although ORG may not be strictly legal in MECARD, it does exist in VCARD and we might as well
62      // honor it when found in the wild.
63      String org = matchSingleDoCoMoPrefixedField("ORG:", rawText, true);
64  
65      return new AddressBookParsedResult(maybeWrap(name),
66                                         null,
67                                         pronunciation,
68                                         phoneNumbers,
69                                         null,
70                                         emails,
71                                         null,
72                                         null,
73                                         note,
74                                         addresses,
75                                         null,
76                                         org,
77                                         birthday,
78                                         null,
79                                         urls,
80                                         null);
81    }
82  
83    private static String parseName(String name) {
84      int comma = name.indexOf((int) ',');
85      if (comma >= 0) {
86        // Format may be last,first; switch it around
87        return name.substring(comma + 1) + ' ' + name.substring(0, comma);
88      }
89      return name;
90    }
91  
92  }
