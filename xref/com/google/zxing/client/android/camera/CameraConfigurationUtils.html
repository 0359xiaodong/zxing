58    public static void setFocus(Camera.Parameters parameters,
59                                boolean autoFocus,
60                                boolean disableContinuous,
61                                boolean safeMode) {
62      List<String> supportedFocusModes = parameters.getSupportedFocusModes();
63      String focusMode = null;
64      if (autoFocus) {
65        if (safeMode || disableContinuous) {
66          focusMode = findSettableValue("focus mode",
67                                         supportedFocusModes,
68                                         Camera.Parameters.FOCUS_MODE_AUTO);
69        } else {
70          focusMode = findSettableValue("focus mode",
71                                        supportedFocusModes,
72                                        Camera.Parameters.FOCUS_MODE_CONTINUOUS_PICTURE,
73                                        Camera.Parameters.FOCUS_MODE_CONTINUOUS_VIDEO,
74                                        Camera.Parameters.FOCUS_MODE_AUTO);
75        }
76      }
77      // Maybe selected auto-focus but not available, so fall through here:
78      if (!safeMode && focusMode == null) {
79        focusMode = findSettableValue("focus mode",
80                                      supportedFocusModes,
81                                      Camera.Parameters.FOCUS_MODE_MACRO,
82                                      Camera.Parameters.FOCUS_MODE_EDOF);
83      }
84      if (focusMode != null) {
85        if (focusMode.equals(parameters.getFocusMode())) {
86          Log.i(TAG, "Focus mode already set to " + focusMode);
87        } else {
88          parameters.setFocusMode(focusMode);
89        }
90      }
91    }

93    public static void setTorch(Camera.Parameters parameters, boolean on) {
94      List<String> supportedFlashModes = parameters.getSupportedFlashModes();
95      String flashMode;
96      if (on) {
97        flashMode = findSettableValue("flash mode",
98                                      supportedFlashModes,
99                                      Camera.Parameters.FLASH_MODE_TORCH,

101     } else {
102       flashMode = findSettableValue("flash mode",
103                                     supportedFlashModes,
104                                     Camera.Parameters.FLASH_MODE_OFF);
105     }
106     if (flashMode != null) {
107       if (flashMode.equals(parameters.getFlashMode())) {
108         Log.i(TAG, "Flash mode already set to " + flashMode);
109       } else {
110         Log.i(TAG, "Setting flash mode to " + flashMode);
111         parameters.setFlashMode(flashMode);
112       }
113     }

116   public static void setBestExposure(Camera.Parameters parameters, boolean lightOn) {
117     int minExposure = parameters.getMinExposureCompensation();
118     int maxExposure = parameters.getMaxExposureCompensation();
119     float step = parameters.getExposureCompensationStep();
120     if ((minExposure != 0 || maxExposure != 0) && step > 0.0f) {
121       // Set low when light is on
122       float targetCompensation = lightOn ? MIN_EXPOSURE_COMPENSATION : MAX_EXPOSURE_COMPENSATION;
123       int compensationSteps = Math.round(targetCompensation / step);
124       float actualCompensation = step * compensationSteps;
125       // Clamp value:
126       compensationSteps = Math.max(Math.min(compensationSteps, maxExposure), minExposure);
127       if (parameters.getExposureCompensation() == compensationSteps) {
128         Log.i(TAG, "Exposure compensation already set to " + compensationSteps + " / " + actualCompensation);
129       } else {
130         Log.i(TAG, "Setting exposure compensation to " + compensationSteps + " / " + actualCompensation);
131         parameters.setExposureCompensation(compensationSteps);
132       }
133     } else {
134       Log.i(TAG, "Camera does not support exposure compensation");
135     }
136   }

142   public static void setBestPreviewFPS(Camera.Parameters parameters, int minFPS, int maxFPS) {
143     List<int[]> supportedPreviewFpsRanges = parameters.getSupportedPreviewFpsRange();
144     Log.i(TAG, "Supported FPS ranges: " + toString(supportedPreviewFpsRanges));
145     if (supportedPreviewFpsRanges != null && !supportedPreviewFpsRanges.isEmpty()) {
146       int[] suitableFPSRange = null;
147       for (int[] fpsRange : supportedPreviewFpsRanges) {
148         int thisMin = fpsRange[Camera.Parameters.PREVIEW_FPS_MIN_INDEX];
149         int thisMax = fpsRange[Camera.Parameters.PREVIEW_FPS_MAX_INDEX];
150         if (thisMin >= minFPS * 1000 && thisMax <= maxFPS * 1000) {
151           suitableFPSRange = fpsRange;
152           break;
153         }
154       }
155       if (suitableFPSRange == null) {
156         Log.i(TAG, "No suitable FPS range?");
157       } else {
158         int[] currentFpsRange = new int[2];
159         parameters.getPreviewFpsRange(currentFpsRange);
160         if (Arrays.equals(currentFpsRange, suitableFPSRange)) {
161           Log.i(TAG, "FPS range already set to " + Arrays.toString(suitableFPSRange));
162         } else {
163           Log.i(TAG, "Setting FPS range to " + Arrays.toString(suitableFPSRange));
164           parameters.setPreviewFpsRange(suitableFPSRange[Camera.Parameters.PREVIEW_FPS_MIN_INDEX],
165                                         suitableFPSRange[Camera.Parameters.PREVIEW_FPS_MAX_INDEX]);
166         }
167       }
168     }
169   }

171   public static void setFocusArea(Camera.Parameters parameters) {
172     if (parameters.getMaxNumFocusAreas() > 0) {
173       Log.i(TAG, "Old focus areas: " + toString(parameters.getFocusAreas()));

176       parameters.setFocusAreas(middleArea);
177     } else {
178       Log.i(TAG, "Device does not support focus areas");
179     }

224   public static void setZoom(Camera.Parameters parameters, double targetZoomRatio) {
225     if (parameters.isZoomSupported()) {
226       Integer zoom = indexOfClosestZoom(parameters, targetZoomRatio);
227       if (zoom == null) {
228         return;
229       }
230       if (parameters.getZoom() == zoom) {
231         Log.i(TAG, "Zoom is already set to " + zoom);
232       } else {
233         Log.i(TAG, "Setting zoom to " + zoom);
234         parameters.setZoom(zoom);
235       }
236     } else {
237       Log.i(TAG, "Zoom is not supported");
238     }
239   }

241   private static Integer indexOfClosestZoom(Camera.Parameters parameters, double targetZoomRatio) {
242     List<Integer> ratios = parameters.getZoomRatios();
243     Log.i(TAG, "Zoom ratios: " + ratios);
244     int maxZoom = parameters.getMaxZoom();
245     if (ratios == null || ratios.isEmpty() || ratios.size() != maxZoom + 1) {
246       Log.w(TAG, "Invalid zoom ratios!");
247       return null;
248     }
249     double target100 = 100.0 * targetZoomRatio;
250     double smallestDiff = Double.POSITIVE_INFINITY;
251     int closestIndex = 0;
252     for (int i = 0; i < ratios.size(); i++) {
253       double diff = Math.abs(ratios.get(i) - target100);
254       if (diff < smallestDiff) {
255         smallestDiff = diff;
256         closestIndex = i;
257       }
258     }
259     Log.i(TAG, "Chose zoom ratio of " + (ratios.get(closestIndex) / 100.0));
260     return closestIndex;
261   }

276   public static Point findBestPreviewSizeValue(Camera.Parameters parameters, Point screenResolution) {
277 
278     List<Camera.Size> rawSupportedSizes = parameters.getSupportedPreviewSizes();
279     if (rawSupportedSizes == null) {
280       Log.w(TAG, "Device returned no supported preview sizes; using default");
281       Camera.Size defaultSize = parameters.getPreviewSize();
282       if (defaultSize == null) {
283         throw new IllegalStateException("Parameters contained no preview size!");
284       }
285       return new Point(defaultSize.width, defaultSize.height);
286     }
287 
288     // Sort by size, descending
289     List<Camera.Size> supportedPreviewSizes = new ArrayList<>(rawSupportedSizes);
290     Collections.sort(supportedPreviewSizes, new Comparator<Camera.Size>() {
291       @Override
292       public int compare(Camera.Size a, Camera.Size b) {
293         int aPixels = a.height * a.width;
294         int bPixels = b.height * b.width;
295         if (bPixels < aPixels) {
296           return -1;
297         }
298         if (bPixels > aPixels) {
299           return 1;
300         }
301         return 0;
302       }
303     });
304 
305     if (Log.isLoggable(TAG, Log.INFO)) {
306       StringBuilder previewSizesString = new StringBuilder();
307       for (Camera.Size supportedPreviewSize : supportedPreviewSizes) {
308         previewSizesString.append(supportedPreviewSize.width).append('x')
309             .append(supportedPreviewSize.height).append(' ');
310       }
311       Log.i(TAG, "Supported preview sizes: " + previewSizesString);
312     }
313 
314     double screenAspectRatio = (double) screenResolution.x / (double) screenResolution.y;
315 
316     // Remove sizes that are unsuitable
317     Iterator<Camera.Size> it = supportedPreviewSizes.iterator();
318     while (it.hasNext()) {
319       Camera.Size supportedPreviewSize = it.next();
320       int realWidth = supportedPreviewSize.width;
321       int realHeight = supportedPreviewSize.height;
322       if (realWidth * realHeight < MIN_PREVIEW_PIXELS) {
323         it.remove();
324         continue;
325       }
326 
327       boolean isCandidatePortrait = realWidth < realHeight;
328       int maybeFlippedWidth = isCandidatePortrait ? realHeight : realWidth;
329       int maybeFlippedHeight = isCandidatePortrait ? realWidth : realHeight;
330       double aspectRatio = (double) maybeFlippedWidth / (double) maybeFlippedHeight;
331       double distortion = Math.abs(aspectRatio - screenAspectRatio);
332       if (distortion > MAX_ASPECT_DISTORTION) {
333         it.remove();
334         continue;
335       }
336 
337       if (maybeFlippedWidth == screenResolution.x && maybeFlippedHeight == screenResolution.y) {
338         Point exactPoint = new Point(realWidth, realHeight);
339         Log.i(TAG, "Found preview size exactly matching screen size: " + exactPoint);
340         return exactPoint;
341       }
342     }
343 
344     // If no exact match, use largest preview size. This was not a great idea on older devices because
345     // of the additional computation needed. We're likely to get here on newer Android 4+ devices, where
346     // the CPU is much more powerful.
347     if (!supportedPreviewSizes.isEmpty()) {
348       Camera.Size largestPreview = supportedPreviewSizes.get(0);
349       Point largestSize = new Point(largestPreview.width, largestPreview.height);
350       Log.i(TAG, "Using largest suitable preview size: " + largestSize);
351       return largestSize;
352     }
353 
354     // If there is nothing at all suitable, return current preview size
355     Camera.Size defaultPreview = parameters.getPreviewSize();
356     if (defaultPreview == null) {
357       throw new IllegalStateException("Parameters contained no preview size!");
358     }
359     Point defaultSize = new Point(defaultPreview.width, defaultPreview.height);
360     Log.i(TAG, "No suitable preview sizes, using default: " + defaultSize);
361     return defaultSize;
362   }

364   private static String findSettableValue(String name,
365                                           Collection<String> supportedValues,
366                                           String... desiredValues) {
367     Log.i(TAG, "Requesting " + name + " value from among: " + Arrays.toString(desiredValues));
368     Log.i(TAG, "Supported " + name + " values: " + supportedValues);
369     if (supportedValues != null) {
370       for (String desiredValue : desiredValues) {
371         if (supportedValues.contains(desiredValue)) {
372           Log.i(TAG, "Can set " + name + " to: " + desiredValue);
373           return desiredValue;
374         }
375       }
376     }
377     Log.i(TAG, "No supported values match");
378     return null;
379   }

381   private static String toString(Collection<int[]> arrays) {
382     if (arrays == null || arrays.isEmpty()) {
383       return "[]";
384     }
385     StringBuilder buffer = new StringBuilder();
386     buffer.append('[');
387     Iterator<int[]> it = arrays.iterator();
388     while (it.hasNext()) {
389       buffer.append(Arrays.toString(it.next()));
390       if (it.hasNext()) {
391         buffer.append(", ");
392       }
393     }
394     buffer.append(']');
395     return buffer.toString();
396   }

398   private static String toString(Iterable<Camera.Area> areas) {
399     if (areas == null) {
400       return null;
401     }
402     StringBuilder result = new StringBuilder();
403     for (Camera.Area area : areas) {
404       result.append(area.rect).append(':').append(area.weight).append(' ');
405     }
406     return result.toString();
407   }

413   public static String collectStats(CharSequence flattenedParams) {
414     StringBuilder result = new StringBuilder(1000);
415 
416     result.append("BOARD=").append(Build.BOARD).append('\n');
417     result.append("BRAND=").append(Build.BRAND).append('\n');
418     result.append("CPU_ABI=").append(Build.CPU_ABI).append('\n');
419     result.append("DEVICE=").append(Build.DEVICE).append('\n');
420     result.append("DISPLAY=").append(Build.DISPLAY).append('\n');
421     result.append("FINGERPRINT=").append(Build.FINGERPRINT).append('\n');
422     result.append("HOST=").append(Build.HOST).append('\n');
423     result.append("ID=").append(Build.ID).append('\n');
424     result.append("MANUFACTURER=").append(Build.MANUFACTURER).append('\n');
425     result.append("MODEL=").append(Build.MODEL).append('\n');
426     result.append("PRODUCT=").append(Build.PRODUCT).append('\n');
427     result.append("TAGS=").append(Build.TAGS).append('\n');
428     result.append("TIME=").append(Build.TIME).append('\n');
429     result.append("TYPE=").append(Build.TYPE).append('\n');
430     result.append("USER=").append(Build.USER).append('\n');
431     result.append("VERSION.CODENAME=").append(Build.VERSION.CODENAME).append('\n');
432     result.append("VERSION.INCREMENTAL=").append(Build.VERSION.INCREMENTAL).append('\n');
433     result.append("VERSION.RELEASE=").append(Build.VERSION.RELEASE).append('\n');
434     result.append("VERSION.SDK_INT=").append(Build.VERSION.SDK_INT).append('\n');
435 
436     if (flattenedParams != null) {
437       String[] params = SEMICOLON.split(flattenedParams);
438       Arrays.sort(params);
439       for (String param : params) {
440         result.append(param).append('\n');
441       }
442     }
443 
444     return result.toString();
445   }
