58  /**
59   * This servlet manages the OAuth 2.0 dance.
60   * 
61   * @author Jenny Murphy - http://google.com/+JennyMurphy
62   * @author Sean Owen
63   * @link https://developers.google.com/glass/develop/mirror/authorization
64   */
65  public final class NewUserSetupServlet extends HttpServlet {
66    
67    private static final Logger log = LoggerFactory.getLogger(NewUserSetupServlet.class);
68    
69    private static final String BS_ID = "barcode-scanner";
70  
71    private AuthUtil authUtil;
72    private String baseURL;
73    private String notificationCallbackURL;
74    private String subscriptionVerifyToken;
75  
76    @Override
77    public void init(ServletConfig servletConfig) {
78      ServletContext context = servletConfig.getServletContext();
79      authUtil = new AuthUtil(context.getInitParameter("CLIENT_ID"), context.getInitParameter("CLIENT_SECRET"));
80      baseURL = context.getInitParameter("BASE_URL");
81      notificationCallbackURL = baseURL + "/notification";
82      subscriptionVerifyToken = context.getInitParameter("SUBSCRIPTION_VERIFY_TOKEN");
83    }
84  
85    @Override
86    public void doGet(HttpServletRequest request, HttpServletResponse response) throws IOException {
87      
88      String redirect;    
89      
90      if (request.getParameter("error") != null) {
91        
92        log.warn("OAuth error: {}", Arrays.toString(request.getParameterValues("error")));
93        redirect = baseURL;
94        
95      } else {
96  
97        GenericUrl url1 = new GenericUrl(request.getRequestURL().toString());
98        url1.setRawPath("/oauth2callback");
99        String callbackURL = url1.build();
100       String code = request.getParameter("code");
101       
102       if (code == null) {
103         
104         log.info("New OAuth flow");
105     
106         AuthorizationCodeFlow flow = authUtil.newAuthorizationCodeFlow();
107         GenericUrl url = flow.newAuthorizationUrl().setRedirectUri(callbackURL);
108         url.set("approval_prompt", "force");
109         redirect = url.build();
110         
111       } else {    
112         // If we have a code, finish the OAuth 2.0 dance
113         
114         log.info("OAuth response, code: {}", code);
115   
116         AuthorizationCodeFlow flow = authUtil.newAuthorizationCodeFlow();
117         TokenResponse tokenResponse = flow.newTokenRequest(code).setRedirectUri(callbackURL).execute();
118   
119         // Extract the Google User ID from the ID token in the auth response
120         String userId = ((GoogleTokenResponse) tokenResponse).parseIdToken().getPayload().getSubject();
121   
122         // Set it into the session
123         //AuthUtil.setUserId(request, userId);
124         flow.createAndStoreCredential(tokenResponse, userId);
125   
126         ResourceBundle resources = ResourceBundle.getBundle("Strings", request.getLocale());
127         
128         bootstrapNewUser(userId, resources);
129   
130         redirect = "glass/index.jspx";
131       }
132     }
133     
134     response.sendRedirect(redirect);
135   }
136   
137   private void bootstrapNewUser(String userId, ResourceBundle resources) throws IOException {
138     Credential credential = authUtil.newAuthorizationCodeFlow().loadCredential(userId);
139 
140     String appName = resources.getString("app.title");
141     
142     MirrorClient mirrorClient = MirrorClient.get();
143     
144     boolean contactExists = false;
145     for (Contact contact : mirrorClient.listContacts(credential)) {
146       if (BS_ID.equals(contact.getId())) {
147         contactExists = true;
148         break;
149       }
150     }
151     
152     if (!contactExists) {
153       Contact contact = new Contact();
154       contact.setId(BS_ID);
155       contact.setDisplayName(appName);
156       contact.setAcceptTypes(Arrays.asList("image/png", "image/jpeg", "image/gif", "image/bmp"));
157       mirrorClient.insertContact(credential, contact);
158     }
159 
160     boolean subscriptionExists = false;
161     for (Subscription subscription : mirrorClient.listSubscriptions(credential)) {
162       if (notificationCallbackURL.equals(subscription.getCallbackUrl())) {
163         subscriptionExists = true;
164         break;
165       }
166     }
167 
168     if (!subscriptionExists) {
169       Subscription subscription = new Subscription();
170       subscription.setCollection("timeline");
171       subscription.setOperation(Collections.singletonList("INSERT"));
172       subscription.setCallbackUrl(notificationCallbackURL);
173       subscription.setVerifyToken(subscriptionVerifyToken);
174       subscription.setUserToken(userId);
175       mirrorClient.insertSubscription(credential, subscription);
176     }
177     
178     TimelineItem timelineItem = new TimelineItem();
179     timelineItem.setTitle(MessageFormat.format(resources.getString("setup.welcome.title"), appName));
180     timelineItem.setText(MessageFormat.format(resources.getString("setup.welcome.text"), appName));
181     mirrorClient.insertTimelineItem(credential, timelineItem);
182     
183   }
